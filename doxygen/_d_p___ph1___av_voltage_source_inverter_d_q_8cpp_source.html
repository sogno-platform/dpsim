13DP::Ph1::AvVoltageSourceInverterDQ::AvVoltageSourceInverterDQ(
14 String uid, String name, Logger::Level logLevel, Bool withTrafo)
15 : CompositePowerComp<Complex>(uid, name, true, true, logLevel),
16 mOmegaN(mAttributes->create<Real>("Omega_nom")),
17 mPref(mAttributes->create<Real>("P_ref")),
18 mQref(mAttributes->create<Real>("Q_ref")),
19 mVcd(mAttributes->create<Real>("Vc_d", 0)),
20 mVcq(mAttributes->create<Real>("Vc_q", 0)),
21 mIrcd(mAttributes->create<Real>("Irc_d", 0)),
22 mIrcq(mAttributes->create<Real>("Irc_q", 0)),
23 mVsref(mAttributes->create<MatrixComp>("Vsref", MatrixComp::Zero(1, 1))),
24 mVs(mAttributes->createDynamic<MatrixComp>("Vs")),
25 mPllOutput(mAttributes->createDynamic<Matrix>("pll_output")),
26 mPowerctrlInputs(mAttributes->createDynamic<Matrix>("powerctrl_inputs")),
27 mPowerctrlOutputs(
28 mAttributes->createDynamic<Matrix>("powerctrl_outputs")),
29 mPowerctrlStates(mAttributes->createDynamic<Matrix>("powerctrl_states")) {
30
31 if (withTrafo) {
32 setVirtualNodeNumber(4);
33 mConnectionTransformer = DP::Ph1::Transformer::make(
34 **mName + "_trans", **mName + "_trans", mLogLevel, false);
35 addMNASubComponent(mConnectionTransformer,
36 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
37 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
38 } else {
39 setVirtualNodeNumber(3);
40 }
41 mWithConnectionTransformer = withTrafo;
42 setTerminalNumber(1);
43
44 SPDLOG_LOGGER_INFO(mSLog, "Create {} {}", type(), name);
45 **mIntfVoltage = MatrixComp::Zero(1, 1);
46 **mIntfCurrent = MatrixComp::Zero(1, 1);
47
48 // Create electrical sub components
49 mSubResistorF = DP::Ph1::Resistor::make(**mName + "_resF", mLogLevel);
50 mSubResistorC = DP::Ph1::Resistor::make(**mName + "_resC", mLogLevel);
51 mSubCapacitorF = DP::Ph1::Capacitor::make(**mName + "_capF", mLogLevel);
52 mSubInductorF = DP::Ph1::Inductor::make(**mName + "_indF", mLogLevel);
53 mSubCtrledVoltageSource =
54 DP::Ph1::VoltageSource::make(**mName + "_src", mLogLevel);
55 addMNASubComponent(mSubResistorF, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
56 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, false);
57 addMNASubComponent(mSubResistorC, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
58 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, false);
59 addMNASubComponent(mSubCapacitorF, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
60 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
61 addMNASubComponent(mSubInductorF, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
62 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
63
64 // Pre-step of the subcontrolled voltage source is handled explicitly in mnaParentPreStep
65 addMNASubComponent(mSubCtrledVoltageSource, MNA_SUBCOMP_TASK_ORDER::NO_TASK,
66 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
67
68 SPDLOG_LOGGER_INFO(mSLog, "Electrical subcomponents: ");
69 for (auto subcomp : mSubComponents)
70 SPDLOG_LOGGER_INFO(mSLog, "- {}", subcomp->name());
71
72 // Create control sub components
73 mPLL = Signal::PLL::make(**mName + "_PLL", mLogLevel);
74 mPowerControllerVSI = Signal::PowerControllerVSI::make(
75 **mName + "_PowerControllerVSI", mLogLevel);
76
77 // Sub voltage source
78 mVs->setReference(mSubCtrledVoltageSource->mIntfVoltage);
79
80 // PLL
81 mPLL->mInputRef->setReference(mVcq);
82 mPllOutput->setReference(mPLL->mOutputCurr);
83
84 // Power controller
85 // input references
86 mPowerControllerVSI->mVc_d->setReference(mVcd);
87 mPowerControllerVSI->mVc_q->setReference(mVcq);
88 mPowerControllerVSI->mIrc_d->setReference(mIrcd);
89 mPowerControllerVSI->mIrc_q->setReference(mIrcq);
90 // input, state and output vector for logging
91 mPowerctrlInputs->setReference(mPowerControllerVSI->mInputCurr);
92 mPowerctrlStates->setReference(mPowerControllerVSI->mStateCurr);
93 mPowerctrlOutputs->setReference(mPowerControllerVSI->mOutputCurr);
94}

193void DP::Ph1::AvVoltageSourceInverterDQ::initializeFromNodesAndTerminals(
194 Real frequency) {
195
196 // set initial interface quantities
197 (**mIntfVoltage)(0, 0) = initialSingleVoltage(0);
198 (**mIntfCurrent)(0, 0) =
199 -std::conj(Complex(**mPref, **mQref) / (**mIntfVoltage)(0, 0));
200
201 Complex filterInterfaceInitialVoltage;
202 Complex filterInterfaceInitialCurrent;
203
204 if (mWithConnectionTransformer) {
205 // calculate quantities of low voltage side of transformer (being the interface quantities of the filter)
206 filterInterfaceInitialVoltage =
207 ((**mIntfVoltage)(0, 0) -
208 Complex(mTransformerResistance, mTransformerInductance * **mOmegaN) *
209 (**mIntfCurrent)(0, 0)) /
210 Complex(mTransformerRatioAbs, mTransformerRatioPhase);
211 filterInterfaceInitialCurrent =
212 (**mIntfCurrent)(0, 0) *
213 Complex(mTransformerRatioAbs, mTransformerRatioPhase);
214
215 // connect transformer
216 mVirtualNodes[3]->setInitialVoltage(filterInterfaceInitialVoltage);
217 mConnectionTransformer->connect({mTerminals[0]->node(), mVirtualNodes[3]});
218 } else {
219 // if no transformer used, filter interface equal to inverter interface
220 filterInterfaceInitialVoltage = (**mIntfVoltage)(0, 0);
221 filterInterfaceInitialCurrent = (**mIntfCurrent)(0, 0);
222 }
223
224 // derive initialization quantities of filter
225 Complex vcInit =
226 filterInterfaceInitialVoltage - filterInterfaceInitialCurrent * mRc;
227 Complex icfInit = vcInit * Complex(0., 2. * PI * frequency * mCf);
228 Complex vfInit = vcInit - (filterInterfaceInitialCurrent - icfInit) *
229 Complex(0., 2. * PI * frequency * mLf);
230 Complex vsInit =
231 vfInit - (filterInterfaceInitialCurrent - icfInit) * Complex(mRf, 0);
232 mVirtualNodes[0]->setInitialVoltage(vsInit);
233 mVirtualNodes[1]->setInitialVoltage(vfInit);
234 mVirtualNodes[2]->setInitialVoltage(vcInit);
235
236 // Set parameters electrical subcomponents
237 (**mVsref)(0, 0) = mVirtualNodes[0]->initialSingleVoltage();
238 mSubCtrledVoltageSource->setParameters((**mVsref)(0, 0));
239
240 // Connect electrical subcomponents
241 mSubCtrledVoltageSource->connect({SimNode::GND, mVirtualNodes[0]});
242 mSubResistorF->connect({mVirtualNodes[0], mVirtualNodes[1]});
243 mSubInductorF->connect({mVirtualNodes[1], mVirtualNodes[2]});
244 mSubCapacitorF->connect({mVirtualNodes[2], SimNode::GND});
245 if (mWithConnectionTransformer)
246 mSubResistorC->connect({mVirtualNodes[2], mVirtualNodes[3]});
247 else
248 mSubResistorC->connect({mVirtualNodes[2], mTerminals[0]->node()});
249
250 // Initialize electrical subcomponents
251 for (auto subcomp : mSubComponents) {
252 subcomp->initialize(mFrequencies);
253 subcomp->initializeFromNodesAndTerminals(frequency);
254 }
255
256 // Initialize control subcomponents
257 // current and voltage inputs to PLL and power controller
258 Complex vcdq, ircdq;
259 vcdq = Math::rotatingFrame2to1(
260 mVirtualNodes[3]->initialSingleVoltage(),
261 std::arg(mVirtualNodes[3]->initialSingleVoltage()), 0);
262 ircdq = Math::rotatingFrame2to1(
263 -1. * (**mSubResistorC->mIntfCurrent)(0, 0),
264 std::arg(mVirtualNodes[3]->initialSingleVoltage()), 0);
265 **mVcd = vcdq.real();
266 **mVcq = vcdq.imag();
267 **mIrcd = ircdq.real();
268 **mIrcq = ircdq.imag();
269 // angle input
270 Matrix matrixStateInit = Matrix::Zero(2, 1);
271 matrixStateInit(0, 0) = std::arg(mVirtualNodes[3]->initialSingleVoltage());
272 mPLL->setInitialValues(**mVcq, matrixStateInit, Matrix::Zero(2, 1));
273
274 SPDLOG_LOGGER_INFO(
275 mSLog,
276 "\n--- Initialization from powerflow ---"
277 "\nInterface voltage across: {:s}"
278 "\nInterface current: {:s}"
279 "\nTerminal 0 initial voltage: {:s}"
280 "\nTerminal 0 connected to {:s} = sim node {:d}"
281 "\nVirtual node 0 initial voltage: {:s}"
282 "\nVirtual node 1 initial voltage: {:s}"
283 "\nVirtual node 2 initial voltage: {:s}",
284 Logger::phasorToString((**mIntfVoltage)(0, 0)),
285 Logger::phasorToString((**mIntfCurrent)(0, 0)),
286 Logger::phasorToString(initialSingleVoltage(0)),
287 mTerminals[0]->node()->name(), mTerminals[0]->node()->matrixNodeIndex(),
288 Logger::phasorToString(mVirtualNodes[0]->initialSingleVoltage()),
289 Logger::phasorToString(mVirtualNodes[1]->initialSingleVoltage()),
290 Logger::phasorToString(mVirtualNodes[2]->initialSingleVoltage()));
291 if (mWithConnectionTransformer)
292 SPDLOG_LOGGER_INFO(
293 mSLog, "\nVirtual node 3 initial voltage: {:s}",
294 Logger::phasorToString(mVirtualNodes[3]->initialSingleVoltage()));
295 SPDLOG_LOGGER_INFO(mSLog, "\n--- Initialization from powerflow finished ---");
296}

344void DP::Ph1::AvVoltageSourceInverterDQ::controlStep(Real time,
345 Int timeStepCount) {
346 // Transformation interface forward
347 Complex vcdq, ircdq;
348 vcdq = Math::rotatingFrame2to1(
349 mVirtualNodes[3]->singleVoltage(),
350 mPLL->attributeTyped<Matrix>("output_prev")->get()(0, 0), mThetaN);
351 ircdq = Math::rotatingFrame2to1(
352 -1. * (**mSubResistorC->mIntfCurrent)(0, 0),
353 mPLL->attributeTyped<Matrix>("output_prev")->get()(0, 0), mThetaN);
354 **mVcd = vcdq.real();
355 **mVcq = vcdq.imag();
356 **mIrcd = ircdq.real();
357 **mIrcq = ircdq.imag();
358
359 // add step of subcomponents
360 mPLL->signalStep(time, timeStepCount);
361 mPowerControllerVSI->signalStep(time, timeStepCount);
362
363 // Transformation interface backward
364 (**mVsref)(0, 0) = Math::rotatingFrame2to1(
365 Complex(mPowerControllerVSI->attributeTyped<Matrix>("output_curr")
366 ->get()(0, 0),
367 mPowerControllerVSI->attributeTyped<Matrix>("output_curr")
368 ->get()(1, 0)),
369 mThetaN, mPLL->attributeTyped<Matrix>("output_prev")->get()(0, 0));
370
371 // Update nominal system angle
372 mThetaN = mThetaN + mTimeStep * **mOmegaN;
373}
