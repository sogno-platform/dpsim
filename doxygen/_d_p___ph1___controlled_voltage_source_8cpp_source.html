73void DP::Ph1::ControlledVoltageSource::mnaCompApplySystemMatrixStamp(
74 SparseMatrixRow &systemMatrix) {
75 for (UInt freq = 0; freq < mNumFreqs; freq++) {
76 if (terminalNotGrounded(0)) {
77 Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
78 matrixNodeIndex(0), Complex(-1, 0), mNumFreqs,
79 freq);
80 Math::setMatrixElement(systemMatrix, matrixNodeIndex(0),
81 mVirtualNodes[0]->matrixNodeIndex(),
82 Complex(-1, 0), mNumFreqs, freq);
83 }
84 if (terminalNotGrounded(1)) {
85 Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
86 matrixNodeIndex(1), Complex(1, 0), mNumFreqs,
87 freq);
88 Math::setMatrixElement(systemMatrix, matrixNodeIndex(1),
89 mVirtualNodes[0]->matrixNodeIndex(), Complex(1, 0),
90 mNumFreqs, freq);
91 }
92
93 SPDLOG_LOGGER_INFO(mSLog, "-- Stamp frequency {:d} ---", freq);
94 if (terminalNotGrounded(0)) {
95 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", -1.,
96 matrixNodeIndex(0),
97 mVirtualNodes[0]->matrixNodeIndex());
98 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", -1.,
99 mVirtualNodes[0]->matrixNodeIndex(),
100 matrixNodeIndex(0));
101 }
102 if (terminalNotGrounded(1)) {
103 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
104 mVirtualNodes[0]->matrixNodeIndex(),
105 matrixNodeIndex(1));
106 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
107 matrixNodeIndex(1),
108 mVirtualNodes[0]->matrixNodeIndex());
109 }
110 }
111}
