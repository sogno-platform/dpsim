34void DP::Ph1::ProfileVoltageSource::readFromFile() {
35 mSamples.clear();
36 std::ifstream file(mSourceFile, std::ios::binary | std::ios::ate);
37
38 if (!file.is_open()) {
39 throw SystemError(
40 "ProfileVoltageSource::ProfileVoltageSource could not open file " +
41 mSourceFile.string());
42 }
43
44 std::streamsize size = file.tellg();
45 file.seekg(0, std::ios::beg);
46
47 size_t sample_num =
48 size / (sizeof(struct villas::node::Sample) + SAMPLE_DATA_LENGTH(1));
49 std::vector<char> buffer(size);
50 if (!file.read(buffer.data(), size)) {
51 throw SystemError(
52 "ProfileVoltageSource::ProfileVoltageSource could not read file " +
53 mSourceFile.string());
54 }
55
56 unsigned i, j;
57 Villas__Node__Message *pb_msg;
58
59 pb_msg =
60 villas__node__message__unpack(nullptr, size, (uint8_t *)buffer.data());
61 if (!pb_msg)
62 throw SystemError("ProfileVoltageSource::ProfileVoltageSource could not "
63 "unpack Protobuf message");
64
65 for (i = 0; i < pb_msg->n_samples; i++) {
66 Villas__Node__Sample *pb_smp = pb_msg->samples[i];
67
68 if (pb_smp->type != VILLAS__NODE__SAMPLE__TYPE__DATA)
69 throw SystemError("ProfileVoltageSource::ProfileVoltageSource could not "
70 "unpack Protobuf message");
71
72 if (pb_smp->n_values != 1) {
73 throw SystemError("ProfileVoltageSource::ProfileVoltageSource could not "
74 "unpack Protobuf message");
75 }
76
77 for (j = 0; j < pb_smp->n_values; j++) {
78 Villas__Node__Value *pb_val = pb_smp->values[j];
79
80 if (pb_val->value_case != VILLAS__NODE__VALUE__VALUE_F)
81 throw SystemError("ProfileVoltageSource::ProfileVoltageSource could "
82 "not unpack Protobuf message");
83 mSamples.push_back(pb_val->f);
84 }
85 }
86
87 villas__node__message__free_unpacked(pb_msg, nullptr);
88
89 std::cout << "Read " << mSamples.size() << " samples from file "
90 << mSourceFile << std::endl;
91 for (double sample : mSamples) {
92 std::cout << sample << std::endl;
93 }
94}

163void DP::Ph1::ProfileVoltageSource::mnaCompApplySystemMatrixStamp(
164 SparseMatrixRow &systemMatrix) {
165 for (UInt freq = 0; freq < mNumFreqs; freq++) {
166 if (terminalNotGrounded(0)) {
167 Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
168 matrixNodeIndex(0), Complex(-1, 0), mNumFreqs,
169 freq);
170 Math::setMatrixElement(systemMatrix, matrixNodeIndex(0),
171 mVirtualNodes[0]->matrixNodeIndex(),
172 Complex(-1, 0), mNumFreqs, freq);
173 }
174 if (terminalNotGrounded(1)) {
175 Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
176 matrixNodeIndex(1), Complex(1, 0), mNumFreqs,
177 freq);
178 Math::setMatrixElement(systemMatrix, matrixNodeIndex(1),
179 mVirtualNodes[0]->matrixNodeIndex(), Complex(1, 0),
180 mNumFreqs, freq);
181 }
182
183 SPDLOG_LOGGER_INFO(mSLog, "-- Stamp frequency {:d} ---", freq);
184 if (terminalNotGrounded(0)) {
185 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", -1.,
186 matrixNodeIndex(0),
187 mVirtualNodes[0]->matrixNodeIndex());
188 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", -1.,
189 mVirtualNodes[0]->matrixNodeIndex(),
190 matrixNodeIndex(0));
191 }
192 if (terminalNotGrounded(1)) {
193 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
194 mVirtualNodes[0]->matrixNodeIndex(),
195 matrixNodeIndex(1));
196 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
197 matrixNodeIndex(1),
198 mVirtualNodes[0]->matrixNodeIndex());
199 }
200 }

292void DP::Ph1::ProfileVoltageSource::daeResidual(double ttime,
293 const double state[],
294 const double dstate_dt[],
295 double resid[],
296 std::vector<int> &off) {
297 /* New state vector definintion:
298 * state[0]=node0_voltage
299 * state[1]=node1_voltage
300 * ....
301 * state[n]=noden_voltage
302 * state[n+1]=component0_voltage
303 * state[n+2]=component0_inductance (not yet implemented)
304 * ...
305 * state[m-1]=componentm_voltage
306 * state[m]=componentm_inductance
307 */
308
309 int Pos1 = matrixNodeIndex(0);
310 int Pos2 = matrixNodeIndex(1);
311 int c_offset = off[0] + off[1]; // current offset for component
312 int n_offset_1 =
313 c_offset + Pos1 + 1; // current offset for first nodal equation
314 int n_offset_2 =
315 c_offset + Pos2 + 1; // current offset for second nodal equation
316 resid[c_offset] = (state[Pos2] - state[Pos1]) -
317 state[c_offset]; // Voltage equation for Resistor
318 // resid[++c_offset] = ; //TODO : add inductance equation
319 resid[n_offset_1] += (**mIntfCurrent)(0, 0).real();
320 resid[n_offset_2] += (**mIntfCurrent)(0, 0).real();
321 off[1] += 1;
322}
