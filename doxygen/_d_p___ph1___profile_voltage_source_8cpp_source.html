30 void DP::Ph1::ProfileVoltageSource::readFromFile() {
31  mSamples.clear();
32  std::ifstream file(mSourceFile, std::ios::binary | std::ios::ate);
33 
34  if (!file.is_open()) {
35  throw SystemError("ProfileVoltageSource::ProfileVoltageSource could not open file " + mSourceFile.string());
36  }
37 
38  std::streamsize size = file.tellg();
39  file.seekg(0, std::ios::beg);
40 
41  size_t sample_num = size / (sizeof(struct villas::node::Sample) + SAMPLE_DATA_LENGTH(1));
42  std::vector<char> buffer(size);
43  if (!file.read(buffer.data(), size)) {
44  throw SystemError("ProfileVoltageSource::ProfileVoltageSource could not read file " + mSourceFile.string());
45  }
46 
47  unsigned i, j;
48  Villas__Node__Message *pb_msg;
49 
50  pb_msg = villas__node__message__unpack(nullptr, size, (uint8_t *)buffer.data());
51  if (!pb_msg)
52  throw SystemError("ProfileVoltageSource::ProfileVoltageSource could not unpack Protobuf message");
53 
54  for (i = 0; i < pb_msg->n_samples; i++) {
55  Villas__Node__Sample *pb_smp = pb_msg->samples[i];
56 
57  if (pb_smp->type != VILLAS__NODE__SAMPLE__TYPE__DATA)
58  throw SystemError("ProfileVoltageSource::ProfileVoltageSource could not unpack Protobuf message");
59 
60  if (pb_smp->n_values != 1) {
61  throw SystemError("ProfileVoltageSource::ProfileVoltageSource could not unpack Protobuf message");
62  }
63 
64  for (j = 0; j < pb_smp->n_values; j++) {
65  Villas__Node__Value *pb_val = pb_smp->values[j];
66 
67  if (pb_val->value_case != VILLAS__NODE__VALUE__VALUE_F)
68  throw SystemError("ProfileVoltageSource::ProfileVoltageSource could not unpack Protobuf message");
69  mSamples.push_back(pb_val->f);
70  }
71  }
72 
73  villas__node__message__free_unpacked(pb_msg, nullptr);
74 
75  std::cout << "Read " << mSamples.size() << " samples from file " << mSourceFile << std::endl;
76  for (double sample : mSamples) {
77  std::cout << sample << std::endl;
78  }
79 }

220 void DP::Ph1::ProfileVoltageSource::daeResidual(double ttime, const double state[], const double dstate_dt[], double resid[], std::vector<int> &off) {
221  /* new state vector definintion:
222  state[0]=node0_voltage
223  state[1]=node1_voltage
224  ....
225  state[n]=noden_voltage
226  state[n+1]=component0_voltage
227  state[n+2]=component0_inductance (not yet implemented)
228  ...
229  state[m-1]=componentm_voltage
230  state[m]=componentm_inductance
231  */
232 
233  int Pos1 = matrixNodeIndex(0);
234  int Pos2 = matrixNodeIndex(1);
235  int c_offset = off[0] + off[1]; // current offset for component
236  int n_offset_1 = c_offset + Pos1 + 1; // current offset for first nodal equation
237  int n_offset_2 = c_offset + Pos2 + 1; // current offset for second nodal equation
238  resid[c_offset] = (state[Pos2] - state[Pos1]) - state[c_offset]; // Voltage equation for Resistor
239  // resid[++c_offset] = ; //TODO : add inductance equation
240  resid[n_offset_1] += (**mIntfCurrent)(0, 0).real();
241  resid[n_offset_2] += (**mIntfCurrent)(0, 0).real();
242  off[1] += 1;
243 }
