32 void DP::Ph1::ReducedOrderSynchronGeneratorVBR::calculateConductanceMatrix() {
33  MatrixFixedSize<2, 2> resistanceMatrix = MatrixFixedSize<2, 2>::Zero(2, 2);
34  Real DeltaTheta = **mThetaMech - mBase_OmMech * mSimTime;
35  resistanceMatrix(0, 0) = -(mA + mB) / 2.0 * sin(2 * DeltaTheta);
36  resistanceMatrix(0, 1) =
37  (mA - mB) / 2.0 + (mA + mB) / 2.0 * cos(2 * DeltaTheta);
38  resistanceMatrix(1, 0) =
39  -(mA - mB) / 2.0 + (mA + mB) / 2.0 * cos(2 * DeltaTheta);
40  resistanceMatrix(1, 1) = (mA + mB) / 2.0 * sin(2 * DeltaTheta);
41  resistanceMatrix = resistanceMatrix * mBase_Z;
42  mConductanceMatrix = resistanceMatrix.inverse();
43 }

126 void DP::Ph1::ReducedOrderSynchronGeneratorVBR::mnaCompApplySystemMatrixStamp(

128  if (mModelAsNortonSource) {
129  // Stamp conductance matrix
130  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 0),
131  matrixNodeIndex(0, 0), mConductanceMatrix);
132  } else {
133  // Stamp voltage source
134  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
135  mVirtualNodes[1]->matrixNodeIndex(), Complex(-1, 0));
136  Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(),
137  mVirtualNodes[0]->matrixNodeIndex(), Complex(1, 0));
138 
139  // Stamp conductance matrix
140  // set upper left block
141  Math::addToMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
142  mVirtualNodes[0]->matrixNodeIndex(),
143  mConductanceMatrix);
144 
145  // set bottom right block
146  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 0),

148 
149  // Set off diagonal blocks

159  mnaCompApplyRightSideVectorStamp(Matrix &rightVector) {

171  mEvbr);

175 void DP::Ph1::ReducedOrderSynchronGeneratorVBR::mnaCompPostStep(
176  const Matrix &leftVector) {
177  // update armature voltage
178  (**mIntfVoltage)(0, 0) =
179  Math::complexFromVectorElement(leftVector, matrixNodeIndex(0, 0));
180 
181  // convert armature voltage to dq reference frame
182  **mVdq = mDomainInterface.applyDPToDQTransform((**mIntfVoltage)(0, 0)) /
183  mBase_V_RMS;
184 
185  // update armature current
186  if (mModelAsNortonSource) {
187  (**mIntfCurrent)(0, 0) =
188  mIvbr -
189  Complex(mConductanceMatrix(0, 0) * (**mIntfVoltage)(0, 0).real() +
190  mConductanceMatrix(0, 1) * (**mIntfVoltage)(0, 0).imag(),
191  mConductanceMatrix(1, 0) * (**mIntfVoltage)(0, 0).real() +
192  mConductanceMatrix(1, 1) * (**mIntfVoltage)(0, 0).imag());
193  } else {
194  (**mIntfCurrent)(0, 0) = Math::complexFromVectorElement(
195  leftVector, mVirtualNodes[1]->matrixNodeIndex());
196  }
197 
198  // convert armature current to dq reference frame
199  **mIdq = mDomainInterface.applyDPToDQTransform((**mIntfCurrent)(0, 0)) /
200  mBase_I_RMS;
201 }
