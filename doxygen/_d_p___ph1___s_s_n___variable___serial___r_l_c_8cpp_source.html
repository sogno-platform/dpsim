19void DP::Ph1::SSN::Variable_Serial_RLC::buildStateSpaceModel(Matrix &A,
20 Matrix &B,
21 Matrix &C,
22 Matrix &D) const {
23 // Packed real state [Re(vC), Im(vC), Re(iL), Im(iL)]: series RLC coupling
24 // applied identically to Re/Im, plus -j*omegaN as a same-state rotation.
25 A = Matrix::Zero(4, 4);
26 A(0, 1) = mOmegaN;
27 A(0, 2) = 1. / mCapacitance;
28 A(1, 0) = -mOmegaN;
29 A(1, 3) = 1. / mCapacitance;
30 A(2, 0) = -1. / mInductance;
31 A(2, 2) = -mResistance / mInductance;
32 A(2, 3) = mOmegaN;
33 A(3, 1) = -1. / mInductance;
34 A(3, 2) = -mOmegaN;
35 A(3, 3) = -mResistance / mInductance;
36
37 B = Matrix::Zero(4, 2);
38 B(2, 0) = 1. / mInductance;
39 B(3, 1) = 1. / mInductance;
40
41 C = Matrix::Zero(2, 4);
42 C(0, 2) = 1.;
43 C(1, 3) = 1.;
44
45 D = Matrix::Zero(2, 2);
46}
