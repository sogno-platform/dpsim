30void DP::Ph1::SynchronGenerator4OrderVBR::specificInitialization() {

46void DP::Ph1::SynchronGenerator4OrderVBR::stepInPerUnit() {
47
48 // update DP-DQ transforms
49 mDomainInterface.updateDQToDPTransform(**mThetaMech, mSimTime);
50 mDomainInterface.updateDPToDQTransform(**mThetaMech, mSimTime);
51
52 // calculate Edq_t at t=k
53 (**mEdq_t)(0, 0) = (**mVdq)(0, 0) - (**mIdq)(1, 0) * mLq_t;
54 (**mEdq_t)(1, 0) = (**mVdq)(1, 0) + (**mIdq)(0, 0) * mLd_t;
55
56 // Update time-varying reactance matrix
57 calculateConductanceMatrix();
58
59 // VBR history voltage
60 mEh_vbr(0, 0) = mAd_t * (**mIdq)(1, 0) + mBd_t * (**mEdq_t)(0, 0);
61 mEh_vbr(1, 0) = mAq_t * (**mIdq)(0, 0) + mBq_t * (**mEdq_t)(1, 0) +
62 mDq_t * mEf_prev + mDq_t * (**mEf);
63
64 // convert Edq_t into the abc reference frame
65 mEvbr = mDomainInterface.applyDQToDPTransform(mEh_vbr) * mBase_V_RMS;
66}
