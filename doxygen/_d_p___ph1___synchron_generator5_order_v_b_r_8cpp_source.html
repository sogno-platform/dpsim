33void DP::Ph1::SynchronGenerator5OrderVBR::specificInitialization() {
34
35 // initial voltage behind the transient reactance in the dq reference frame
36 (**mEdq_t)(0, 0) = 0.0;
37 (**mEdq_t)(1, 0) =
38 (1 - mTaa / mTd0_t) * **mEf - (mLd - mLd_t - mYd) * (**mIdq)(0, 0);
39
40 // initial dq behind the subtransient reactance in the dq reference frame
41 (**mEdq_s)(0, 0) = (**mVdq)(0, 0) - (mLq_s) * (**mIdq)(1, 0);
42 (**mEdq_s)(1, 0) = (**mVdq)(1, 0) + (mLd_s) * (**mIdq)(0, 0);
43
44 // initial history term behind the transient reactance
45 mEh_t(0, 0) = 0.0;
46 mEh_t(1, 0) = mAq_t * (**mIdq)(0, 0) + mBq_t * (**mEdq_t)(1, 0) +
47 mDq_t * (**mEf) + mDq_t * mEf_prev;
48
49 SPDLOG_LOGGER_DEBUG(mSLog,
50 "\n--- Model DPecific initialization ---"
51 "\nSG model: 5th order"
52 "\nInitial Ed_t (per unit): {:f}"
53 "\nInitial Eq_t (per unit): {:f}"
54 "\nInitial Ed_s (per unit): {:f}"
55 "\nInitial Eq_s (per unit): {:f}"
56 "\n--- Model DPecific initialization finished ---",
57 (**mEdq_t)(0, 0), (**mEdq_t)(1, 0), (**mEdq_s)(0, 0),
58 (**mEdq_s)(1, 0));
59 mSLog->flush();
60}

62void DP::Ph1::SynchronGenerator5OrderVBR::stepInPerUnit() {
63 // update DP-DQ transforms
64 mDomainInterface.updateDQToDPTransform(**mThetaMech, mSimTime);
65 mDomainInterface.updateDPToDQTransform(**mThetaMech, mSimTime);
66
67 // calculate Edq_t at t=k
68 (**mEdq_t)(0, 0) = 0.0;
69 (**mEdq_t)(1, 0) = mAq_t * (**mIdq)(0, 0) + mEh_t(1, 0);
70
71 // calculate Edq_s at t=k
72 (**mEdq_s)(0, 0) = (**mVdq)(0, 0) - mLq_s * (**mIdq)(1, 0);
73 (**mEdq_s)(1, 0) = (**mVdq)(1, 0) + mLd_s * (**mIdq)(0, 0);
74
75 // VBR history voltage
76 calculateConductanceMatrix();
77
78 // calculate history term behind the transient reactance
79 mEh_t(0, 0) = 0.0;
80 mEh_t(1, 0) = mAq_t * (**mIdq)(0, 0) + mBq_t * (**mEdq_t)(1, 0) +
81 mDq_t * (**mEf) + mDq_t * mEf_prev;
82
83 // calculate history term behind the subtransient reactance
84 mEh_s(0, 0) = mAd_s * (**mIdq)(1, 0) + mCd_s * (**mEdq_s)(0, 0);
85 mEh_s(1, 0) = mAq_s * (**mIdq)(0, 0) + mBq_s * (**mEdq_t)(1, 0) +
86 mCq_s * (**mEdq_s)(1, 0) + mDq_s * (**mEf) + mDq_s * mEf_prev;
87
88 // convert Edq_t into the abc reference frame
89 mEvbr = mDomainInterface.applyDQToDPTransform(mEh_s) * mBase_V_RMS;
90}
