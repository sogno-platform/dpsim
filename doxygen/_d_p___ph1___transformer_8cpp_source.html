51 void DP::Ph1::Transformer::initializeFromNodesAndTerminals(Real frequency) {
52 
53  // Component parameters are referred to high voltage side.
54  // Switch terminals if transformer is connected the other way around.
55  if (Math::abs(mRatio) < 1.) {
56  mRatio = 1. / mRatio;
57  std::shared_ptr<SimTerminal<Complex>> tmp = mTerminals[0];
58  mTerminals[0] = mTerminals[1];
59  mTerminals[1] = tmp;
60  }
61 
62  // Set initial voltage of virtual node in between
63  mVirtualNodes[0]->setInitialVoltage( initialSingleVoltage(1) * mRatio );
64 
65  // Static calculations from load flow data
66  Real omega = 2.*PI* frequency;
67  Complex impedance = { mResistance, omega * mInductance };
68  mSLog->info("Reactance={} [Ohm] (referred to primary side)", omega * mInductance );
69  mIntfVoltage(0,0) = mVirtualNodes[0]->initialSingleVoltage() - initialSingleVoltage(0);
70  mIntfCurrent(0,0) = mIntfVoltage(0,0) / impedance;
71 
72  // Create series sub components
73  mSubInductor = std::make_shared<DP::Ph1::Inductor>(mName + "_ind", mLogLevel);
74  mSubInductor->setParameters(mInductance);
75 
76  if (mNumVirtualNodes == 3) {
77  mVirtualNodes[2]->setInitialVoltage(initialSingleVoltage(0));
78  mSubResistor = std::make_shared<DP::Ph1::Resistor>(mName + "_res", mLogLevel);
79  mSubResistor->setParameters(mResistance);
80  mSubResistor->connect({node(0), mVirtualNodes[2]});
81  mSubResistor->initialize(mFrequencies);
82  mSubResistor->initializeFromNodesAndTerminals(frequency);
83  mSubInductor->connect({mVirtualNodes[2], mVirtualNodes[0]});
84  } else {
85  mSubInductor->connect({node(0), mVirtualNodes[0]});
86  }
87  mSubInductor->initialize(mFrequencies);
88  mSubInductor->initializeFromNodesAndTerminals(frequency);
89 
90  // Create parallel sub components
91  // A snubber conductance is added on the low voltage side (resistance scaled with lower voltage side)
92  mSnubberResistance = mNominalVoltageEnd1 > mNominalVoltageEnd2 ? std::abs(mNominalVoltageEnd2)*1e6 : std::abs(mNominalVoltageEnd1)*1e6;
93  mSubSnubResistor = std::make_shared<DP::Ph1::Resistor>(mName + "_snub_res", mLogLevel);
94  mSubSnubResistor->setParameters(mSnubberResistance);
95  mSubSnubResistor->connect({ node(1), DP::SimNode::GND });
96  mSubSnubResistor->initialize(mFrequencies);
97  mSubSnubResistor->initializeFromNodesAndTerminals(frequency);
98  mSLog->info("Snubber Resistance={} [Ohm] (connected to LV side)", mSnubberResistance);
99 
100  mSLog->info(
101  "\n--- Initialization from powerflow ---"
102  "\nVoltage across: {:s}"
103  "\nCurrent: {:s}"
104  "\nTerminal 0 voltage: {:s}"
105  "\nTerminal 1 voltage: {:s}"
106  "\nVirtual Node 1 voltage: {:s}"
107  "\n--- Initialization from powerflow finished ---",
108  Logger::phasorToString(mIntfVoltage(0,0)),
109  Logger::phasorToString(mIntfCurrent(0,0)),
110  Logger::phasorToString(initialSingleVoltage(0)),
111  Logger::phasorToString(initialSingleVoltage(1)),
112  Logger::phasorToString(mVirtualNodes[0]->initialSingleVoltage()));
113 }

115 void DP::Ph1::Transformer::mnaInitialize(Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
116  MNAInterface::mnaInitialize(omega, timeStep);
117  updateMatrixNodeIndices();
118 
119  mRightVector = Matrix::Zero(leftVector->get().rows(), 1);
120  auto subComponents = MNAInterface::List({mSubInductor, mSubSnubResistor});
121 
122  if (mSubResistor)
123  subComponents.push_back(mSubResistor);
124 
125  for (auto comp : subComponents) {
126  comp->mnaInitialize(omega, timeStep, leftVector);
127 
128  for (auto task : comp->mnaTasks())
129  mMnaTasks.push_back(task);
130 
131  }
132 
133  mMnaTasks.push_back(std::make_shared<MnaPreStep>(*this));
134  mMnaTasks.push_back(std::make_shared<MnaPostStep>(*this, leftVector));
135 
136  mSLog->info(
137  "\nTerminal 0 connected to {:s} = sim node {:d}"
138  "\nTerminal 1 connected to {:s} = sim node {:d}",
139  mTerminals[0]->node()->name(), mTerminals[0]->node()->matrixNodeIndex(),
140  mTerminals[1]->node()->name(), mTerminals[1]->node()->matrixNodeIndex());
141 }

143 void DP::Ph1::Transformer::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
144  // Ideal transformer equations
145  if (terminalNotGrounded(0)) {
146  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(), mVirtualNodes[1]->matrixNodeIndex(), Complex(-1.0, 0));
147  Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(), mVirtualNodes[0]->matrixNodeIndex(), Complex(1.0, 0));
148  }
149  if (terminalNotGrounded(1)) {
150  Math::setMatrixElement(systemMatrix, matrixNodeIndex(1), mVirtualNodes[1]->matrixNodeIndex(), mRatio);
151  Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(), matrixNodeIndex(1), -mRatio);
152  }
153 
154  // Add inductive part to system matrix
155  mSubInductor->mnaApplySystemMatrixStamp(systemMatrix);
156  mSubSnubResistor->mnaApplySystemMatrixStamp(systemMatrix);
157 
158  if (mNumVirtualNodes == 3) {
159  mSubResistor->mnaApplySystemMatrixStamp(systemMatrix);
160  }
161 
162  if (terminalNotGrounded(0)) {
163  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(Complex(-1.0, 0)),
164  mVirtualNodes[0]->matrixNodeIndex(), mVirtualNodes[1]->matrixNodeIndex());
165  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(Complex(1.0, 0)),
166  mVirtualNodes[1]->matrixNodeIndex(), mVirtualNodes[0]->matrixNodeIndex());
167  }
168  if (terminalNotGrounded(1)) {
169  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(mRatio),
170  matrixNodeIndex(1), mVirtualNodes[1]->matrixNodeIndex());
171  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(-mRatio),
172  mVirtualNodes[1]->matrixNodeIndex(), matrixNodeIndex(1));
173  }
174 }
