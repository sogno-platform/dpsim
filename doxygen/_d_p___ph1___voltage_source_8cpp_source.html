70 void DP::Ph1::VoltageSource::initializeFromNodesAndTerminals(Real frequency) {
72  if (**mVoltageRef == Complex(0, 0))
73  **mVoltageRef = initialSingleVoltage(1) - initialSingleVoltage(0);
74 
75  if (mSrcSig == nullptr) {
76  auto srcSigSine = Signal::SineWaveGenerator::make(**mName);
77  srcSigSine->mVoltageRef->setReference(mVoltageRef);
78  srcSigSine->mFreq->setReference(mSrcFreq);
79  srcSigSine->setParameters(**mVoltageRef);
80  mSrcSig = srcSigSine;
81  }
82 
83  SPDLOG_LOGGER_INFO(mSLog,
84  "\n--- Initialization from node voltages ---"
85  "\nVoltage across: {:s}"
86  "\nTerminal 0 voltage: {:s}"
87  "\nTerminal 1 voltage: {:s}"
88  "\n--- Initialization from node voltages ---",
89  Logger::phasorToString(mSrcSig->getSignal()),
90  Logger::phasorToString(initialSingleVoltage(0)),
91  Logger::phasorToString(initialSingleVoltage(1)));
92 }

141 void DP::Ph1::VoltageSource::mnaCompApplySystemMatrixStamp(
142  SparseMatrixRow &systemMatrix) {
143  for (UInt freq = 0; freq < mNumFreqs; freq++) {
144  if (terminalNotGrounded(0)) {
145  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
146  matrixNodeIndex(0), Complex(-1, 0), mNumFreqs,
147  freq);
148  Math::setMatrixElement(systemMatrix, matrixNodeIndex(0),
149  mVirtualNodes[0]->matrixNodeIndex(),
150  Complex(-1, 0), mNumFreqs, freq);
151  }
152  if (terminalNotGrounded(1)) {
153  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
154  matrixNodeIndex(1), Complex(1, 0), mNumFreqs,
155  freq);
156  Math::setMatrixElement(systemMatrix, matrixNodeIndex(1),
157  mVirtualNodes[0]->matrixNodeIndex(), Complex(1, 0),
158  mNumFreqs, freq);
159  }
160 
161  SPDLOG_LOGGER_INFO(mSLog, "-- Stamp frequency {:d} ---", freq);
162  if (terminalNotGrounded(0)) {
163  SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", -1.,
164  matrixNodeIndex(0),
165  mVirtualNodes[0]->matrixNodeIndex());
166  SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", -1.,
167  mVirtualNodes[0]->matrixNodeIndex(),
168  matrixNodeIndex(0));
169  }
170  if (terminalNotGrounded(1)) {
171  SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
172  mVirtualNodes[0]->matrixNodeIndex(),
173  matrixNodeIndex(1));
174  SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
175  matrixNodeIndex(1),
176  mVirtualNodes[0]->matrixNodeIndex());
177  }
178  }

275 void DP::Ph1::VoltageSource::daeResidual(double ttime, const double state[],
276  const double dstate_dt[],
277  double resid[],
278  std::vector<int> &off) {
279  /* new state vector definintion:
280  state[0]=node0_voltage
281  state[1]=node1_voltage
282  ....
283  state[n]=noden_voltage
284  state[n+1]=component0_voltage
285  state[n+2]=component0_inductance (not yet implemented)
286  ...
287  state[m-1]=componentm_voltage
288  state[m]=componentm_inductance
289  */
290 
291  int Pos1 = matrixNodeIndex(0);
292  int Pos2 = matrixNodeIndex(1);
293  int c_offset = off[0] + off[1]; //current offset for component
294  int n_offset_1 =
295  c_offset + Pos1 + 1; // current offset for first nodal equation
296  int n_offset_2 =
297  c_offset + Pos2 + 1; // current offset for second nodal equation
298  resid[c_offset] = (state[Pos2] - state[Pos1]) -
299  state[c_offset]; // Voltage equation for Resistor
300  //resid[++c_offset] = ; //TODO : add inductance equation
301  resid[n_offset_1] += (**mIntfCurrent)(0, 0).real();
302  resid[n_offset_2] += (**mIntfCurrent)(0, 0).real();
303  off[1] += 1;
304 }
