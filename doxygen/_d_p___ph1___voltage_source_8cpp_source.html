80void DP::Ph1::VoltageSource::initializeFromNodesAndTerminals(Real frequency) {
82 if (**mVoltageRef == Complex(0, 0))
83 **mVoltageRef = initialSingleVoltage(1) - initialSingleVoltage(0);
84
85 if (mSrcSig == nullptr) {
86 auto srcSigSine = Signal::SineWaveGenerator::make(**mName);
87 srcSigSine->mVoltageRef->setReference(mVoltageRef);
88 srcSigSine->mFreq->setReference(mSrcFreq);
89 srcSigSine->setParameters(**mVoltageRef);
90 mSrcSig = srcSigSine;
91 }
92
93 SPDLOG_LOGGER_INFO(mSLog,
94 "\n--- Initialization from node voltages ---"
95 "\nVoltage across: {:s}"
96 "\nTerminal 0 voltage: {:s}"
97 "\nTerminal 1 voltage: {:s}"
98 "\n--- Initialization from node voltages ---",
99 Logger::phasorToString(mSrcSig->getSignal()),
100 Logger::phasorToString(initialSingleVoltage(0)),
101 Logger::phasorToString(initialSingleVoltage(1)));
102}

151void DP::Ph1::VoltageSource::mnaCompApplySystemMatrixStamp(
152 SparseMatrixRow &systemMatrix) {
153 for (UInt freq = 0; freq < mNumFreqs; freq++) {
154 if (terminalNotGrounded(0)) {
155 Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
156 matrixNodeIndex(0), Complex(-1, 0), mNumFreqs,
157 freq);
158 Math::setMatrixElement(systemMatrix, matrixNodeIndex(0),
159 mVirtualNodes[0]->matrixNodeIndex(),
160 Complex(-1, 0), mNumFreqs, freq);
161 }
162 if (terminalNotGrounded(1)) {
163 Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
164 matrixNodeIndex(1), Complex(1, 0), mNumFreqs,
165 freq);
166 Math::setMatrixElement(systemMatrix, matrixNodeIndex(1),
167 mVirtualNodes[0]->matrixNodeIndex(), Complex(1, 0),
168 mNumFreqs, freq);
169 }
170
171 SPDLOG_LOGGER_INFO(mSLog, "-- Stamp frequency {:d} ---", freq);
172 if (terminalNotGrounded(0)) {
173 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", -1.,
174 matrixNodeIndex(0),
175 mVirtualNodes[0]->matrixNodeIndex());
176 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", -1.,
177 mVirtualNodes[0]->matrixNodeIndex(),
178 matrixNodeIndex(0));
179 }
180 if (terminalNotGrounded(1)) {
181 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
182 mVirtualNodes[0]->matrixNodeIndex(),
183 matrixNodeIndex(1));
184 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
185 matrixNodeIndex(1),
186 mVirtualNodes[0]->matrixNodeIndex());
187 }
188 }

285void DP::Ph1::VoltageSource::daeResidual(double ttime, const double state[],
286 const double dstate_dt[],
287 double resid[],
288 std::vector<int> &off) {
289 /* New state vector definintion:
290 * state[0]=node0_voltage
291 * state[1]=node1_voltage
292 * ....
293 * state[n]=noden_voltage
294 * state[n+1]=component0_voltage
295 * state[n+2]=component0_inductance (not yet implemented)
296 * ...
297 * state[m-1]=componentm_voltage
298 * state[m]=componentm_inductance
299 */
300
301 int Pos1 = matrixNodeIndex(0);
302 int Pos2 = matrixNodeIndex(1);
303 int c_offset = off[0] + off[1]; //current offset for component
304 int n_offset_1 =
305 c_offset + Pos1 + 1; // current offset for first nodal equation
306 int n_offset_2 =
307 c_offset + Pos2 + 1; // current offset for second nodal equation
308 resid[c_offset] = (state[Pos2] - state[Pos1]) -
309 state[c_offset]; // Voltage equation for Resistor
310 //resid[++c_offset] = ; //TODO : add inductance equation
311 resid[n_offset_1] += (**mIntfCurrent)(0, 0).real();
312 resid[n_offset_2] += (**mIntfCurrent)(0, 0).real();
313 off[1] += 1;
314}
