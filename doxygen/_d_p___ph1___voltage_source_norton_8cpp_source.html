51 void DP::Ph1::VoltageSourceNorton::mnaCompApplySystemMatrixStamp(
52  SparseMatrixRow &systemMatrix) {
53  // Apply matrix stamp for equivalent resistance
54  if (terminalNotGrounded(0))
55  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0),
56  matrixNodeIndex(0), Complex(mConductance, 0));
57  if (terminalNotGrounded(1))
58  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1),
59  matrixNodeIndex(1), Complex(mConductance, 0));
60  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
61  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0),
62  matrixNodeIndex(1), Complex(-mConductance, 0));
63  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1),
64  matrixNodeIndex(0), Complex(-mConductance, 0));
65  }
66 
67  if (terminalNotGrounded(0))
68  SPDLOG_LOGGER_INFO(mSLog, "Add {:s} to system at ({:d},{:d})",
69  Logger::complexToString(Complex(mConductance, 0)),
70  matrixNodeIndex(0), matrixNodeIndex(0));
71  if (terminalNotGrounded(1))
72  SPDLOG_LOGGER_INFO(mSLog, "Add {:s} to system at ({:d},{:d})",
73  Logger::complexToString(Complex(mConductance, 0)),
74  matrixNodeIndex(1), matrixNodeIndex(1));
75  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
76  SPDLOG_LOGGER_INFO(mSLog, "Add {:s} to system at ({:d},{:d})",
77  Logger::complexToString(Complex(-mConductance, 0)),
78  matrixNodeIndex(0), matrixNodeIndex(1));
79  SPDLOG_LOGGER_INFO(mSLog, "Add {:s} to system at ({:d},{:d})",
80  Logger::complexToString(Complex(-mConductance, 0)),
81  matrixNodeIndex(1), matrixNodeIndex(0));
82  }
83 }

88 void DP::Ph1::VoltageSourceNorton::mnaCompApplyRightSideVectorStamp(
89  Matrix &rightVector) {

99 void DP::Ph1::VoltageSourceNorton::updateState(Real time) {
100  if (**mSrcFreq >= 0) {
101  (**mIntfVoltage)(0, 0) = Complex(
102  Math::abs(**mVoltageRef) *
103  cos(time * 2. * PI * **mSrcFreq + Math::phase(**mVoltageRef)),
104  Math::abs(**mVoltageRef) *
105  sin(time * 2. * PI * **mSrcFreq + Math::phase(**mVoltageRef)));
106  } else {
107  // If source frequency -1, use system frequency.
108  (**mIntfVoltage)(0, 0) = **mVoltageRef;
109  }

141 void DP::Ph1::VoltageSourceNorton::mnaCompUpdateVoltage(
142  const Matrix &leftVector) {
143  // Calculate v1 - v0
144  (**mIntfVoltage)(0, 0) = 0;

147  Math::complexFromVectorElement(leftVector, matrixNodeIndex(1));
