67 void DP::Ph1::VoltageSourceRamp::updateState(Real time) {
68  (**mIntfVoltage)(0, 0) = **mVoltageRef;
69 
70  if (time >= mSwitchTime && time < mSwitchTime + mRampTime) {
71  Real voltageAbs = Math::abs(**mVoltageRef +
72  (time - mSwitchTime) / mRampTime * mAddVoltage);
73  Real voltagePhase = Math::phase(
74  **mVoltageRef + (time - mSwitchTime) / mRampTime * mAddVoltage);
75  Real fadeInOut =
76  0.5 + 0.5 * sin((time - mSwitchTime) / mRampTime * PI + -PI / 2);
77  (**mIntfVoltage)(0, 0) =
78  Math::polar(voltageAbs, voltagePhase + fadeInOut * mAddSrcFreq * time);
79  } else if (time >= mSwitchTime + mRampTime) {
80  Real voltageAbs = Math::abs(**mVoltageRef + mAddVoltage);
81  Real voltagePhase = Math::phase(**mVoltageRef + mAddVoltage);
82  (**mIntfVoltage)(0, 0) =
83  Math::polar(voltageAbs, voltagePhase + mAddSrcFreq * time);
84  }
85 }
