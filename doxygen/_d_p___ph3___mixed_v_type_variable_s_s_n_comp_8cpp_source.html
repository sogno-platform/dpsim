46Matrix DP::Ph3::MixedVTypeVariableSSNComp::packComplex(const MatrixComp &c) {
47 const Matrix::Index m = c.rows();
48 Matrix v(2 * m, 1);
49 for (Matrix::Index i = 0; i < m; ++i) {
50 v(2 * i, 0) = c(i, 0).real();
51 v(2 * i + 1, 0) = c(i, 0).imag();
52 }
53 return v;
54}

56MatrixComp DP::Ph3::MixedVTypeVariableSSNComp::unpackComplex(const Matrix &v) {
57 const Matrix::Index m = v.rows() / 2;
58 MatrixComp c(m, 1);
59 for (Matrix::Index i = 0; i < m; ++i)
60 c(i, 0) = Complex(v(2 * i, 0), v(2 * i + 1, 0));
61 return c;
62}

64MatrixComp
65DP::Ph3::MixedVTypeVariableSSNComp::foldComplexMatrix(const Matrix &real) {
66 const Matrix::Index m = real.rows() / 2;
67 MatrixComp c(m, real.cols() / 2);
68 for (Matrix::Index i = 0; i < m; ++i)
69 for (Matrix::Index j = 0; j < real.cols() / 2; ++j)
70 // Trust the [[a,-b],[b,a]] complex-multiplication structure of each
71 // 2x2 block (Irc = (U - Vc)/Rc is an exact real-scalar per-phase
72 // relation), as DP::SSNComp does; not re-checked at runtime.
73 c(i, j) = Complex(real(2 * i, 2 * j), real(2 * i + 1, 2 * j));
74 return c;
75}

104void DP::Ph3::MixedVTypeVariableSSNComp::setParameters(
105 const Matrix &A, const Matrix &B, const Matrix &C, const Matrix &D,
106 const Matrix &E, const Matrix &F) {
107 mParametersSet = false;
108
109 const Matrix::Index n = stateSize();
110 const Matrix::Index m = 2 * mPhaseCount;
111
112 if (A.rows() != n || A.cols() != n)
113 throw std::invalid_argument("A has invalid dimensions.");
114
115 if (B.rows() != n || B.cols() != m)
116 throw std::invalid_argument("B has invalid dimensions.");
117
118 if (C.rows() != m || C.cols() != n)
119 throw std::invalid_argument("C has invalid dimensions.");
120
121 if (D.rows() != m || D.cols() != m)
122 throw std::invalid_argument("D has invalid dimensions.");
123
124 if (E.rows() != n || E.cols() != 1)
125 throw std::invalid_argument("E has invalid dimensions.");
126
127 if (F.rows() != m || F.cols() != 1)
128 throw std::invalid_argument("F has invalid dimensions.");
129
130 mA = A;
131 mB = B;
132 mC = C;
133 mD = D;
134 mE = E;
135 mF = F;
136
137 **mX = Matrix::Zero(n, 1);
138
139 mdA = Matrix::Zero(n, n);
140 mdB = Matrix::Zero(n, m);
141 mdE = Matrix::Zero(n, 1);
142
143 mW = MatrixComp::Zero(mPhaseCount, mPhaseCount);
144 mYHist = MatrixComp::Zero(mPhaseCount, 1);
145
146 mParametersSet = true;
147}

176void DP::Ph3::MixedVTypeVariableSSNComp::updateState(const MatrixComp &uOld,
177 const MatrixComp &uNew) {
178 const Matrix uOldReal = packComplex(uOld);
179 const Matrix uNewReal = packComplex(uNew);
180 **mX = mdA * (**mX) + mdB * (uNewReal + uOldReal) + mdE;

183void DP::Ph3::MixedVTypeVariableSSNComp::recomputeDiscreteModel() {
184 Math::calculateStateSpaceTrapezoidalMatrices(mA, mB, mE, mTimeStep, mdA, mdB,
185 mdE);
186
187 // Fold the 2m x 2m real u->y block into a 3x3 complex admittance; each 2x2
188 // block is required to have the [[a,-b],[b,a]] complex-multiplication form.
189 mW = foldComplexMatrix(mC * mdB + mD);
190}

206void DP::Ph3::MixedVTypeVariableSSNComp::initializeFromNodesAndTerminals(
207 Real frequency) {
208 if (!mParametersSet)
209 throw std::logic_error("setParameters() must be called before "
210 "initializeFromNodesAndTerminals().");
211
212 if (mRealStateCount != 0)
213 throw std::logic_error(
214 "The default initializeFromNodesAndTerminals() only supports a "
215 "purely complex-envelope state (realStateCount() == 0); components "
216 "with real control states must override it.");
217
218 const MatrixComp uInit = buildInitialInputFromNodes(frequency);
219
220 **mIntfVoltage = uInit;
221 **mX = Matrix::Zero(stateSize(), 1);
222
223 SPDLOG_LOGGER_INFO(mSLog,
224 "\n--- Initialization from nodes and terminals ---");
225
226 Bool converged = false;
227
228 const Matrix uColumn = packComplex(uInit);
229
230 for (Int iter = 0; iter < mInitializationMaxIterations; ++iter) {
231 const Matrix xPrev = **mX;
232
233 updateComponentParameters();
234
235 // mA is already carrier-shifted, so steady state is 0 = mA*x + mB*u + mE.
236 **mX = -mA.inverse() * (mB * uColumn + mE);
237
238 if ((**mX).isApprox(xPrev, mInitializationTolerance)) {
239 converged = true;
240 break;
241 }
242 }
243
244 if (!converged) {
245 SPDLOG_LOGGER_WARN(
246 mSLog,
247 "Fixed-point initialization did not converge within {} iterations.",
248 mInitializationMaxIterations);
249 }
250
251 updateComponentParameters();
252
253 const Matrix yReal = mC * (**mX) + mD * uColumn + mF;
254 **mIntfCurrent = unpackComplex(yReal);
255
256 mParameterChanged = false;
257
258 SPDLOG_LOGGER_INFO(
259 mSLog,
260 "\nInput u: {:s}"
261 "\nOutput y: {:s}"
262 "\nState x: {:s}"
263 "\n--- Initialization from nodes and terminals finished ---",
264 Logger::matrixCompToString(**mIntfVoltage),
265 Logger::matrixCompToString(**mIntfCurrent), Logger::matrixToString(**mX));
266}

315void DP::Ph3::MixedVTypeVariableSSNComp::mnaCompApplyRightSideVectorStamp(
316 Matrix &rightVector) {
317 if (terminalNotGrounded(0)) {
318 for (Int phase = 0; phase < mPhaseCount; ++phase)
319 Math::setVectorElement(rightVector, matrixNodeIndex(0, phase),
320 mYHist(phase, 0));
321 }
322 if (terminalNotGrounded(1)) {
323 for (Int phase = 0; phase < mPhaseCount; ++phase)
324 Math::setVectorElement(rightVector, matrixNodeIndex(1, phase),
325 -mYHist(phase, 0));
326 }
327}

329void DP::Ph3::MixedVTypeVariableSSNComp::mnaCompUpdateVoltage(
330 const Matrix &leftVector) {
331 **mIntfVoltage = MatrixComp::Zero(mPhaseCount, 1);
332
333 if (terminalNotGrounded(1)) {
334 for (Int phase = 0; phase < mPhaseCount; ++phase)
335 (**mIntfVoltage)(phase, 0) =
336 Math::complexFromVectorElement(leftVector, matrixNodeIndex(1, phase));
337 }
338 if (terminalNotGrounded(0)) {
339 for (Int phase = 0; phase < mPhaseCount; ++phase)
340 (**mIntfVoltage)(phase, 0) -=
341 Math::complexFromVectorElement(leftVector, matrixNodeIndex(0, phase));
342 }
343}
