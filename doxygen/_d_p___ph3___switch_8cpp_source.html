23void DP::Ph3::Switch::initializeFromNodesAndTerminals(Real frequency) {
24 mTerminals[0]->setPhaseType(PhaseType::ABC);
25 mTerminals[1]->setPhaseType(PhaseType::ABC);
26
27 Matrix impedance = (**mIsClosed) ? **mClosedResistance : **mOpenResistance;
28 // Voltage across component is defined as V1 - V0.
29 **mIntfVoltage = initialVoltage(1) - initialVoltage(0);
30 **mIntfCurrent = impedance.inverse() * **mIntfVoltage;
31
32 SPDLOG_LOGGER_INFO(mSLog,
33 "\n--- Initialization from powerflow ---"
34 "\nVoltage across phasor: \n{:s}"
35 "\nCurrent phasor: \n{:s}"
36 "\nTerminal 0 voltage phasor: \n{:s}"
37 "\nTerminal 1 voltage phasor: \n{:s}"
38 "\n--- Initialization from powerflow finished ---",
39 Logger::phasorMatrixToString(**mIntfVoltage),
40 Logger::phasorMatrixToString(**mIntfCurrent),
41 Logger::phasorMatrixToString(initialVoltage(0)),
42 Logger::phasorMatrixToString(initialVoltage(1)));
43}

91void DP::Ph3::Switch::mnaCompUpdateVoltage(const Matrix &leftVector) {
92 // Voltage across component is defined as V1 - V0.
93 **mIntfVoltage = MatrixComp::Zero(3, 1);
94 if (terminalNotGrounded(1)) {
95 (**mIntfVoltage)(0, 0) =
96 Math::complexFromVectorElement(leftVector, matrixNodeIndex(1, 0));
97 (**mIntfVoltage)(1, 0) =
98 Math::complexFromVectorElement(leftVector, matrixNodeIndex(1, 1));
99 (**mIntfVoltage)(2, 0) =
100 Math::complexFromVectorElement(leftVector, matrixNodeIndex(1, 2));
101 }
102 if (terminalNotGrounded(0)) {
103 (**mIntfVoltage)(0, 0) =
104 (**mIntfVoltage)(0, 0) -
105 Math::complexFromVectorElement(leftVector, matrixNodeIndex(0, 0));
106 (**mIntfVoltage)(1, 0) =
107 (**mIntfVoltage)(1, 0) -
108 Math::complexFromVectorElement(leftVector, matrixNodeIndex(0, 1));
109 (**mIntfVoltage)(2, 0) =
110 (**mIntfVoltage)(2, 0) -
111 Math::complexFromVectorElement(leftVector, matrixNodeIndex(0, 2));
112 }
113}
