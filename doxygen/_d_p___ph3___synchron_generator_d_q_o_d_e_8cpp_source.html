48void DP::Ph3::SynchronGeneratorDQODE::odePreStep() {
49 for (int i = 0; i < mDim - 2; i++)
50 (**mOdePreState)(i) = mPsisr(i, 0);
51
52 (**mOdePreState)(mDim - 2) = mThetaMech;
53 (**mOdePreState)(mDim - 1) = **mOmMech;

65void DP::Ph3::SynchronGeneratorDQODE::odePostStep() {
66 for (int i = 0; i < mDim - 2; i++)
67 mPsisr(i, 0) = (**mOdePostState)(i);
68
69 mThetaMech = (**mOdePostState)(mDim - 2);
70 **mOmMech = (**mOdePostState)(mDim - 1);
71
72 //copied from stepInPerUnit ; makes only sense to call this after write back
73 mIsr = mFluxToCurrentMat * mPsisr;
74
75 mIdq0(0, 0) = mIsr(0, 0);
76 mIdq0(1, 0) = mIsr(3, 0);
77 mIdq0(2, 0) = mIsr(6, 0);
78 **mIntfCurrent = mBase_I * dq0ToAbcTransform(mThetaMech, mIdq0);
79
80 SPDLOG_LOGGER_DEBUG(mSLog, "\nCurrent: \n{:s}",
81 Logger::matrixCompToString(**mIntfCurrent));
82}

84// ODE-Class simulation state-space
85void DP::Ph3::SynchronGeneratorDQODE::odeStateSpace(double t, const double y[],
86 double ydot[]) {
87 if (mDim == 9) { //flux-order: ds, fd, kd, qs, kq1, (kq2,) 0s
88 //multiply voltages with mBase_OmElec (similar to stepInPerUnit):
89 ydot[0] = mBase_OmElec * (mVdq0(0, 0) + y[8] * y[3]); //v_d+omega*lambda_q
90 ydot[1] = mBase_OmElec * mVsr(1, 0); //v_fd
91 ydot[2] = 0;
92 ydot[3] = mBase_OmElec * (mVdq0(1, 0) - y[8] * y[0]); //v_q-omega*lambda_d
93 ydot[4] = 0;
94 ydot[5] = 0;
95 ydot[6] = mBase_OmElec * mVdq0(2, 0); //v_0
96 } else {
97 ydot[0] = mBase_OmElec * (mVdq0(0, 0) + y[8] * y[3]); //v_d+omega*lambda_q
98 ydot[1] = mBase_OmElec * mVsr(1, 0); //v_fd
99 ydot[2] = 0;
100 ydot[3] = mBase_OmElec * (mVdq0(1, 0) - y[8] * y[0]); //v_q-omega*lambda_d
101 ydot[4] = 0;
102 ydot[5] = mBase_OmElec * mVdq0(2, 0); //v_0
103 }
104
105 // Matrix Vector-Mult:
106 for (int i = 0; i < mDim - 2; i++)
107 for (int j = 0; j < mDim - 2; j++)
108 ydot[i] += mBase_OmElec * mFluxStateSpaceMat(i, j) * y[j];
109
110 // Mechanical equations:
111
112 // Compute theta with transformation (following stepInPerUnit):
113 ydot[mDim - 2] = (y[mDim - 1] - 1) * mBase_OmMech;
114
115 /* Auxiliary variables to compute
116 * T_e= lambda_d * i_q - lambda_q * i_d
117 * needed for omega: */
118 realtype i_d = 0, i_q = 0;
119 // Compute new currents (depending on updated fluxes)
120 for (int i = 0; i < mDim - 2; i++) {
121 i_d += mFluxToCurrentMat(0, i) * y[i];
122 i_q += mFluxToCurrentMat(3, i) * y[i];
123 }
124
125 // Compute Omega (according to stepInPerUnit)
126 ydot[mDim - 1] =
127 1 / (2 * **mInertia) * (**mMechTorque - (y[3] * i_d - y[0] * i_q));
128}

130void DP::Ph3::SynchronGeneratorDQODE::odeJacobian(double t, const double y[],
131 double fy[], double J[],
132 double tmp1[], double tmp2[],
133 double tmp3[]) {
134 // Fill Jacobian:
135 // Fluxes:
136 for (int i = 0; i < mDim - 2; i++)
137 for (int j = 0; j < mDim - 2; j++)
138 J[i * (mDim - 2) + j] =
139 mBase_OmElec *
140 mFluxStateSpaceMat(i, j); //Note: J is row-major ordered
141
142 //omega in flux eqs.:
143 J[0 * (mDim - 2) + mDim - 1] = mBase_OmElec * y[3];
144 J[3 * (mDim - 2) + mDim - 1] = mBase_OmElec * (-1) * y[0];
145 //theta-row:
146 J[(mDim - 2) * (mDim - 2) + mDim - 1] = 1 * mBase_OmMech;
147
148 realtype i_d = 0, i_q = 0;
149
150 for (int i = 0; i < mDim - 2; i++) {
151 i_d += mFluxToCurrentMat(0, i) * y[i];
152 i_q += mFluxToCurrentMat(3, i) * y[i];
153 }
154 //fluxes show up in omega eq. (with mBase_OmMech ?)
155 J[(mDim - 1) * (mDim - 2) + 0] = 1 / (2 * **mInertia) * i_q;
156 J[(mDim - 1) * (mDim - 2) + 3] = -1 / (2 * **mInertia) * i_d;
157}
