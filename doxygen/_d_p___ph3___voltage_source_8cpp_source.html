40void DP::Ph3::VoltageSource::mnaCompInitialize(
41 Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
42 updateMatrixNodeIndices();
43 (**mIntfVoltage)(0, 0) = **mVoltageRef;
44 (**mIntfVoltage)(1, 0) =
45 Complex(Math::abs(**mVoltageRef) *
46 cos(Math::phase(**mVoltageRef) - 2. / 3. * M_PI),
47 Math::abs(**mVoltageRef) *
48 sin(Math::phase(**mVoltageRef) - 2. / 3. * M_PI));
49 (**mIntfVoltage)(2, 0) =
50 Complex(Math::abs(**mVoltageRef) *
51 cos(Math::phase(**mVoltageRef) + 2. / 3. * M_PI),
52 Math::abs(**mVoltageRef) *
53 sin(Math::phase(**mVoltageRef) + 2. / 3. * M_PI));
54}

56void DP::Ph3::VoltageSource::mnaCompApplySystemMatrixStamp(
57 SparseMatrixRow &systemMatrix) {
58 if (terminalNotGrounded(0)) {
59 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 0),
60 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
61 Complex(-1, 0));
62 Math::addToMatrixElement(systemMatrix,
63 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
64 matrixNodeIndex(0, 0), Complex(-1, 0));
65
66 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 1),
67 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
68 Complex(-1, 0));
69 Math::addToMatrixElement(systemMatrix,
70 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
71 matrixNodeIndex(0, 1), Complex(-1, 0));
72
73 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 2),
74 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
75 Complex(-1, 0));
76 Math::addToMatrixElement(systemMatrix,
77 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
78 matrixNodeIndex(0, 2), Complex(-1, 0));
79 }
80 if (terminalNotGrounded(1)) {
81 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 0),
82 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
83 Complex(1, 0));
84 Math::addToMatrixElement(systemMatrix,
85 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
86 matrixNodeIndex(1, 0), Complex(1, 0));
87
88 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 1),
89 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
90 Complex(1, 0));
91 Math::addToMatrixElement(systemMatrix,
92 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
93 matrixNodeIndex(1, 1), Complex(1, 0));
94
95 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 2),
96 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
97 Complex(1, 0));
98 Math::addToMatrixElement(systemMatrix,
99 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
100 matrixNodeIndex(1, 2), Complex(1, 0));
101 }
102}
