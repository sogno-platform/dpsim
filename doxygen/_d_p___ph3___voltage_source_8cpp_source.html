40 void DP::Ph3::VoltageSource::mnaCompInitialize(
41  Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
42  updateMatrixNodeIndices();
43  (**mIntfVoltage)(0, 0) = **mVoltageRef;
44  (**mIntfVoltage)(1, 0) =
45  Complex(Math::abs(**mVoltageRef) *
46  cos(Math::phase(**mVoltageRef) - 2. / 3. * M_PI),

51  cos(Math::phase(**mVoltageRef) + 2. / 3. * M_PI),

54 }

56 void DP::Ph3::VoltageSource::mnaCompApplySystemMatrixStamp(

66  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 1),

68  Complex(-1, 0));
69  Math::addToMatrixElement(systemMatrix,
70  mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
71  matrixNodeIndex(0, 1), Complex(-1, 0));
72 
73  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 2),
74  mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),

81  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 0),

88  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 1),
89  mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
90  Complex(1, 0));

92  mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
93  matrixNodeIndex(1, 1), Complex(1, 0));

95  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 2),
