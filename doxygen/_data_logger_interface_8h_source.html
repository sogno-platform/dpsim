36  virtual void logAttribute(const String &name, CPS::AttributeBase::Ptr attr,
37  UInt rowsMax = 0, UInt colsMax = 0) {
38  if (auto attrReal =
39  std::dynamic_pointer_cast<CPS::Attribute<Real>>(attr.getPtr())) {
40  mAttributes[name] = attrReal;
41  } else if (auto attrComp =
42  std::dynamic_pointer_cast<CPS::Attribute<Complex>>(
43  attr.getPtr())) {
44  mAttributes[name + ".re"] = attrComp->deriveReal();
45  mAttributes[name + ".im"] = attrComp->deriveImag();
46  } else if (auto attrInt = std::dynamic_pointer_cast<CPS::Attribute<Int>>(
47  attr.getPtr())) {
48  mAttributes[name] = attrInt;
49  } else if (auto attrMatrix =
50  std::dynamic_pointer_cast<CPS::Attribute<Matrix>>(
51  attr.getPtr())) {
52  UInt rows = static_cast<UInt>((**attrMatrix).rows());
53  UInt cols = static_cast<UInt>((**attrMatrix).cols());
54  if (rowsMax == 0 || rowsMax > rows)
55  rowsMax = rows;
56  if (colsMax == 0 || colsMax > cols)
57  colsMax = cols;
58  if (rows == 1 && cols == 1) {
59  mAttributes[name] = attrMatrix->deriveCoeff<Real>(0, 0);
60  } else if (cols == 1) {
61  for (UInt k = 0; k < rowsMax; ++k) {
62  mAttributes[name + "_" + std::to_string(k)] =
63  attrMatrix->deriveCoeff<Real>(k, 0);
64  }
65  } else {
66  for (UInt k = 0; k < rowsMax; ++k) {
67  for (UInt l = 0; l < colsMax; ++l) {
68  mAttributes[name + "_" + std::to_string(k) + "_" +
69  std::to_string(l)] =
70  attrMatrix->deriveCoeff<Real>(k, l);
71  }
72  }
73  }
74  } else if (auto attrMatrix =
75  std::dynamic_pointer_cast<CPS::Attribute<MatrixComp>>(
76  attr.getPtr())) {
77  UInt rows = static_cast<UInt>((**attrMatrix).rows());
78  UInt cols = static_cast<UInt>((**attrMatrix).cols());
79  if (rowsMax == 0 || rowsMax > rows)
80  rowsMax = rows;
81  if (colsMax == 0 || colsMax > cols)
82  colsMax = cols;
83  if (rows == 1 && cols == 1) {
84  mAttributes[name + ".re"] =
85  attrMatrix->deriveCoeff<Complex>(0, 0)->deriveReal();
86  mAttributes[name + ".im"] =
87  attrMatrix->deriveCoeff<Complex>(0, 0)->deriveImag();
88  } else if (cols == 1) {
89  for (UInt k = 0; k < rowsMax; ++k) {
90  mAttributes[name + "_" + std::to_string(k) + ".re"] =
91  attrMatrix->deriveCoeff<Complex>(k, 0)->deriveReal();
92  mAttributes[name + "_" + std::to_string(k) + ".im"] =
93  attrMatrix->deriveCoeff<Complex>(k, 0)->deriveImag();
94  }
95  } else {
96  for (UInt k = 0; k < rowsMax; ++k) {
97  for (UInt l = 0; l < colsMax; ++l) {
98  mAttributes[name + "_" + std::to_string(k) + "_" +
99  std::to_string(l) + ".re"] =
100  attrMatrix->deriveCoeff<Complex>(k, l)->deriveReal();
101  mAttributes[name + "_" + std::to_string(k) + "_" +
102  std::to_string(l) + ".im"] =
103  attrMatrix->deriveCoeff<Complex>(k, l)->deriveImag();
104  }
105  }
106  }
107  } else {
108  throw std::runtime_error(
109  "DataLoggerInterface: Unknown attribute type for attribute " + name);
110  }
111  }

115  void logAttribute(const std::vector<String> &name,
116  CPS::AttributeBase::Ptr attr) {
117  if (auto attrMatrix =
118  std::dynamic_pointer_cast<CPS::Attribute<Matrix>>(attr.getPtr())) {
119  if ((**attrMatrix).rows() == 1 && (**attrMatrix).cols() == 1) {
120  logAttribute(name[0], attrMatrix->deriveCoeff<CPS::Real>(0, 0));
121  } else if ((**attrMatrix).cols() == 1) {
122  for (UInt k = 0; k < (**attrMatrix).rows(); ++k) {
123  logAttribute(name[k], attrMatrix->deriveCoeff<CPS::Real>(k, 0));
124  }
125  } else {
126  for (UInt k = 0; k < (**attrMatrix).rows(); ++k) {
127  for (UInt l = 0; l < (**attrMatrix).cols(); ++l) {
128  logAttribute(name[k * (**attrMatrix).cols() + l],
129  attrMatrix->deriveCoeff<CPS::Real>(k, l));
130  }
131  }
132  }
133  } else if (auto attrMatrix =
134  std::dynamic_pointer_cast<CPS::Attribute<MatrixComp>>(
135  attr.getPtr())) {
136  if ((**attrMatrix).rows() == 1 && (**attrMatrix).cols() == 1) {
137  logAttribute(name[0], attrMatrix->deriveCoeff<CPS::Complex>(0, 0));
138  } else if ((**attrMatrix).cols() == 1) {
139  for (UInt k = 0; k < (**attrMatrix).rows(); ++k) {
140  logAttribute(name[k], attrMatrix->deriveCoeff<CPS::Complex>(k, 0));
141  }
142  } else {
143  for (UInt k = 0; k < (**attrMatrix).rows(); ++k) {
144  for (UInt l = 0; l < (**attrMatrix).cols(); ++l) {
145  logAttribute(name[k * (**attrMatrix).cols() + l],
146  attrMatrix->deriveCoeff<CPS::Complex>(k, l));
147  }
148  }
149  }
150  }
