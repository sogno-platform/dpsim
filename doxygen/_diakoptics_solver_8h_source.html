116 SubnetSolveTask(DiakopticsSolver<VarType> &solver, UInt net)
117 : Task(solver.mName + ".SubnetSolve_" + std::to_string(net)),
118 mSolver(solver), mSubnet(solver.mSubnets[net]) {
119 for (auto it : mSubnet.components) {
