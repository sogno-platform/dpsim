46void EMT::Ph1::VoltageSource::mnaCompApplySystemMatrixStamp(
47 SparseMatrixRow &systemMatrix) {
48 if (terminalNotGrounded(0)) {
49 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0),
50 mVirtualNodes[0]->matrixNodeIndex(), -1);
51 Math::addToMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
52 matrixNodeIndex(0), -1);
53 }
54 if (terminalNotGrounded(1)) {
55 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1),
56 mVirtualNodes[0]->matrixNodeIndex(), 1);
57 Math::addToMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
58 matrixNodeIndex(1), 1);
59 }
60
61 if (terminalNotGrounded(0)) {
62 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", -1.,
63 matrixNodeIndex(0), mVirtualNodes[0]->matrixNodeIndex());
64 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", -1.,
65 mVirtualNodes[0]->matrixNodeIndex(), matrixNodeIndex(0));
66 }
67 if (terminalNotGrounded(1)) {
68 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
69 matrixNodeIndex(1), mVirtualNodes[0]->matrixNodeIndex());
70 SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
71 mVirtualNodes[0]->matrixNodeIndex(), matrixNodeIndex(1));
72 }
73}

81void EMT::Ph1::VoltageSource::updateVoltage(Real time) {
82 Complex voltageRef = mVoltageRef->get();
83 Real srcFreq = mSrcFreq->get();
84 if (srcFreq > 0)
85 (**mIntfVoltage)(0, 0) =
86 Math::abs(voltageRef) *
87 cos((time) * 2. * PI * srcFreq + Math::phase(voltageRef));
88 else
89 (**mIntfVoltage)(0, 0) = voltageRef.real();
90}
