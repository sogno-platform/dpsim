46 void EMT::Ph1::VoltageSource::mnaCompApplySystemMatrixStamp(

63  matrixNodeIndex(0), mVirtualNodes[0]->matrixNodeIndex());

65  mVirtualNodes[0]->matrixNodeIndex(), matrixNodeIndex(0));
66  }

68  SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
69  matrixNodeIndex(1), mVirtualNodes[0]->matrixNodeIndex());

81 void EMT::Ph1::VoltageSource::updateVoltage(Real time) {
82  Complex voltageRef = mVoltageRef->get();
83  Real srcFreq = mSrcFreq->get();
84  if (srcFreq > 0)
85  (**mIntfVoltage)(0, 0) =
86  Math::abs(voltageRef) *
87  cos((time) * 2. * PI * srcFreq + Math::phase(voltageRef));
88  else
89  (**mIntfVoltage)(0, 0) = voltageRef.real();
90 }
