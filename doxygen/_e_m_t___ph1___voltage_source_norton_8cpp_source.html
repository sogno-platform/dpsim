46 void EMT::Ph1::VoltageSourceNorton::mnaCompApplySystemMatrixStamp(
47  SparseMatrixRow &systemMatrix) {
48  // Apply matrix stamp for equivalent resistance
49  if (terminalNotGrounded(0))
50  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0),
51  matrixNodeIndex(0), mConductance);
52  if (terminalNotGrounded(1))
53  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1),
54  matrixNodeIndex(1), mConductance);
55  if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
56  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0),
57  matrixNodeIndex(1), -mConductance);
58  Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1),
59  matrixNodeIndex(0), -mConductance);
60  }
61 }

72 void EMT::Ph1::VoltageSourceNorton::updateState(Real time) {
73  // Check if set source was called
74  if (Math::abs(**mVoltageRef) > 0)
75  (**mIntfVoltage)(0, 0) =
76  Math::abs(**mVoltageRef) *
77  cos(2. * PI * **mSrcFreq * time + Math::phase(**mVoltageRef));
78 
79  mEquivCurrent = (**mIntfVoltage)(0, 0) / **mResistance;
80 }

112 void EMT::Ph1::VoltageSourceNorton::mnaCompUpdateVoltage(
113  const Matrix &leftVector) {
114  // Calculate v1 - v0
115  if (terminalNotGrounded(1))
116  (**mIntfVoltage)(0, 0) =
117  Math::realFromVectorElement(leftVector, matrixNodeIndex(1));
118  if (terminalNotGrounded(0))
