64void EMT::Ph1::VoltageSourceRamp::updateState(Real time) {
65 Real voltageAbs = Math::abs(**mVoltageRef);
66 Real voltagePhase = Math::phase(**mVoltageRef);
67 (**mIntfVoltage)(0, 0) = voltageAbs * cos(voltagePhase + **mSrcFreq * time);
68
69 if (time >= mSwitchTime && time < mSwitchTime + mRampTime) {
70 voltageAbs = Math::abs(**mVoltageRef +
71 (time - mSwitchTime) / mRampTime * mAddVoltage);
72 voltagePhase = Math::phase(**mVoltageRef +
73 (time - mSwitchTime) / mRampTime * mAddVoltage);
74 Real fadeInOut =
75 0.5 + 0.5 * sin((time - mSwitchTime) / mRampTime * PI + -PI / 2);
76 (**mIntfVoltage)(0, 0) =
77 voltageAbs *
78 cos(voltagePhase + (**mSrcFreq + fadeInOut * mAddSrcFreq) * time);
79 } else if (time >= mSwitchTime + mRampTime) {
80 voltageAbs = Math::abs(**mVoltageRef + mAddVoltage);
81 voltagePhase = Math::phase(**mVoltageRef + mAddVoltage);
82 (**mIntfVoltage)(0, 0) =
83 voltageAbs * cos(voltagePhase + (**mSrcFreq + mAddSrcFreq) * time);
84 }
85}
