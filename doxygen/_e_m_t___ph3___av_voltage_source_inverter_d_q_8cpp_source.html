199 void EMT::Ph3::AvVoltageSourceInverterDQ::initializeFromNodesAndTerminals(
200  Real frequency) {
201 
202  // use complex interface quantities for initialization calculations
203  MatrixComp intfVoltageComplex = Matrix::Zero(3, 1);
204  MatrixComp intfCurrentComplex = Matrix::Zero(3, 1);
205 
206  // derive complex threephase initialization from single phase initial values (only valid for balanced systems)
207  intfVoltageComplex(0, 0) = RMS3PH_TO_PEAK1PH * initialSingleVoltage(0);
208  intfVoltageComplex(1, 0) = intfVoltageComplex(0, 0) * SHIFT_TO_PHASE_B;
209  intfVoltageComplex(2, 0) = intfVoltageComplex(0, 0) * SHIFT_TO_PHASE_C;
210  intfCurrentComplex(0, 0) = -std::conj(2. / 3. * Complex(**mPref, **mQref) /
211  intfVoltageComplex(0, 0));
212  intfCurrentComplex(1, 0) = intfCurrentComplex(0, 0) * SHIFT_TO_PHASE_B;
213  intfCurrentComplex(2, 0) = intfCurrentComplex(0, 0) * SHIFT_TO_PHASE_C;
214 
215  MatrixComp filterInterfaceInitialVoltage = MatrixComp::Zero(3, 1);
216  MatrixComp filterInterfaceInitialCurrent = MatrixComp::Zero(3, 1);
217  if (mWithConnectionTransformer) {
218  // calculate quantities of low voltage side of transformer (being the interface quantities of the filter, calculations only valid for symmetrical systems)
219  filterInterfaceInitialVoltage =
220  (intfVoltageComplex -
221  Complex(mTransformerResistance, mTransformerInductance * **mOmegaN) *
222  intfCurrentComplex) /
223  Complex(mTransformerRatioAbs, mTransformerRatioPhase);
224  filterInterfaceInitialCurrent =
225  intfCurrentComplex *
226  Complex(mTransformerRatioAbs, mTransformerRatioPhase);
227 
228  // connect and init transformer
229  mVirtualNodes[3]->setInitialVoltage(PEAK1PH_TO_RMS3PH *
230  filterInterfaceInitialVoltage);
231  mConnectionTransformer->connect({mTerminals[0]->node(), mVirtualNodes[3]});
232  } else {
233  // if no transformer used, filter interface equal to inverter interface
234  filterInterfaceInitialVoltage = intfVoltageComplex;
235  filterInterfaceInitialCurrent = intfCurrentComplex;
236  }
237 
238  // derive initialization quantities of filter (calculations only valid for symmetrical systems)
239  MatrixComp vcInit = filterInterfaceInitialVoltage -
240  filterInterfaceInitialCurrent * Complex(mRc, 0);
241  MatrixComp icfInit = vcInit * Complex(0., 2. * PI * frequency * mCf);
242  MatrixComp vfInit = vcInit - (filterInterfaceInitialCurrent - icfInit) *
243  Complex(0., 2. * PI * frequency * mLf);
244  MatrixComp vsInit =
245  vfInit - (filterInterfaceInitialCurrent - icfInit) * Complex(mRf, 0);
246  mVirtualNodes[0]->setInitialVoltage(PEAK1PH_TO_RMS3PH * vsInit);
247  mVirtualNodes[1]->setInitialVoltage(PEAK1PH_TO_RMS3PH * vfInit);
248  mVirtualNodes[2]->setInitialVoltage(PEAK1PH_TO_RMS3PH * vcInit);
249 
250  // save real interface quantities calculated from complex ones
251  **mIntfVoltage = intfVoltageComplex.real();
252  **mIntfCurrent = intfCurrentComplex.real();
253 
254  // Initialize controlled source
255  mSubCtrledVoltageSource->setParameters(mVirtualNodes[0]->initialVoltage(),
256  0.0);
257 
258  // Connect electrical subcomponents
259  mSubCtrledVoltageSource->connect({SimNode::GND, mVirtualNodes[0]});
260  mSubResistorF->connect({mVirtualNodes[0], mVirtualNodes[1]});
261  mSubInductorF->connect({mVirtualNodes[1], mVirtualNodes[2]});
262  mSubCapacitorF->connect({mVirtualNodes[2], SimNode::GND});
263  if (mWithConnectionTransformer)
264  mSubResistorC->connect({mVirtualNodes[2], mVirtualNodes[3]});
265  else
266  mSubResistorC->connect({mVirtualNodes[2], mTerminals[0]->node()});
267 
268  // Initialize electrical subcomponents
269  for (auto subcomp : mSubComponents) {
270  subcomp->initialize(mFrequencies);
271  subcomp->initializeFromNodesAndTerminals(frequency);
272  }
273 
274  // Initialize control subcomponents
275  // current and voltage inputs to PLL and power controller
276  Matrix vcdq, ircdq;
277  Real theta = std::arg(mVirtualNodes[3]->initialSingleVoltage());
278  vcdq =
279  parkTransformPowerInvariant(theta, filterInterfaceInitialVoltage.real());
280  ircdq = parkTransformPowerInvariant(
281  theta, -1 * filterInterfaceInitialCurrent.real());
282 
283  **mVcd = vcdq(0, 0);
284  **mVcq = vcdq(1, 0);
285  **mIrcd = ircdq(0, 0);
286  **mIrcq = ircdq(1, 0);
287 
288  // angle input
289  Matrix matrixStateInit = Matrix::Zero(2, 1);
290  Matrix matrixOutputInit = Matrix::Zero(2, 1);
291  matrixStateInit(0, 0) = std::arg(mVirtualNodes[3]->initialSingleVoltage());
292  matrixOutputInit(0, 0) = std::arg(mVirtualNodes[3]->initialSingleVoltage());
293  mPLL->setInitialValues(**mVcq, matrixStateInit, matrixOutputInit);
294 
295  SPDLOG_LOGGER_INFO(
296  mSLog,
297  "\n--- Initialization from powerflow ---"
298  "\nInterface voltage across: {:s}"
299  "\nInterface current: {:s}"
300  "\nTerminal 0 initial voltage: {:s}"
301  "\nTerminal 0 connected to {:s} = sim node {:d}"
302  "\nVirtual node 0 initial voltage: {:s}"
303  "\nVirtual node 1 initial voltage: {:s}"
304  "\nVirtual node 2 initial voltage: {:s}",
305  Logger::phasorToString(intfVoltageComplex(0, 0)),
306  Logger::phasorToString(intfCurrentComplex(0, 0)),
307  Logger::phasorToString(initialSingleVoltage(0)),
308  mTerminals[0]->node()->name(), mTerminals[0]->node()->matrixNodeIndex(),
309  Logger::phasorToString(mVirtualNodes[0]->initialSingleVoltage()),
310  Logger::phasorToString(mVirtualNodes[1]->initialSingleVoltage()),
311  Logger::phasorToString(mVirtualNodes[2]->initialSingleVoltage()));
312  if (mWithConnectionTransformer)
313  SPDLOG_LOGGER_INFO(
314  mSLog, "\nVirtual node 3 initial voltage: {:s}",
315  Logger::phasorToString(mVirtualNodes[3]->initialSingleVoltage()));
316  SPDLOG_LOGGER_INFO(mSLog, "\n--- Initialization from powerflow finished ---");
317 }

365 Matrix EMT::Ph3::AvVoltageSourceInverterDQ::parkTransformPowerInvariant(
366  Real theta, const Matrix &fabc) {
367  // Calculates fdq = Tdq * fabc
368  // Assumes that d-axis starts aligned with phase a
369  Matrix Tdq = getParkTransformMatrixPowerInvariant(theta);
370  Matrix dqvector = Tdq * fabc;
371  return dqvector;
372 }

374 Matrix
375 EMT::Ph3::AvVoltageSourceInverterDQ::getParkTransformMatrixPowerInvariant(
376  Real theta) {
377  // Return park matrix for theta
378  // Assumes that d-axis starts aligned with phase a
379  Matrix Tdq = Matrix::Zero(2, 3);
380  Real k = sqrt(2. / 3.);
381  Tdq << k * cos(theta), k * cos(theta - 2. * M_PI / 3.),
382  k * cos(theta + 2. * M_PI / 3.), -k * sin(theta),
383  -k * sin(theta - 2. * M_PI / 3.), -k * sin(theta + 2. * M_PI / 3.);
384  return Tdq;
385 }

387 Matrix EMT::Ph3::AvVoltageSourceInverterDQ::inverseParkTransformPowerInvariant(
388  Real theta, const Matrix &fdq) {
389  // Calculates fabc = Tabc * fdq
390  // with d-axis starts aligned with phase a
391  Matrix Tabc = getInverseParkTransformMatrixPowerInvariant(theta);
392  Matrix fabc = Tabc * fdq;
393  return fabc;

396 Matrix EMT::Ph3::AvVoltageSourceInverterDQ::
397  getInverseParkTransformMatrixPowerInvariant(Real theta) {
398  // Return inverse park matrix for theta
400  Matrix Tabc = Matrix::Zero(3, 2);
401  Real k = sqrt(2. / 3.);
402  Tabc << k * cos(theta), -k * sin(theta), k * cos(theta - 2. * M_PI / 3.),
403  -k * sin(theta - 2. * M_PI / 3.), k * cos(theta + 2. * M_PI / 3.),
404  -k * sin(theta + 2. * M_PI / 3.);
405  return Tabc;
406 }

408 void EMT::Ph3::AvVoltageSourceInverterDQ::controlStep(Real time,
409  Int timeStepCount) {
410  // Transformation interface forward
411  Matrix vcdq, ircdq;
412  Real theta = mPLL->mOutputPrev->get()(0, 0);
413  vcdq = parkTransformPowerInvariant(theta, **mVirtualNodes[3]->mVoltage);
414  ircdq = parkTransformPowerInvariant(theta, -**mSubResistorC->mIntfCurrent);
415 
416  **mVcd = vcdq(0, 0);
417  **mVcq = vcdq(1, 0);
418  **mIrcd = ircdq(0, 0);
419  **mIrcq = ircdq(1, 0);
420 
421  // add step of subcomponents
422  mPLL->signalStep(time, timeStepCount);
423  mPowerControllerVSI->signalStep(time, timeStepCount);
424 
425  // Transformation interface backward
426  **mVsref = inverseParkTransformPowerInvariant(
427  mPLL->mOutputPrev->get()(0, 0), mPowerControllerVSI->mOutputCurr->get());
428  mThetaN = mThetaN + mTimeStep * **mOmegaN;
429 }
