67void EMT::Ph3::ControlledVoltageSource::mnaCompApplySystemMatrixStamp(
68 SparseMatrixRow &systemMatrix) {
69 if (terminalNotGrounded(0)) {
70 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 0),
71 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
72 -1);
73 Math::addToMatrixElement(systemMatrix,
74 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
75 matrixNodeIndex(0, 0), -1);
76
77 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 1),
78 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
79 -1);
80 Math::addToMatrixElement(systemMatrix,
81 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
82 matrixNodeIndex(0, 1), -1);
83
84 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 2),
85 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
86 -1);
87 Math::addToMatrixElement(systemMatrix,
88 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
89 matrixNodeIndex(0, 2), -1);
90 }
91 if (terminalNotGrounded(1)) {
92 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 0),
93 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
94 1);
95 Math::addToMatrixElement(systemMatrix,
96 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
97 matrixNodeIndex(1, 0), 1);
98
99 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 1),
100 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
101 1);
102 Math::addToMatrixElement(systemMatrix,
103 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
104 matrixNodeIndex(1, 1), 1);
105
106 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 2),
107 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
108 1);
109 Math::addToMatrixElement(systemMatrix,
110 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
111 matrixNodeIndex(1, 2), 1);
112 }
113}
