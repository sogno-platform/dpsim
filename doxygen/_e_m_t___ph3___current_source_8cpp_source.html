26 void EMT::Ph3::CurrentSource::initializeFromNodesAndTerminals(Real frequency) {
27  SPDLOG_LOGGER_INFO(
28  mSLog, "\n--- Initialization from node voltages and terminal ---");
29  if (!mParametersSet) {
30  auto srcSigSine =
31  Signal::SineWaveGenerator::make(**mName + "_sw", Logger::Level::off);
32  // Complex(1,0) is used as initialPhasor for signal generator as only phase is used
33  srcSigSine->setParameters(Complex(1, 0), frequency);
34  mSrcSig = srcSigSine;
35 
36  Complex v_ref = initialSingleVoltage(1) - initialSingleVoltage(0);
37  Complex s_ref = terminal(1)->singlePower() - terminal(0)->singlePower();
38 
39  // Current flowing from T1 to T0 (rms value)
40  Complex i_ref = std::conj(s_ref / v_ref / sqrt(3.));
41 
42  **mCurrentRef = CPS::Math::singlePhaseVariableToThreePhase(i_ref);
43  mSrcFreq->setReference(mSrcSig->attributeTyped<Real>("freq"));
44 
45  SPDLOG_LOGGER_INFO(mSLog,
46  "\nReference current: {:s}"
47  "\nReference voltage: {:s}"
48  "\nReference power: {:s}"
49  "\nTerminal 0 voltage: {:s}"
50  "\nTerminal 1 voltage: {:s}"
51  "\nTerminal 0 power: {:s}"
52  "\nTerminal 1 power: {:s}",
53  Logger::phasorToString(i_ref),
54  Logger::phasorToString(v_ref),
55  Logger::complexToString(s_ref),
56  Logger::phasorToString(initialSingleVoltage(0)),
57  Logger::phasorToString(initialSingleVoltage(1)),
58  Logger::complexToString(terminal(0)->singlePower()),
59  Logger::complexToString(terminal(1)->singlePower()));
60  } else {
61  SPDLOG_LOGGER_INFO(
62  mSLog,
63  "\nInitialization from node voltages and terminal omitted (parameter "
64  "already set)."
65  "\nReference voltage: {:s}",
66  Logger::matrixCompToString(attributeTyped<MatrixComp>("I_ref")->get()));
67  }
68  SPDLOG_LOGGER_INFO(
69  mSLog, "\n--- Initialization from node voltages and terminal ---");
70  mSLog->flush();
71 }

105 void EMT::Ph3::CurrentSource::updateCurrent(Real time) {
106  if (mSrcSig != nullptr) {
107  mSrcSig->step(time);
108  for (int i = 0; i < 3; i++) {
109  (**mIntfCurrent)(i, 0) = RMS_TO_PEAK * Math::abs((**mCurrentRef)(i, 0)) *
110  cos(Math::phase(mSrcSig->getSignal()) +
111  Math::phase((**mCurrentRef)(i, 0)));
112  }
113  } else {
114  **mIntfCurrent = RMS_TO_PEAK * (**mCurrentRef).real();
115  }
116  SPDLOG_LOGGER_DEBUG(mSLog, "\nUpdate current: {:s}",
117  Logger::matrixToString(**mIntfCurrent));
118 }

148 void EMT::Ph3::CurrentSource::mnaCompUpdateVoltage(const Matrix &leftVector) {
149  // v1 - v0
150  **mIntfVoltage = Matrix::Zero(3, 1);
151  if (terminalNotGrounded(1)) {
152  (**mIntfVoltage)(0, 0) =
153  Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 0));
154  (**mIntfVoltage)(1, 0) =
155  Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 1));
156  (**mIntfVoltage)(2, 0) =
157  Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 2));
158  }
159  if (terminalNotGrounded(0)) {
160  (**mIntfVoltage)(0, 0) =
161  (**mIntfVoltage)(0, 0) -
162  Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 0));
163  (**mIntfVoltage)(1, 0) =
164  (**mIntfVoltage)(1, 0) -
165  Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 1));
166  (**mIntfVoltage)(2, 0) =
167  (**mIntfVoltage)(2, 0) -
168  Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 2));
169  }
