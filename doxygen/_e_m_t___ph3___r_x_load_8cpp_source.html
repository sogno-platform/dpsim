74 void EMT::Ph3::RXLoad::initializeFromNodesAndTerminals(Real frequency) {
75 
76  if (initPowerFromTerminal) {
77  **mActivePower = Matrix::Zero(3, 3);
78  (**mActivePower)(0, 0) = mTerminals[0]->singleActivePower() / 3.;
79  (**mActivePower)(1, 1) = mTerminals[0]->singleActivePower() / 3.;
80  (**mActivePower)(2, 2) = mTerminals[0]->singleActivePower() / 3.;
81 
82  **mReactivePower = Matrix::Zero(3, 3);
83  (**mReactivePower)(0, 0) = mTerminals[0]->singleReactivePower() / 3.;
84  (**mReactivePower)(1, 1) = mTerminals[0]->singleReactivePower() / 3.;
85  (**mReactivePower)(2, 2) = mTerminals[0]->singleReactivePower() / 3.;
86 
87  // complex power
88  mPower = MatrixComp::Zero(3, 3);
89  mPower(0, 0) = {(**mActivePower)(0, 0), (**mReactivePower)(0, 0)};
90  mPower(1, 1) = {(**mActivePower)(1, 1), (**mReactivePower)(1, 1)};
91  mPower(2, 2) = {(**mActivePower)(2, 2), (**mReactivePower)(2, 2)};
92 
93  **mNomVoltage = std::abs(mTerminals[0]->initialSingleVoltage());
94 
95  SPDLOG_LOGGER_INFO(mSLog,
96  "\nActive Power [W]: {}"
97  "\nReactive Power [VAr]: {}",
98  Logger::matrixToString(**mActivePower),
99  Logger::matrixToString(**mReactivePower));
100  SPDLOG_LOGGER_INFO(mSLog, "Nominal Voltage={} [V]", **mNomVoltage);
101  }
102 
103  MatrixComp vInitABC = MatrixComp::Zero(3, 1);
104  vInitABC(0, 0) = RMS3PH_TO_PEAK1PH * mTerminals[0]->initialSingleVoltage();
105  vInitABC(1, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_B;
106  vInitABC(2, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_C;
107  **mIntfVoltage = vInitABC.real();
108 
109  if ((**mActivePower)(0, 0) != 0) {
110  mResistance = std::pow(**mNomVoltage / sqrt(3), 2) * (**mActivePower).inverse();
111  }
112 
113  if ((**mReactivePower)(0, 0) != 0)
114  mReactance = std::pow(**mNomVoltage / sqrt(3), 2) * (**mReactivePower).inverse();
115  else
116  mReactance = Matrix::Zero(1, 1);
117 
118  if (mReactanceInSeries) {
119  MatrixComp impedance = MatrixComp::Zero(3, 3);
120  impedance << Complex(mResistance(0, 0), mReactance(0, 0)), Complex(mResistance(0, 1), mReactance(0, 1)), Complex(mResistance(0, 2), mReactance(0, 2)), Complex(mResistance(1, 0), mReactance(1, 0)),
121  Complex(mResistance(1, 1), mReactance(1, 1)), Complex(mResistance(1, 2), mReactance(1, 2)), Complex(mResistance(2, 0), mReactance(2, 0)), Complex(mResistance(2, 1), mReactance(2, 1)),
122  Complex(mResistance(2, 2), mReactance(2, 2));
123  **mIntfCurrent = (impedance.inverse() * vInitABC).real();
124 
125  // Initialization of virtual node
126  // Initial voltage of phase B,C is set after A
127  MatrixComp vInitTerm0 = MatrixComp::Zero(3, 1);
128  vInitTerm0(0, 0) = initialSingleVoltage(0);
129  vInitTerm0(1, 0) = vInitTerm0(0, 0) * SHIFT_TO_PHASE_B;
130  vInitTerm0(2, 0) = vInitTerm0(0, 0) * SHIFT_TO_PHASE_C;
131  mVirtualNodes[0]->setInitialVoltage(vInitTerm0 + mResistance * **mIntfCurrent);
132  }
133 
134  if ((**mActivePower)(0, 0) != 0) {
135  mSubResistor = std::make_shared<EMT::Ph3::Resistor>(**mName + "_res", mLogLevel);
136  mSubResistor->setParameters(mResistance);
137  if (mReactanceInSeries) {
138  mSubResistor->connect({mTerminals[0]->node(), mVirtualNodes[0]});
139  } else {
140  mSubResistor->connect({SimNode::GND, mTerminals[0]->node()});
141  }
142  mSubResistor->initialize(mFrequencies);
143  mSubResistor->initializeFromNodesAndTerminals(frequency);
144  addMNASubComponent(mSubResistor, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
145 
146  if (!mReactanceInSeries) {
147  **mIntfCurrent += mSubResistor->intfCurrent();
148  }
149  }
150 
151  if (mReactance(0, 0) > 0) {
152  mInductance = mReactance / (2 * PI * frequency);
153 
154  mSubInductor =
155  std::make_shared<EMT::Ph3::Inductor>(**mName + "_ind", mLogLevel);
156  mSubInductor->setParameters(mInductance);
157  if (mReactanceInSeries) {
158  mSubInductor->connect({SimNode::GND, mVirtualNodes[0]});
159  } else {
160  mSubInductor->connect({SimNode::GND, mTerminals[0]->node()});
161  }
162  mSubInductor->initialize(mFrequencies);
163  mSubInductor->initializeFromNodesAndTerminals(frequency);
164  addMNASubComponent(mSubInductor, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
165  MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
166 
167  if (!mReactanceInSeries) {
168  **mIntfCurrent += mSubInductor->intfCurrent();
169  }
170  } else if (mReactance(0, 0) < 0) {
171  mCapacitance = -1 / (2 * PI * frequency) * mReactance.inverse();
172 
173  mSubCapacitor =
174  std::make_shared<EMT::Ph3::Capacitor>(**mName + "_cap", mLogLevel);
175  mSubCapacitor->setParameters(mCapacitance);
176  if (mReactanceInSeries) {
177  mSubCapacitor->connect({SimNode::GND, mVirtualNodes[0]});
178  } else {
179  mSubCapacitor->connect({SimNode::GND, mTerminals[0]->node()});
180  }
181  mSubCapacitor->initialize(mFrequencies);
182  mSubCapacitor->initializeFromNodesAndTerminals(frequency);
183  addMNASubComponent(mSubCapacitor,
184  MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
185  MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
186  if (!mReactanceInSeries) {
187  **mIntfCurrent += mSubCapacitor->intfCurrent();
188  }
189  }
190 
191  SPDLOG_LOGGER_INFO(
192  mSLog,
193  "\n--- Initialization from powerflow ---"
194  "\nVoltage across: {:s}"
195  "\nCurrent: {:s}"
196  "\nTerminal 0 voltage: {:s}"
197  "\nActive Power: {:s}"
198  "\nReactive Power: {:s}"
199  "\nResistance: {:s}"
200  "\nReactance: {:s}"
201  "\n--- Initialization from powerflow finished ---",
202  Logger::matrixToString(**mIntfVoltage),
203  Logger::matrixToString(**mIntfCurrent),
204  Logger::phasorToString(RMS3PH_TO_PEAK1PH * initialSingleVoltage(0)),
205  Logger::matrixToString(**mActivePower),
206  Logger::matrixToString(**mReactivePower),
207  Logger::matrixToString(mResistance), Logger::matrixToString(mReactance));
208  mSLog->flush();
209 }
