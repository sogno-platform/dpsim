29void EMT::Ph3::ReducedOrderSynchronGeneratorVBR::initializeResistanceMatrix() {
30 // dq0 resistance matrix
31 mResistanceMatrixDq0 = MatrixFixedSize<3, 3>::Zero(3, 3);
32 mResistanceMatrixDq0 << 0.0, mA, 0.0, mB, 0.0, 0.0, 0.0, 0.0, mL0;
33
34 // initialize conductance matrix
35 mConductanceMatrix = MatrixFixedSize<3, 3>::Zero(3, 3);
36}
37
38void EMT::Ph3::ReducedOrderSynchronGeneratorVBR::calculateResistanceMatrix() {
39 MatrixFixedSize<3, 3> resistanceMatrix =
40 mDq0ToAbc * mResistanceMatrixDq0 * mAbcToDq0;
41 resistanceMatrix = resistanceMatrix * mBase_Z;
42 mConductanceMatrix = resistanceMatrix.inverse();
43}

183void EMT::Ph3::ReducedOrderSynchronGeneratorVBR::mnaCompApplySystemMatrixStamp(
184 SparseMatrixRow &systemMatrix) {
185
186 if (mModelAsNortonSource) {
187 MNAStampUtils::stamp3x3ConductanceMatrixNodeToGround(
188 mConductanceMatrix, systemMatrix, matrixNodeIndex(0), mSLog);
189 } else {
190 // Stamp voltage source
191 Math::addToMatrixElement(
192 systemMatrix, mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
193 mVirtualNodes[1]->matrixNodeIndex(PhaseType::A), -1);
194 Math::addToMatrixElement(
195 systemMatrix, mVirtualNodes[1]->matrixNodeIndex(PhaseType::A),
196 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A), 1);
197 Math::addToMatrixElement(
198 systemMatrix, mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
199 mVirtualNodes[1]->matrixNodeIndex(PhaseType::B), -1);
200 Math::addToMatrixElement(
201 systemMatrix, mVirtualNodes[1]->matrixNodeIndex(PhaseType::B),
202 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B), 1);
203 Math::addToMatrixElement(
204 systemMatrix, mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
205 mVirtualNodes[1]->matrixNodeIndex(PhaseType::C), -1);
206 Math::addToMatrixElement(
207 systemMatrix, mVirtualNodes[1]->matrixNodeIndex(PhaseType::C),
208 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C), 1);
209
210 // Stamp conductance matrix
211 MNAStampUtils::stamp3x3ConductanceMatrixBetween2Nodes(
212 mConductanceMatrix, systemMatrix,
213 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A), matrixNodeIndex(0),
214 mSLog);
215 }
216}

218void EMT::Ph3::ReducedOrderSynchronGeneratorVBR::
219 mnaCompApplyRightSideVectorStamp(Matrix &rightVector) {
220 if (mModelAsNortonSource) {
221 // compute equivalent northon circuit in abc reference frame
222 mIvbr = mConductanceMatrix * mEvbr;
223
224 Math::setVectorElement(rightVector, matrixNodeIndex(0, 0), mIvbr(0, 0));
225 Math::setVectorElement(rightVector, matrixNodeIndex(0, 1), mIvbr(1, 0));
226 Math::setVectorElement(rightVector, matrixNodeIndex(0, 2), mIvbr(2, 0));
227 } else {
228 Math::setVectorElement(rightVector,
229 mVirtualNodes[1]->matrixNodeIndex(PhaseType::A),
230 mEvbr(0, 0));
231 Math::setVectorElement(rightVector,
232 mVirtualNodes[1]->matrixNodeIndex(PhaseType::B),
233 mEvbr(1, 0));
234 Math::setVectorElement(rightVector,
235 mVirtualNodes[1]->matrixNodeIndex(PhaseType::C),
236 mEvbr(2, 0));
237 }

240void EMT::Ph3::ReducedOrderSynchronGeneratorVBR::mnaCompPostStep(
241 const Matrix &leftVector) {
242 // update armature voltage
243 (**mIntfVoltage)(0, 0) =
244 Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 0));
245 (**mIntfVoltage)(1, 0) =
246 Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 1));
247 (**mIntfVoltage)(2, 0) =
248 Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 2));
249
250 // convert terminal voltage into dq reference frame
251 **mVdq0 = mAbcToDq0 * **mIntfVoltage / mBase_V;
252
253 // update armature current
254 if (mModelAsNortonSource) {
255 Matrix Iconductance = mConductanceMatrix * **mIntfVoltage;
256 (**mIntfCurrent) = mIvbr - Iconductance;
257 } else {
258 (**mIntfCurrent)(0, 0) = Math::realFromVectorElement(
259 leftVector, mVirtualNodes[1]->matrixNodeIndex(PhaseType::A));
260 (**mIntfCurrent)(1, 0) = Math::realFromVectorElement(
261 leftVector, mVirtualNodes[1]->matrixNodeIndex(PhaseType::B));
262 (**mIntfCurrent)(2, 0) = Math::realFromVectorElement(
263 leftVector, mVirtualNodes[1]->matrixNodeIndex(PhaseType::C));
264 }
265
266 // convert armature current into dq reference frame
267 **mIdq0 = mAbcToDq0 * **mIntfCurrent / mBase_I;
268}

270Matrix
271EMT::Ph3::ReducedOrderSynchronGeneratorVBR::get_parkTransformMatrix() const {
272 Matrix abcToDq0(3, 3);
273
274 abcToDq0 << 2. / 3. * cos(**mThetaMech),
275 2. / 3. * cos(**mThetaMech - 2. * PI / 3.),
276 2. / 3. * cos(**mThetaMech + 2. * PI / 3.), -2. / 3. * sin(**mThetaMech),
277 -2. / 3. * sin(**mThetaMech - 2. * PI / 3.),
278 -2. / 3. * sin(**mThetaMech + 2. * PI / 3.), 1. / 3., 1. / 3., 1. / 3.;
279
280 return abcToDq0;
281}
