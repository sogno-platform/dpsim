20void EMT::Ph3::SSN::Full_Serial_RLC::setParameters(const Matrix &resistance,
21 const Matrix &inductance,
22 const Matrix &capacitance) {
23 if (resistance.rows() != 3 || resistance.cols() != 3)
24 throw std::invalid_argument("Resistance matrix must have size 3 x 3.");
25
26 if (inductance.rows() != 3 || inductance.cols() != 3)
27 throw std::invalid_argument("Inductance matrix must have size 3 x 3.");
28
29 if (capacitance.rows() != 3 || capacitance.cols() != 3)
30 throw std::invalid_argument("Capacitance matrix must have size 3 x 3.");
31
32 **mResistance = resistance;
33 **mInductance = inductance;
34 **mCapacitance = capacitance;
35
36 Matrix inverseInductance = inductance.inverse();
37 Matrix inverseCapacitance = capacitance.inverse();
38 Matrix identity3 = Matrix::Identity(3, 3);
39
40 // State choice:
41 // x = [uC_abc; i_abc] -> 6x1
42 // u = v_abc -> 3x1
43 // y = i_abc -> 3x1
44
45 Matrix aMatrix = Matrix::Zero(6, 6);
46 aMatrix.block(0, 3, 3, 3) = inverseCapacitance;
47 aMatrix.block(3, 0, 3, 3) = -inverseInductance;
48 aMatrix.block(3, 3, 3, 3) = -inverseInductance * resistance;
49
50 Matrix bMatrix = Matrix::Zero(6, 3);
51 bMatrix.block(3, 0, 3, 3) = inverseInductance;
52
53 Matrix cMatrix = Matrix::Zero(3, 6);
54 cMatrix.block(0, 3, 3, 3) = identity3;
55
56 Matrix dMatrix = Matrix::Zero(3, 3);
57
58 SSNComp::setParameters(aMatrix, bMatrix, cMatrix, dMatrix);
59}
