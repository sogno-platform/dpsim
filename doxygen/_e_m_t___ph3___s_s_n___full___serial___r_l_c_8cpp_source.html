32 void EMT::Ph3::SSN::Full_Serial_RLC::initializeFromNodesAndTerminals(
33  Real frequency) {
34 
35  Real omega = 2 * PI * frequency;
36 
37  MatrixComp impedance = MatrixComp::Zero(3, 3);
38 
39  impedance << Complex((**mResistance)(0, 0),
40  omega * (**mInductance)(0, 0) -
41  1. / (omega * (**mCapacitance)(0, 0))),
42  Complex((**mResistance)(0, 1), omega * (**mInductance)(0, 1) -
43  1. / (omega * (**mCapacitance)(0, 1))),
44  Complex((**mResistance)(0, 2), omega * (**mInductance)(0, 2) -
45  1. / (omega * (**mCapacitance)(0, 2))),
46  Complex((**mResistance)(1, 0), omega * (**mInductance)(1, 0) -
47  1. / (omega * (**mCapacitance)(1, 0))),
48  Complex((**mResistance)(1, 1), omega * (**mInductance)(1, 1) -
49  1. / (omega * (**mCapacitance)(1, 1))),
50  Complex((**mResistance)(1, 2), omega * (**mInductance)(1, 2) -
51  1. / (omega * (**mCapacitance)(1, 2))),
52  Complex((**mResistance)(2, 0), omega * (**mInductance)(2, 0) -
53  1. / (omega * (**mCapacitance)(2, 0))),
54  Complex((**mResistance)(2, 1), omega * (**mInductance)(2, 1) -
55  1. / (omega * (**mCapacitance)(2, 1))),
56  Complex((**mResistance)(2, 2), omega * (**mInductance)(2, 2) -
57  1. / (omega * (**mCapacitance)(2, 2)));
58 
59  MatrixComp vInitABC = Matrix::Zero(3, 1);
60  vInitABC(0, 0) = RMS3PH_TO_PEAK1PH * initialSingleVoltage(1) -
61  RMS3PH_TO_PEAK1PH * initialSingleVoltage(0);
62  vInitABC(1, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_B;
63  vInitABC(2, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_C;
64  **mIntfVoltage = vInitABC.real();
65  MatrixComp admittance = impedance.inverse();
66  **mIntfCurrent = (admittance * vInitABC).real();
67 
69  **mIntfCurrent = Matrix::Zero(3, 3);
70 
71  SPDLOG_LOGGER_INFO(mSLog, "\nImpedance [Ohm]: {:s}",
72  Logger::matrixCompToString(impedance));
73  SPDLOG_LOGGER_INFO(
74  mSLog,
75  "\n--- Initialization from powerflow ---"
76  "\nVoltage across: {:s}"
77  "\nCurrent: {:s}"
78  "\nTerminal 0 voltage: {:s}"
79  "\nTerminal 1 voltage: {:s}"
80  "\n--- Initialization from powerflow finished ---",
81  Logger::matrixToString(**mIntfVoltage),
82  Logger::matrixToString(**mIntfCurrent),
83  Logger::phasorToString(RMS3PH_TO_PEAK1PH * initialSingleVoltage(0)),
84  Logger::phasorToString(RMS3PH_TO_PEAK1PH * initialSingleVoltage(1)));
85 }

87 void EMT::Ph3::SSN::Full_Serial_RLC::mnaCompInitialize(
88  Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
89 
90  updateMatrixNodeIndices();
91 
92  mState = Matrix::Zero(6, 1);
93  mYHistory = Matrix::Zero(3, 1);
94 
95  //Fill mDufourAKHat
96  //State Equation one, phases A,B,C: top left submatrix
97  mDufourAKHat(0, 0) =
98  1. - ((2. * (timeStep * timeStep)) /
99  (4. * (**mInductance)(0, 0) * (**mCapacitance)(0, 0) +
100  2. * timeStep * (**mCapacitance)(0, 0) * (**mResistance)(0, 0) +
101  timeStep * timeStep));
102  mDufourAKHat(1, 1) =
103  1. - ((2. * (timeStep * timeStep)) /
104  (4. * (**mInductance)(1, 1) * (**mCapacitance)(1, 1) +
105  2. * timeStep * (**mCapacitance)(1, 1) * (**mResistance)(1, 1) +
106  timeStep * timeStep));
107  mDufourAKHat(2, 2) =
108  1. - ((2. * (timeStep * timeStep)) /
109  (4. * (**mInductance)(2, 2) * (**mCapacitance)(2, 2) +
110  2. * timeStep * (**mCapacitance)(2, 2) * (**mResistance)(2, 2) +
111  timeStep * timeStep));
112  //State Equation one, phases A,B,C: top right submatrix
113  mDufourAKHat(0, 3) =
114  (timeStep / (2. * (**mCapacitance)(0, 0))) *
115  (1. + ((4. * (**mInductance)(0, 0) * (**mCapacitance)(0, 0) -
116  2. * timeStep * (**mResistance)(0, 0) * (**mCapacitance)(0, 0) -
117  (timeStep * timeStep)) /
118  (4. * (**mInductance)(0, 0) * (**mCapacitance)(0, 0) +
119  2. * timeStep * (**mResistance)(0, 0) * (**mCapacitance)(0, 0) +
120  (timeStep * timeStep))));
121  mDufourAKHat(1, 4) =
122  (timeStep / (2. * (**mCapacitance)(1, 1))) *
123  (1. + ((4. * (**mInductance)(1, 1) * (**mCapacitance)(1, 1) -
124  2. * timeStep * (**mResistance)(1, 1) * (**mCapacitance)(1, 1) -
125  (timeStep * timeStep)) /
126  (4. * (**mInductance)(1, 1) * (**mCapacitance)(1, 1) +
127  2. * timeStep * (**mResistance)(1, 1) * (**mCapacitance)(1, 1) +
128  (timeStep * timeStep))));
129  mDufourAKHat(2, 5) =
130  (timeStep / (2. * (**mCapacitance)(2, 2))) *
131  (1. + ((4. * (**mInductance)(2, 2) * (**mCapacitance)(2, 2) -
132  2. * timeStep * (**mResistance)(2, 2) * (**mCapacitance)(2, 2) -
133  (timeStep * timeStep)) /
134  (4. * (**mInductance)(2, 2) * (**mCapacitance)(2, 2) +
135  2. * timeStep * (**mResistance)(2, 2) * (**mCapacitance)(2, 2) +
136  (timeStep * timeStep))));
137  //State Equation two, phases A,B,C: bottom left submatrix
138  mDufourAKHat(3, 0) =
139  -((4. * (**mCapacitance)(0, 0) * timeStep) /
140  (4. * (**mInductance)(0, 0) * (**mCapacitance)(0, 0) +
141  2. * timeStep * (**mCapacitance)(0, 0) * (**mResistance)(0, 0) +
142  (timeStep * timeStep)));
143  mDufourAKHat(4, 1) =
144  -((4. * (**mCapacitance)(1, 1) * timeStep) /
145  (4. * (**mInductance)(1, 1) * (**mCapacitance)(1, 1) +
146  2. * timeStep * (**mCapacitance)(1, 1) * (**mResistance)(1, 1) +
147  (timeStep * timeStep)));
148  mDufourAKHat(5, 2) =
149  -((4. * (**mCapacitance)(2, 2) * timeStep) /
150  (4. * (**mInductance)(2, 2) * (**mCapacitance)(2, 2) +
151  2. * timeStep * (**mCapacitance)(2, 2) * (**mResistance)(2, 2) +
152  (timeStep * timeStep)));
153  //State Equation two, phases A,B,C: bottom right submatrix
154  mDufourAKHat(3, 3) =
155  (4. * (**mInductance)(0, 0) * (**mCapacitance)(0, 0) -
156  2. * timeStep * (**mResistance)(0, 0) * (**mCapacitance)(0, 0) -
157  (timeStep * timeStep)) /
158  (4. * (**mInductance)(0, 0) * (**mCapacitance)(0, 0) +
159  2. * timeStep * (**mResistance)(0, 0) * (**mCapacitance)(0, 0) +
160  (timeStep * timeStep));
161  mDufourAKHat(4, 4) =
162  (4. * (**mInductance)(1, 1) * (**mCapacitance)(1, 1) -
163  2. * timeStep * (**mResistance)(1, 1) * (**mCapacitance)(1, 1) -
164  (timeStep * timeStep)) /
165  (4. * (**mInductance)(1, 1) * (**mCapacitance)(1, 1) +
166  2. * timeStep * (**mResistance)(1, 1) * (**mCapacitance)(1, 1) +
167  (timeStep * timeStep));
168  mDufourAKHat(5, 5) =
169  (4. * (**mInductance)(2, 2) * (**mCapacitance)(2, 2) -
170  2. * timeStep * (**mResistance)(2, 2) * (**mCapacitance)(2, 2) -
171  (timeStep * timeStep)) /
172  (4. * (**mInductance)(2, 2) * (**mCapacitance)(2, 2) +
173  2. * timeStep * (**mResistance)(2, 2) * (**mCapacitance)(2, 2) +
174  (timeStep * timeStep));
175 
177  //State Equation one, phases A,B,C: top submatrix
178  mDufourBKHat(0, 0) =
179  (timeStep * timeStep) /
180  (4. * (**mInductance)(0, 0) * (**mCapacitance)(0, 0) +
181  2. * timeStep * (**mCapacitance)(0, 0) * (**mResistance)(0, 0) +
182  (timeStep * timeStep));
183  mDufourBKHat(1, 1) =
184  (timeStep * timeStep) /
185  (4. * (**mInductance)(1, 1) * (**mCapacitance)(1, 1) +
186  2. * timeStep * (**mCapacitance)(1, 1) * (**mResistance)(1, 1) +
187  (timeStep * timeStep));
188  mDufourBKHat(2, 2) =
189  (timeStep * timeStep) /
190  (4. * (**mInductance)(2, 2) * (**mCapacitance)(2, 2) +
191  2. * timeStep * (**mCapacitance)(2, 2) * (**mResistance)(2, 2) +
192  (timeStep * timeStep));
193 
194  //State Equation two, phases A,B,C: bottom submatrix
195  mDufourBKHat(3, 0) =
196  (timeStep * 2. * (**mCapacitance)(0, 0)) /
197  (4. * (**mInductance)(0, 0) * (**mCapacitance)(0, 0) +
198  2. * timeStep * (**mCapacitance)(0, 0) * (**mResistance)(0, 0) +
199  (timeStep * timeStep));
200  mDufourBKHat(4, 1) =
201  (timeStep * 2. * (**mCapacitance)(1, 1)) /
202  (4. * (**mInductance)(1, 1) * (**mCapacitance)(1, 1) +
203  2. * timeStep * (**mCapacitance)(1, 1) * (**mResistance)(1, 1) +
204  (timeStep * timeStep));
205  mDufourBKHat(5, 2) =
206  (timeStep * 2. * (**mCapacitance)(2, 2)) /
207  (4. * (**mInductance)(2, 2) * (**mCapacitance)(2, 2) +
208  2. * timeStep * (**mCapacitance)(2, 2) * (**mResistance)(2, 2) +
209  (timeStep * timeStep));
210 
211  mDufourBKNHat = mDufourBKHat;
212 
213  mDufourCKN(0, 3) = 1.;
214  mDufourCKN(1, 4) = 1.;
215  mDufourCKN(2, 5) = 1.;
216 
217  mDufourWKN = mDufourCKN * mDufourBKHat;
218 
225  mState(3, 0) = (**mIntfCurrent)(0, 0);
226  mState(4, 0) = (**mIntfCurrent)(1, 0);
227  mState(5, 0) = (**mIntfCurrent)(2, 0);
228  ssnUpdateState();
229  mState(3, 0) = (**mIntfCurrent)(0, 0);
230  mState(4, 0) = (**mIntfCurrent)(1, 0);
231  mState(5, 0) = (**mIntfCurrent)(2, 0);
232 
233  **mRightVector = Matrix::Zero(leftVector->get().rows(), 1);
234 
235  SPDLOG_LOGGER_INFO(mSLog,
236  "\n--- MNA initialization ---"
237  "\nInitial voltage {:s}"
238  "\nInitial current {:s}"
239  "\n--- MNA initialization finished ---",
240  Logger::matrixToString(**mIntfVoltage),
241  Logger::matrixToString(**mIntfCurrent));
242  mSLog->flush();
243 }

255 void EMT::Ph3::SSN::Full_Serial_RLC::mnaCompApplyRightSideVectorStamp(
256  Matrix &rightVector) {
257  // Update history term
258  mYHistory =
259  mDufourCKN * (mDufourAKHat * mState + mDufourBKHat * **mIntfVoltage);
260 
261  if (terminalNotGrounded(0)) {
262  Math::setVectorElement(rightVector, matrixNodeIndex(0, 0), mYHistory(0, 0));
263  Math::setVectorElement(rightVector, matrixNodeIndex(0, 1), mYHistory(1, 0));
264  Math::setVectorElement(rightVector, matrixNodeIndex(0, 2), mYHistory(2, 0));
265  }
266  if (terminalNotGrounded(1)) {
267  Math::setVectorElement(rightVector, matrixNodeIndex(1, 0),
268  -mYHistory(0, 0));
269  Math::setVectorElement(rightVector, matrixNodeIndex(1, 1),
270  -mYHistory(1, 0));
271  Math::setVectorElement(rightVector, matrixNodeIndex(1, 2),
272  -mYHistory(2, 0));
273  }
274  SPDLOG_LOGGER_DEBUG(
275  mSLog, "\nHistory current term (mnaCompApplyRightSideVectorStamp): {:s}",
276  Logger::matrixToString(mYHistory));
277  mSLog->flush();
278 }

312 void EMT::Ph3::SSN::Full_Serial_RLC::mnaCompUpdateVoltage(
313  const Matrix &leftVector) {
314  // v1 - v0
315  mDufourUNT = **mIntfVoltage;
316  **mIntfVoltage = Matrix::Zero(3, 1);

318  (**mIntfVoltage)(0, 0) =
319  Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 0));
320  (**mIntfVoltage)(1, 0) =
321  Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 1));
322  (**mIntfVoltage)(2, 0) =
323  Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 2));
324  }

326  (**mIntfVoltage)(0, 0) =
327  (**mIntfVoltage)(0, 0) -
328  Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 0));
329  (**mIntfVoltage)(1, 0) =
330  (**mIntfVoltage)(1, 0) -
331  Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 1));
332  (**mIntfVoltage)(2, 0) =
333  (**mIntfVoltage)(2, 0) -
334  Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 2));
335  }
336  SPDLOG_LOGGER_DEBUG(mSLog, "\nUpdate Voltage: {:s}",
337  Logger::matrixToString(**mIntfVoltage));
338  mSLog->flush();
