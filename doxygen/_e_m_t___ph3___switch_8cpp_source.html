30 void EMT::Ph3::Switch::initializeFromNodesAndTerminals(Real frequency) {
31 
32  Matrix impedance =
33  (**mSwitchClosed) ? **mClosedResistance : **mOpenResistance;
34  MatrixComp vInitABC = MatrixComp::Zero(3, 1);
35  vInitABC(0, 0) = initialSingleVoltage(1) - initialSingleVoltage(0);
36  vInitABC(1, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_B;
37  vInitABC(2, 0) = vInitABC(0, 0) * SHIFT_TO_PHASE_C;
38  **mIntfVoltage = vInitABC.real();
39  **mIntfCurrent = (impedance.inverse() * vInitABC).real();
40 
41  SPDLOG_LOGGER_INFO(mSLog,
42  "\n--- Initialization from powerflow ---"
43  "\nVoltage across: {:s}"
44  "\nCurrent: {:s}"
45  "\nTerminal 0 voltage: {:s}"
46  "\nTerminal 1 voltage: {:s}"
47  "\n--- Initialization from powerflow finished ---",
48  Logger::matrixToString(**mIntfVoltage),
49  Logger::matrixToString(**mIntfCurrent),
50  Logger::phasorToString(initialSingleVoltage(0)),
51  Logger::phasorToString(initialSingleVoltage(1)));

266 void EMT::Ph3::Switch::mnaCompUpdateVoltage(const Matrix &leftVector) {
267  // Voltage across component is defined as V1 - V0
268  **mIntfVoltage = Matrix::Zero(3, 1);
269  if (terminalNotGrounded(1)) {
270  (**mIntfVoltage)(0, 0) =
271  Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 0));
272  (**mIntfVoltage)(1, 0) =
273  Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 1));
274  (**mIntfVoltage)(2, 0) =
275  Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 2));
276  }
277  if (terminalNotGrounded(0)) {
278  (**mIntfVoltage)(0, 0) =
279  (**mIntfVoltage)(0, 0) -
280  Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 0));
281  (**mIntfVoltage)(1, 0) =
282  (**mIntfVoltage)(1, 0) -
283  Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 1));
284  (**mIntfVoltage)(2, 0) =
285  (**mIntfVoltage)(2, 0) -
286  Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 2));
287  }
288 }
