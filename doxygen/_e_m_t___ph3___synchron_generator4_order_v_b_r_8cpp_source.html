30void EMT::Ph3::SynchronGenerator4OrderVBR::specificInitialization() {

46void EMT::Ph3::SynchronGenerator4OrderVBR::stepInPerUnit() {
47
48 if (mSimTime > 0.0) {
49 // calculate Edq_t at t=k
50 (**mEdq0_t)(0, 0) = -(**mIdq0)(1, 0) * mLq_t + (**mVdq0)(0, 0);
51 (**mEdq0_t)(1, 0) = (**mIdq0)(0, 0) * mLd_t + (**mVdq0)(1, 0);
52 (**mEdq0_t)(2, 0) = 0.0;
53 }
54
55 // get transformation matrix
56 mAbcToDq0 = get_parkTransformMatrix();
57 mDq0ToAbc = get_inverseParkTransformMatrix();
58
59 // calculate resistance matrix at t=k+1
60 calculateResistanceMatrix();
61
62 // VBR history voltage
63 mEhs_vbr(0, 0) = mAd_t * (**mIdq0)(1, 0) + mBd_t * (**mEdq0_t)(0, 0);
64 mEhs_vbr(1, 0) = mAq_t * (**mIdq0)(0, 0) + mBq_t * (**mEdq0_t)(1, 0) +
65 mDq_t * mEf_prev + mDq_t * (**mEf);
66 mEhs_vbr(2, 0) = 0.0;
67
68 // convert Edq_t into the abc reference frame
69 mEvbr = mDq0ToAbc * mEhs_vbr * mBase_V;
70}
