33void EMT::Ph3::SynchronGenerator5OrderVBR::specificInitialization() {
34 // initial voltage behind the transient reactance in the dq reference frame
35 (**mEdq0_t)(0, 0) = 0.0;
36 (**mEdq0_t)(1, 0) =
37 (1 - mTaa / mTd0_t) * **mEf - (mLd - mLd_t - mYd) * (**mIdq0)(0, 0);
38
39 // initial dq behind the subtransient reactance in the dq reference frame
40 (**mEdq0_s)(0, 0) = (**mVdq0)(0, 0) - mLq_s * (**mIdq0)(1, 0);
41 (**mEdq0_s)(1, 0) = (**mVdq0)(1, 0) + mLd_s * (**mIdq0)(0, 0);
42
43 // initialize history term behind the transient reactance
44 mEh_t(0, 0) = 0.0;
45 mEh_t(1, 0) = mAq_t * (**mIdq0)(0, 0) + mBq_t * (**mEdq0_t)(1, 0) +
46 mDq_t * (**mEf) + mDq_t * mEf_prev;
47 mEh_t(2, 0) = 0.0;
48
49 SPDLOG_LOGGER_DEBUG(mSLog,
50 "\n--- Model specific initialization ---"
51 "\nSG model: 5th order"
52 "\nInitial Ed_t (per unit): {:f}"
53 "\nInitial Eq_t (per unit): {:f}"
54 "\nInitial Ed_s (per unit): {:f}"
55 "\nInitial Eq_s (per unit): {:f}"
56 "\n--- Model specific initialization finished ---",
57
58 (**mEdq0_t)(0, 0), (**mEdq0_t)(1, 0), (**mEdq0_s)(0, 0),
59 (**mEdq0_s)(1, 0));
60 mSLog->flush();
61}

63void EMT::Ph3::SynchronGenerator5OrderVBR::stepInPerUnit() {
64
65 // calculate Edq_t at t=k
66 (**mEdq0_t)(0, 0) = 0.0;
67 (**mEdq0_t)(1, 0) = mAq_t * (**mIdq0)(0, 0) + mEh_t(1, 0);
68 (**mEdq0_t)(2, 0) = 0.0;
69
70 // calculate Edq_s at t=k
71 (**mEdq0_s)(0, 0) = (**mVdq0)(0, 0) - mLq_s * (**mIdq0)(1, 0);
72 (**mEdq0_s)(1, 0) = (**mVdq0)(1, 0) + mLd_s * (**mIdq0)(0, 0);
73
74 // get transformation matrix
75 mAbcToDq0 = get_parkTransformMatrix();
76 mDq0ToAbc = get_inverseParkTransformMatrix();
77
78 // calculate resistance matrix at t=k+1
79 calculateResistanceMatrix();
80
81 // calculate history term behind the transient reactance
82 mEh_t(0, 0) = 0.0;
83 mEh_t(1, 0) = mAq_t * (**mIdq0)(0, 0) + mBq_t * (**mEdq0_t)(1, 0) +
84 mDq_t * (**mEf) + mDq_t * mEf_prev;
85 mEh_t(2, 0) = 0.0;
86
87 // calculate history term behind the subtransient reactance
88 mEh_s(0, 0) = mAd_s * (**mIdq0)(1, 0) + mCd_s * (**mEdq0_s)(0, 0);
89 mEh_s(1, 0) = mAq_s * (**mIdq0)(0, 0) + mBq_s * (**mEdq0_t)(1, 0) +
90 mCq_s * (**mEdq0_s)(1, 0) + mDq_s * (**mEf) + mDq_s * mEf_prev;
91 mEh_s(2, 0) = 0.0;
92
93 // convert Edq_s into the abc reference frame
94 mEvbr = mDq0ToAbc * mEh_s * mBase_V;
95}
