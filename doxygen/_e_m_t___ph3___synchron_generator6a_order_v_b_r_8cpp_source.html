33void EMT::Ph3::SynchronGenerator6aOrderVBR::specificInitialization() {
34
35 // initial voltage behind the transient reactance in the dq reference frame
36 (**mEdq0_t)(0, 0) = (mLq - mLq_t - mYq) * (**mIdq0)(1, 0);
37 (**mEdq0_t)(1, 0) =
38 (1 - mTaa / mTd0_t) * **mEf - (mLd - mLd_t - mYd) * (**mIdq0)(0, 0);
39
40 // initial dq behind the subtransient reactance in the dq reference frame
41 (**mEdq0_s)(0, 0) = (**mVdq0)(0, 0) - mLq_s * (**mIdq0)(1, 0);
42 (**mEdq0_s)(1, 0) = (**mVdq0)(1, 0) + mLd_s * (**mIdq0)(0, 0);
43
44 // initialize history term behind the transient reactance
45 mEh_t(0, 0) = mAd_t * (**mIdq0)(1, 0) + mBd_t * (**mEdq0_t)(0, 0);
46 mEh_t(1, 0) = mAq_t * (**mIdq0)(0, 0) + mBq_t * (**mEdq0_t)(1, 0) +
47 mDq_t * (**mEf) + mDq_t * mEf_prev;
48 mEh_t(2, 0) = 0.0;
49
50 SPDLOG_LOGGER_INFO(mSLog,
51 "\n--- Model specific initialization ---"
52 "\nSG model: 6th order type a (Marconato's model)"
53 "\nInitial Ed_t (per unit): {:f}"
54 "\nInitial Eq_t (per unit): {:f}"
55 "\nInitial Ed_s (per unit): {:f}"
56 "\nInitial Eq_s (per unit): {:f}"
57 "\n--- Model specific initialization finished ---",
58
59 (**mEdq0_t)(0, 0), (**mEdq0_t)(1, 0), (**mEdq0_s)(0, 0),
60 (**mEdq0_s)(1, 0));
61 mSLog->flush();
62}

64void EMT::Ph3::SynchronGenerator6aOrderVBR::stepInPerUnit() {
65
66 // calculate Edq_t at t=k
67 (**mEdq0_t)(0, 0) = mAd_t * (**mIdq0)(1, 0) + mEh_t(0, 0);
68 (**mEdq0_t)(1, 0) = mAq_t * (**mIdq0)(0, 0) + mEh_t(1, 0);
69 (**mEdq0_t)(2, 0) = 0.0;
70
71 // calculate Edq_s at t=k
72 (**mEdq0_s)(0, 0) = -(**mIdq0)(1, 0) * mLq_s + (**mVdq0)(0, 0);
73 (**mEdq0_s)(1, 0) = (**mIdq0)(0, 0) * mLd_s + (**mVdq0)(1, 0);
74
75 // get transformation matrix
76 mAbcToDq0 = get_parkTransformMatrix();
77 mDq0ToAbc = get_inverseParkTransformMatrix();
78
79 // calculate resistance matrix at t=k+1
80 calculateResistanceMatrix();
81
82 // calculate history term behind the transient reactance
83 mEh_t(0, 0) = mAd_t * (**mIdq0)(1, 0) + mBd_t * (**mEdq0_t)(0, 0);
84 mEh_t(1, 0) = mAq_t * (**mIdq0)(0, 0) + mBq_t * (**mEdq0_t)(1, 0) +
85 mDq_t * (**mEf) + mDq_t * mEf_prev;
86 mEh_t(2, 0) = 0.0;
87
88 // calculate history term behind the subtransient reactance
89 mEh_s(0, 0) = mAd_s * (**mIdq0)(1, 0) + mBd_s * (**mEdq0_t)(0, 0) +
90 mCd_s * (**mEdq0_s)(0, 0);
91 mEh_s(1, 0) = mAq_s * (**mIdq0)(0, 0) + mBq_s * (**mEdq0_t)(1, 0) +
92 mCq_s * (**mEdq0_s)(1, 0) + mDq_s * (**mEf) + mDq_s * mEf_prev;
93 mEh_s(2, 0) = 0.0;
94
95 // convert Edq_s into the abc reference frame
96 mEvbr = mDq0ToAbc * mEh_s * mBase_V;
97}
