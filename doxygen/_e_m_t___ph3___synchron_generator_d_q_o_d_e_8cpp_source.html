53void EMT::Ph3::SynchronGeneratorDQODE::odePreStep() {
54 for (int i = 0; i < mDim - 2; i++)
55 (**mOdePreState)(i) = mPsisr(i, 0);
56
57 (**mOdePreState)(mDim - 2) = mThetaMech;
58 (**mOdePreState)(mDim - 1) = **mOmMech;
59
60 //copied from stepInPerUnit
61 mVdq0 = abcToDq0Transform(mThetaMech, **mIntfVoltage);
62 mVdq0 = mVdq0 / mBase_V;

70void EMT::Ph3::SynchronGeneratorDQODE::odePostStep() {
71 for (int i = 0; i < mDim - 2; i++)
72 mPsisr(i, 0) = (**mOdePostState)(i);
73
74 mThetaMech = (**mOdePostState)(mDim - 2);
75 **mOmMech = (**mOdePostState)(mDim - 1);
76
77 //copied from stepInPerUnit ; makes only sense to call this after write back
78 mIsr = mFluxToCurrentMat * mPsisr;
79
80 mIdq0(0, 0) = mIsr(0, 0);
81 mIdq0(1, 0) = mIsr(3, 0);
82 mIdq0(2, 0) = mIsr(6, 0);
83 **mIntfCurrent = mBase_I * dq0ToAbcTransform(mThetaMech, mIdq0);
84
85 SPDLOG_LOGGER_DEBUG(mSLog, "\nCurrent: \n{:s}",
86 Logger::matrixCompToString(**mIntfCurrent));
87}

89// ODE-Class simulation state-space
90void EMT::Ph3::SynchronGeneratorDQODE::odeStateSpace(double t, const double y[],
91 double ydot[]) {
92 if (mDim == 9) { //flux-order: ds, fd, kd, qs, kq1, (kq2,) 0s
93 //multiply voltages with mBase_OmElec (similar to stepInPerUnit):
94 ydot[0] = mBase_OmElec * (mVdq0(0, 0) + y[8] * y[3]); //v_d+omega*lambda_q
95 ydot[1] = mBase_OmElec * mVsr(1, 0); //v_fd
96 ydot[2] = 0;
97 ydot[3] = mBase_OmElec * (mVdq0(1, 0) - y[8] * y[0]); //v_q-omega*lambda_d
98 ydot[4] = 0;
99 ydot[5] = 0;
100 ydot[6] = mBase_OmElec * mVdq0(2, 0); //v_0
101 } else {
102 ydot[0] = mBase_OmElec * (mVdq0(0, 0) + y[8] * y[3]); //v_d+omega*lambda_q
103 ydot[1] = mBase_OmElec * mVsr(1, 0); //v_fd
104 ydot[2] = 0;
105 ydot[3] = mBase_OmElec * (mVdq0(1, 0) - y[8] * y[0]); //v_q-omega*lambda_d
106 ydot[4] = 0;
107 ydot[5] = mBase_OmElec * mVdq0(2, 0); //v_0
108 }
109
110 // Matrix Vector-Mult:
111 for (int i = 0; i < mDim - 2; i++)
112 for (int j = 0; j < mDim - 2; j++)
113 ydot[i] += mBase_OmElec * mFluxStateSpaceMat(i, j) * y[j];
114
115 // Mechanical equations:
116
117 // Compute theta with transformation (following stepInPerUnit):
118 ydot[mDim - 2] = y[mDim - 1] * mBase_OmMech;
119
120 /* Auxiliary variables to compute
121 * T_e= lambda_d * i_q - lambda_q * i_d
122 * needed for omega: */
123 realtype i_d = 0, i_q = 0;
124 // Compute new currents (depending on updated fluxes)
125 for (int i = 0; i < mDim - 2; i++) {
126 i_d += mFluxToCurrentMat(0, i) * y[i];
127 i_q += mFluxToCurrentMat(3, i) * y[i];
128 }
129
130 // Compute Omega (according to stepInPerUnit)
131 ydot[mDim - 1] =
132 1 / (2 * **mInertia) * (**mMechTorque - (y[3] * i_d - y[0] * i_q));
133}

135void EMT::Ph3::SynchronGeneratorDQODE::odeJacobian(double t, const double y[],
136 double fy[], double J[],
137 double tmp1[], double tmp2[],
138 double tmp3[]) {
139 // Fill Jacobian:
140 // Fluxes:
141 for (int i = 0; i < mDim - 2; i++)
142 for (int j = 0; j < mDim - 2; j++)
143 J[i * (mDim - 2) + j] =
144 mBase_OmElec *
145 mFluxStateSpaceMat(i, j); //Note: J is row-major ordered
146
147 //omega in flux eqs.:
148 J[0 * (mDim - 2) + mDim - 1] = mBase_OmElec * y[3];
149 J[3 * (mDim - 2) + mDim - 1] = mBase_OmElec * (-1) * y[0];
150 //theta-row:
151 J[(mDim - 2) * (mDim - 2) + mDim - 1] = 1 * mBase_OmMech;
152
153 realtype i_d = 0, i_q = 0;
154
155 for (int i = 0; i < mDim - 2; i++) {
156 i_d += mFluxToCurrentMat(0, i) * y[i];
157 i_q += mFluxToCurrentMat(3, i) * y[i];
158 }
159 //fluxes show up in omega eq. (with mBase_OmMech ?)
160 J[(mDim - 1) * (mDim - 2) + 0] = 1 / (2 * **mInertia) * i_q;
161 J[(mDim - 1) * (mDim - 2) + 3] = -1 / (2 * **mInertia) * i_d;
162}
