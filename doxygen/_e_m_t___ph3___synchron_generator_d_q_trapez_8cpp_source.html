49 void EMT::Ph3::SynchronGeneratorDQTrapez::stepInPerUnit(Real time) {
50  // Receives the voltage of time step k and has to
51  // calculate the current output for time step k+1
52 
53  // Calculate per unit values and
54  // transform per unit voltages from abc to dq0
55  mVdq0 = abcToDq0Transform(mThetaMech, **mIntfVoltage) / mBase_V;
56  mVsr(0, 0) = mVdq0(0, 0);
57  mVsr(3, 0) = mVdq0(1, 0);
58  mVsr(6, 0) = mVdq0(2, 0);
59  // Current is already available in pu dq0 from previous step
60 
61  // Update of fd winding voltage from exciter
62  if (mHasExciter) {
63  // Get exciter output voltage
64  // Note: scaled by Rfd/Lmd to transform from exciter pu system
65  // to the synchronous generator pu system
66  mVsr(1, 0) =
67  (mRfd / mLmd) * mExciter->step(mVsr(0, 0), mVsr(3, 0), mTimeStep);
68  }
69 
70  // Update of mechanical torque from turbine governor
71  if (mHasTurbineGovernor)
72  **mMechTorque = mTurbineGovernor->step(
73  **mOmMech, 1, mInitElecPower.real() / mNomPower, mTimeStep);
74 
75  // Calculation of electrical torque
76  **mElecTorque = (mPsisr(3, 0) * mIsr(0, 0) - mPsisr(0, 0) * mIsr(3, 0));
77 
78  // Update mechanical rotor angle with respect to electrical angle
79  // using Euler and previous states
80  mThetaMech = mThetaMech + mTimeStep * (**mOmMech * mBase_OmMech);
81 
82  // Update of omega using Euler
83  **mOmMech = **mOmMech + mTimeStep * (1. / (2. * **mInertia) *
84  (**mMechTorque - **mElecTorque));
85 
86  // Update using euler
87  **mDelta = **mDelta + mTimeStep * (**mOmMech - 1);
88 
89  // Update of fluxes
90  if (mNumericalMethod == NumericalMethod::Euler)
91  mPsisr = Math::StateSpaceEuler(
92  mPsisr, mBase_OmElec * (mFluxStateSpaceMat + mOmegaFluxMat * **mOmMech),
93  mBase_OmElec * mVsr, mTimeStep);
94  else
95  mPsisr = Math::StateSpaceTrapezoidal(
96  mPsisr, mBase_OmElec * (mFluxStateSpaceMat + mOmegaFluxMat * **mOmMech),
97  mBase_OmElec * mVsr, mTimeStep);
98 
99  // Calculate new currents from fluxes
100  mIsr = mFluxToCurrentMat * mPsisr;
101 
102  mIdq0(0, 0) = mIsr(0, 0);
103  mIdq0(1, 0) = mIsr(3, 0);
104  mIdq0(2, 0) = mIsr(6, 0);
105  **mIntfCurrent = mBase_I * dq0ToAbcTransform(mThetaMech, mIdq0);
106 }
