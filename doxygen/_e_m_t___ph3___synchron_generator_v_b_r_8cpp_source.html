109 void EMT::Ph3::SynchronGeneratorVBR::initializeFromNodesAndTerminals(
110  Real frequency) {
111  if (!mInitialValuesSet) {
112  SPDLOG_LOGGER_INFO(mSLog, "--- Initialization from powerflow ---");
113 
114  // terminal powers in consumer system -> convert to generator system
115  Real activePower = -terminal(0)->singlePower().real();
116  Real reactivePower = -terminal(0)->singlePower().imag();
117 
118  // voltage magnitude in phase-to-phase RMS -> convert to phase-to-ground peak expected by setInitialValues
119  Real voltMagnitude = RMS3PH_TO_PEAK1PH * Math::abs(initialSingleVoltage(0));
120 
121  this->setInitialValues(activePower, reactivePower, voltMagnitude,
122  Math::phase(initialSingleVoltage(0)), activePower);
123 
124  SPDLOG_LOGGER_INFO(mSLog,
125  "\nTerminal 0 voltage: {:s}"
126  "\nTerminal 0 power: {:s}"
127  "\n--- Initialization from powerflow finished ---",
128  Logger::phasorToString(initialSingleVoltage(0)),
129  Logger::complexToString(terminal(0)->singlePower()));
130  } else {
131  SPDLOG_LOGGER_INFO(mSLog, "Initial values already set, skipping "
132  "initializeFromNodesAndTerminals.");
133  }
134  mSLog->flush();
135 }

137 void EMT::Ph3::SynchronGeneratorVBR::mnaCompInitialize(
138  Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
139  updateMatrixNodeIndices();
140 
141  for (UInt phase1Idx = 0; phase1Idx < 3; ++phase1Idx)
142  for (UInt phase2Idx = 0; phase2Idx < 3; ++phase2Idx)
143  mVariableSystemMatrixEntries.push_back(std::make_pair<UInt, UInt>(
144  matrixNodeIndex(0, phase1Idx), matrixNodeIndex(0, phase2Idx)));
145 
146  SPDLOG_LOGGER_INFO(mSLog, "List of index pairs of varying matrix entries: ");
147  for (auto indexPair : mVariableSystemMatrixEntries)
148  SPDLOG_LOGGER_INFO(mSLog, "({}, {})", indexPair.first, indexPair.second);
149 
150  mSystemOmega = omega;
151  mTimeStep = timeStep;
152 
153  mResistanceMat = MatrixFixedSize<3, 3>::Zero(3, 3);
154  mResistanceMat << **mRs, 0, 0, 0, **mRs, 0, 0, 0, **mRs;
155 
156  //Dynamic mutual inductances
157  mDLmd = 1. / (1. / mLmd + 1. / mLlfd + 1. / mLlkd);
158 
159  if (mNumDampingWindings == 2)
160  mDLmq = 1. / (1. / mLmq + 1. / mLlkq1 + 1. / mLlkq2);
161  else {
162  mDLmq = 1. / (1. / mLmq + 1. / mLlkq1);
163  K1a = Matrix::Zero(2, 1);
164  K1 = Matrix::Zero(2, 1);
165  }
166 
167  mLa = (mDLmq + mDLmd) / 3.;
168  mLb = (mDLmd - mDLmq) / 3.;
169 
170  // steady state per unit initial value
171  initPerUnitStates();
172 
173  // Correcting variables
174  mThetaMech = mThetaMech + PI / 2;
175  **mMechTorque = -**mMechTorque;
176  mIq = -mIq;
177  mId = -mId;
178 
179  // Init stator currents
180  mIq = mIsr(3, 0);
181  mId = mIsr(0, 0);
182  mI0 = mIsr(6, 0);
183 
184  // Init stator voltages
185  mVq = mVsr(3, 0);
186  mVd = mVsr(0, 0);
187  mV0 = mVsr(6, 0);
188 
189  // Init magnetizing flux linkage
190  mPsikq1 = mPsisr(4, 0);
191  mPsikq2 = mPsisr(5, 0);
192  mPsikd = mPsisr(2, 0);
193  mPsifd = mPsisr(1, 0);
194  mPsimq = mPsisr(3, 0);
195  mPsimd = mPsisr(0, 0);
196 
198  mVfd = mVsr(1, 0);
199 
200  // #### VBR Model Dynamic variables #######################################
201  CalculateAuxiliarConstants(mTimeStep * mBase_OmElec);
202 
203  if (mNumDampingWindings == 2)
204  mPsimq = mDLmq * (mPsikq1 / mLlkq1 + mPsikq2 / mLlkq2 + mIq);
205  else
206  mPsimq = mDLmq * (mPsikq1 / mLlkq1 + mIq);
207 
208  mPsimd = mDLmd * (mPsifd / mLlfd + mPsikd / mLlkd + mId);
209 
210  mDqStatorCurrents << mIq, mId;
211 
212  mPsikq1kq2 << mPsikq1, mPsikq2;
213  mPsifdkd << mPsifd, mPsikd;
214 
215  CalculateAuxiliarVariables();
216  K1K2 << K1, K2;
217  mDVqd = K1K2 * mDqStatorCurrents + h_qdr;
218  mDVq = mDVqd(0);
219  mDVd = mDVqd(1);
220 
221  mDVa = inverseParkTransform(mThetaMech, mDVq, mDVd, 0)(0);
222  mDVb = inverseParkTransform(mThetaMech, mDVq, mDVd, 0)(1);
223  mDVc = inverseParkTransform(mThetaMech, mDVq, mDVd, 0)(2);
224 
225  mDVabc << mDVa, mDVb, mDVc;
226 
227  mVa = inverseParkTransform(mThetaMech, mVq, mVd, mV0)(0);
228  mVb = inverseParkTransform(mThetaMech, mVq, mVd, mV0)(1);
229  mVc = inverseParkTransform(mThetaMech, mVq, mVd, mV0)(2);
230 
231  mIa = inverseParkTransform(mThetaMech, mIq, mId, mI0)(0);
232  mIb = inverseParkTransform(mThetaMech, mIq, mId, mI0)(1);
233  mIc = inverseParkTransform(mThetaMech, mIq, mId, mI0)(2);
234 
235  CalculateL();
236 
237  SPDLOG_LOGGER_INFO(mSLog, "Initialize right side vector of size {}",
238  leftVector->get().rows());
239  SPDLOG_LOGGER_INFO(
240  mSLog, "Component affects right side vector entries {}, {} and {}",
241  matrixNodeIndex(0, 0), matrixNodeIndex(0, 1), matrixNodeIndex(0, 2));
242 
243  // set initial interface current
244  mVabc << mVa, mVb, mVc;
245  **mIntfVoltage = mVabc * mBase_V;
246 
247  // set initial interface voltage
248  mIabc << mIa, mIb, mIc;
249  **mIntfCurrent = mIabc * mBase_I;
250 }

304 void EMT::Ph3::SynchronGeneratorVBR::stepInPerUnit() {
305 
306  // Update of mechanical torque from turbine governor
307  if (mHasTurbineGovernor)
308  **mMechTorque = -mTurbineGovernor->step(
309  **mOmMech, 1, mInitElecPower.real() / mNomPower, mTimeStep);
310 
311  // Estimate mechanical variables with euler
312  **mElecTorque = (mPsimd * mIq - mPsimq * mId);
313  **mOmMech = **mOmMech + mTimeStep * (1. / (2. * **mInertia) *
314  (**mElecTorque - **mMechTorque));
315  mThetaMech = mThetaMech + mTimeStep * (**mOmMech * mBase_OmMech);
316 
317  // Calculate equivalent Resistance and current source
318  mVabc << mVa, mVb, mVc;
319 
320  mIabc << mIa, mIb, mIc;
321 
322  mEsh_vbr =
323  (mResistanceMat - (2 / (mTimeStep * mBase_OmElec)) * mDInductanceMat) *
324  mIabc +
325  mDVabc - mVabc;
326 
327  CalculateL();
328 
329  mPsikq1kq2 << mPsikq1, mPsikq2;
330  mPsifdkd << mPsifd, mPsikd;
331 
332  CalculateAuxiliarVariables();
333 
334  R_eq_vbr =
335  mResistanceMat + (2 / (mTimeStep * mBase_OmElec)) * mDInductanceMat + K;
336  E_eq_vbr = mEsh_vbr + E_r_vbr;
337 
338  MatrixFixedSize<3, 3> R_eq_vbr_mBase_Z = R_eq_vbr * mBase_Z;
339 
340  mConductanceMat = R_eq_vbr_mBase_Z.inverse();
341  mISourceEq = R_eq_vbr.inverse() * E_eq_vbr * mBase_I;
342 }

344 void EMT::Ph3::SynchronGeneratorVBR::mnaCompPostStep(
345  Real time, Int timeStepCount, Attribute<Matrix>::Ptr &leftVector) {
346  if (terminalNotGrounded(0)) {
347  mVa = Math::realFromVectorElement(*leftVector, matrixNodeIndex(0, 0)) /
348  mBase_V;
349  mVb = Math::realFromVectorElement(*leftVector, matrixNodeIndex(0, 1)) /
350  mBase_V;
351  mVc = Math::realFromVectorElement(*leftVector, matrixNodeIndex(0, 2)) /
352  mBase_V;
353  } else {
354  mVa = 0;
355  mVb = 0;
356  mVc = 0;
357  }
358 
359  // ################ Update machine stator and rotor variables ############################
360  mVabc << mVa, mVb, mVc;
361 
362  mVq = parkTransform(mThetaMech, mVa, mVb, mVc)(0);
363  mVd = parkTransform(mThetaMech, mVa, mVb, mVc)(1);
364  mV0 = parkTransform(mThetaMech, mVa, mVb, mVc)(2);
365 
366  if (mHasExciter) {
367  // Get exciter output voltage
368  // Note: scaled by Rfd/Lmd to transform from exciter pu system
369  // to the synchronous generator pu system
370  mVfd = (mRfd / mLmd) * mExciter->step(mVd, mVq, mTimeStep);
371  }
372  mIabc = R_eq_vbr.inverse() * (mVabc - E_eq_vbr);
373 
374  mIa = mIabc(0);
375  mIb = mIabc(1);
376  mIc = mIabc(2);
377 
378  mIq_hist = mIq;
379  mId_hist = mId;
380 
381  mIq = parkTransform(mThetaMech, mIa, mIb, mIc)(0);
382  mId = parkTransform(mThetaMech, mIa, mIb, mIc)(1);
383  mI0 = parkTransform(mThetaMech, mIa, mIb, mIc)(2);
384 
385  // Calculate rotor flux likanges
386  if (mNumDampingWindings == 2) {
387  mDqStatorCurrents << mIq, mId;
388 
389  mPsikq1kq2 = E1 * mIq + E2 * mPsikq1kq2 + E1 * mIq_hist;
390  mPsifdkd = F1 * mId + F2 * mPsifdkd + F1 * mId_hist + F3 * mVfd;
391 
392  mPsikq1 = mPsikq1kq2(0);
393  mPsikq2 = mPsikq1kq2(1);
394  mPsifd = mPsifdkd(0);
395  mPsikd = mPsifdkd(1);
396 
397  } else {
398 
399  mDqStatorCurrents << mIq, mId;
400 
401  mPsikq1 = E1_1d * mIq + E2_1d * mPsikq1 + E1_1d * mIq_hist;
402  mPsifdkd = F1 * mId + F2 * mPsifdkd + F1 * mId_hist + F3 * mVfd;
403 
404  mPsifd = mPsifdkd(0);
405  mPsikd = mPsifdkd(1);
406  }
407 
408  // Calculate dynamic flux likages
409  if (mNumDampingWindings == 2) {
410  mPsimq = mDLmq * (mPsikq1 / mLlkq1 + mPsikq2 / mLlkq2 + mIq);
411  } else {
412  mPsimq = mDLmq * (mPsikq1 / mLlkq1 + mIq);
413  }
414 
415  mPsimd = mDLmd * (mPsifd / mLlfd + mPsikd / mLlkd + mId);
416 
417  K1K2 << K1, K2;
418  mDVqd = K1K2 * mDqStatorCurrents + h_qdr;
419  mDVq = mDVqd(0);
420  mDVd = mDVqd(1);
421 
422  mDVa = inverseParkTransform(mThetaMech, mDVq, mDVd, 0)(0);
423  mDVb = inverseParkTransform(mThetaMech, mDVq, mDVd, 0)(1);
424  mDVc = inverseParkTransform(mThetaMech, mDVq, mDVd, 0)(2);
425  mDVabc << mDVa, mDVb, mDVc;
426 
427  **mIntfVoltage = mVabc * mBase_V;
428  **mIntfCurrent = mIabc * mBase_I;
429 }

442 void EMT::Ph3::SynchronGeneratorVBR::CalculateL() {
443  mDInductanceMat << **mLl + mLa - mLb * cos(2 * mThetaMech),
444  -mLa / 2 - mLb * cos(2 * mThetaMech - 2 * PI / 3),
445  -mLa / 2 - mLb * cos(2 * mThetaMech + 2 * PI / 3),
446  -mLa / 2 - mLb * cos(2 * mThetaMech - 2 * PI / 3),
447  **mLl + mLa - mLb * cos(2 * mThetaMech - 4 * PI / 3),
448  -mLa / 2 - mLb * cos(2 * mThetaMech),
449  -mLa / 2 - mLb * cos(2 * mThetaMech + 2 * PI / 3),
450  -mLa / 2 - mLb * cos(2 * mThetaMech),
451  **mLl + mLa - mLb * cos(2 * mThetaMech + 4 * PI / 3);
452 }

454 void EMT::Ph3::SynchronGeneratorVBR::CalculateAuxiliarConstants(Real dt) {
455 
456  b11 = (mRkq1 / mLlkq1) * (mDLmq / mLlkq1 - 1);
457  b13 = mRkq1 * mDLmq / mLlkq1;
458  b31 = (mRfd / mLlfd) * (mDLmd / mLlfd - 1);
459  b32 = mRfd * mDLmd / (mLlfd * mLlkd);
460  b33 = mRfd * mDLmd / mLlfd;
461  b41 = mRkd * mDLmd / (mLlfd * mLlkd);
462  b42 = (mRkd / mLlkd) * (mDLmd / mLlkd - 1);
463  b43 = mRkd * mDLmd / mLlkd;
464 
465  c23 = mDLmd * mRfd / (mLlfd * mLlfd) * (mDLmd / mLlfd - 1) +
466  mDLmd * mDLmd * mRkd / (mLlkd * mLlkd * mLlfd);
467  c24 = mDLmd * mRkd / (mLlkd * mLlkd) * (mDLmd / mLlkd - 1) +
468  mDLmd * mDLmd * mRfd / (mLlfd * mLlfd * mLlkd);
469  c25 = (mRfd / (mLlfd * mLlfd) + mRkd / (mLlkd * mLlkd)) * mDLmd * mDLmd;
470  c26 = mDLmd / mLlfd;
471 
472  if (mNumDampingWindings == 2) {
473  b12 = mRkq1 * mDLmq / (mLlkq1 * mLlkq2);
474  b21 = mRkq2 * mDLmq / (mLlkq1 * mLlkq2);
475  b22 = (mRkq2 / mLlkq2) * (mDLmq / mLlkq2 - 1);
476  b23 = mRkq2 * mDLmq / mLlkq2;
477  c11 = mDLmq * mRkq1 / (mLlkq1 * mLlkq1) * (mDLmq / mLlkq1 - 1) +
478  mDLmq * mDLmq * mRkq2 / (mLlkq2 * mLlkq2 * mLlkq1);
479  c12 = mDLmq * mRkq2 / (mLlkq2 * mLlkq2) * (mDLmq / mLlkq2 - 1) +
480  mDLmq * mDLmq * mRkq1 / (mLlkq1 * mLlkq1 * mLlkq2);
481  c15 =
