198void EMT::Ph3::VSIVoltageControlVCO::initializeParentFromNodesAndTerminals(
199 Real frequency) {
200
201 // use complex interface quantities for initialization calculations
202 MatrixComp intfVoltageComplex = Matrix::Zero(3, 1);
203 MatrixComp intfCurrentComplex = Matrix::Zero(3, 1);
204 // terminal powers in consumer system -> convert to generator system
205 Real activePower = terminal(0)->singlePower().real();
206 Real reactivePower = terminal(0)->singlePower().imag();
207
208 // derive complex threephase initialization from single phase initial values (only valid for balanced systems)
209 intfVoltageComplex(0, 0) = RMS3PH_TO_PEAK1PH * initialSingleVoltage(0);
210 intfVoltageComplex(1, 0) = intfVoltageComplex(0, 0) * SHIFT_TO_PHASE_B;
211 intfVoltageComplex(2, 0) = intfVoltageComplex(0, 0) * SHIFT_TO_PHASE_C;
212 intfCurrentComplex(0, 0) = -std::conj(
213 2. / 3. * Complex(activePower, reactivePower) / intfVoltageComplex(0, 0));
214 intfCurrentComplex(1, 0) = intfCurrentComplex(0, 0) * SHIFT_TO_PHASE_B;
215 intfCurrentComplex(2, 0) = intfCurrentComplex(0, 0) * SHIFT_TO_PHASE_C;
216
217 MatrixComp filterInterfaceInitialVoltage = MatrixComp::Zero(3, 1);
218 MatrixComp filterInterfaceInitialCurrent = MatrixComp::Zero(3, 1);
219 if (mWithConnectionTransformer) {
220 // calculate quantities of low voltage side of transformer (being the interface quantities of the filter, calculations only valid for symmetrical systems)
221 filterInterfaceInitialVoltage =
222 (intfVoltageComplex -
223 Complex(mTransformerResistance, mTransformerInductance * **mOmegaN) *
224 intfCurrentComplex) /
225 Complex(mTransformerRatioAbs, mTransformerRatioPhase);
226 filterInterfaceInitialCurrent =
227 intfCurrentComplex *
228 Complex(mTransformerRatioAbs, mTransformerRatioPhase);
229
230 // connect and init transformer
231 mVirtualNodes[3]->setInitialVoltage(PEAK1PH_TO_RMS3PH *
232 filterInterfaceInitialVoltage);
233 mConnectionTransformer->connect({mTerminals[0]->node(), mVirtualNodes[3]});
234 } else {
235 // if no transformer used, filter interface equal to inverter interface
236 filterInterfaceInitialVoltage = intfVoltageComplex;
237 filterInterfaceInitialCurrent = intfCurrentComplex;
238 }
239
240 // derive initialization quantities of filter (calculations only valid for symmetrical systems)
241 MatrixComp vcInit = filterInterfaceInitialVoltage -
242 filterInterfaceInitialCurrent * Complex(mRc, 0);
243 MatrixComp icfInit = vcInit * Complex(0., 2. * PI * frequency * mCf);
244 MatrixComp vfInit = vcInit - (filterInterfaceInitialCurrent - icfInit) *
245 Complex(0., 2. * PI * frequency * mLf);
246 MatrixComp vsInit =
247 vfInit - (filterInterfaceInitialCurrent - icfInit) * Complex(mRf, 0);
248 mVirtualNodes[0]->setInitialVoltage(PEAK1PH_TO_RMS3PH * vsInit);
249 mVirtualNodes[1]->setInitialVoltage(PEAK1PH_TO_RMS3PH * vfInit);
250 mVirtualNodes[2]->setInitialVoltage(PEAK1PH_TO_RMS3PH * vcInit);
251
252 // save real interface quantities calculated from complex ones
253 **mIntfVoltage = intfVoltageComplex.real();
254 **mIntfCurrent = intfCurrentComplex.real();
255 **mElecActivePower = (3. / 2. * intfVoltageComplex(0, 0) *
256 std::conj(-intfCurrentComplex(0, 0)))
257 .real();
258 **mElecReactivePower = (3. / 2. * intfVoltageComplex(0, 0) *
259 std::conj(-intfCurrentComplex(0, 0)))
260 .imag();
261
262 // Initialize controlled source
263 mSubCtrledVoltageSource->setParameters(mVirtualNodes[0]->initialVoltage(),
264 0.0);
265
266 // Connect electrical subcomponents
267 mSubCtrledVoltageSource->connect({SimNode::GND, mVirtualNodes[0]});
268 mSubResistorF->connect({mVirtualNodes[0], mVirtualNodes[1]});
269 mSubInductorF->connect({mVirtualNodes[1], mVirtualNodes[2]});
270 mSubCapacitorF->connect({mVirtualNodes[2], SimNode::GND});
271 if (mWithConnectionTransformer)
272 mSubResistorC->connect({mVirtualNodes[2], mVirtualNodes[3]});
273 else
274 mSubResistorC->connect({mVirtualNodes[2], mTerminals[0]->node()});
275
276 // Initialize electrical subcomponents
277 for (auto subcomp : mSubComponents) {
278 subcomp->initialize(mFrequencies);
279 subcomp->initializeFromNodesAndTerminals(frequency);
280 }
281
282 if (mWithConnectionTransformer) {
283 // Initialize control subcomponents
284 // current and voltage inputs to VCO and voltage controller
285 Matrix vcdq, ircdq;
286 Real theta = std::arg(mVirtualNodes[3]->initialSingleVoltage());
287 vcdq = Math::parkTransformPowerInvariant(
288 theta, filterInterfaceInitialVoltage.real());
289 ircdq = Math::parkTransformPowerInvariant(
290 theta, -1 * filterInterfaceInitialCurrent.real());
291
292 **mVcd = vcdq(0, 0);
293 **mVcq = vcdq(1, 0);
294 **mIrcd = ircdq(0, 0);
295 **mIrcq = ircdq(1, 0);
296
297 // VCO start settings
298 mVCO->setInitialValues(**mOmegaN, theta, theta);
299 } else {
300 // Initialize control subcomponents
301 // current and voltage inputs to VCO and voltage controller
302 Matrix vcdq, ircdq;
303 Real theta = std::arg(mVirtualNodes[2]->initialSingleVoltage());
304 vcdq = Math::parkTransformPowerInvariant(
305 theta, filterInterfaceInitialVoltage.real());
306 ircdq = Math::parkTransformPowerInvariant(
307 theta, -1 * filterInterfaceInitialCurrent.real());
308
309 **mVcd = vcdq(0, 0);
310 **mVcq = vcdq(1, 0);
311 **mIrcd = ircdq(0, 0);
312 **mIrcq = ircdq(1, 0);
313
314 // VCO start settings
315 mVCO->setInitialValues(**mOmegaN, theta, theta);
316 }
317 SPDLOG_LOGGER_INFO(
318 mSLog,
319 "\n--- Initialization from powerflow ---"
320 "\nInterface voltage across: {:s}"
321 "\nInterface current: {:s}"
322 "\nTerminal 0 initial voltage: {:s}"
323 "\nTerminal 0 connected to {:s} = sim node {:d}"
324 "\nVirtual node 0 initial voltage: {:s}"
325 "\nVirtual node 1 initial voltage: {:s}"
326 "\nVirtual node 2 initial voltage: {:s}",
327 Logger::phasorToString(intfVoltageComplex(0, 0)),
328 Logger::phasorToString(intfCurrentComplex(0, 0)),
329 Logger::phasorToString(initialSingleVoltage(0)),
330 mTerminals[0]->node()->name(), mTerminals[0]->node()->matrixNodeIndex(),
331 Logger::phasorToString(mVirtualNodes[0]->initialSingleVoltage()),
332 Logger::phasorToString(mVirtualNodes[1]->initialSingleVoltage()),
333 Logger::phasorToString(mVirtualNodes[2]->initialSingleVoltage()));
334 if (mWithConnectionTransformer)
335 SPDLOG_LOGGER_INFO(
336 mSLog, "\nVirtual node 3 initial voltage: {:s}",
337 Logger::phasorToString(mVirtualNodes[3]->initialSingleVoltage()));
338 SPDLOG_LOGGER_INFO(mSLog, "\n--- Initialization from powerflow finished ---");
339}

387void EMT::Ph3::VSIVoltageControlVCO::controlStep(Real time, Int timeStepCount) {
388 // Transformation interface forward
389 Matrix vcdq, ircdq;
390 Real theta = mVCO->mOutputPrev->get();
391 vcdq = Math::parkTransformPowerInvariant(theta, **mVirtualNodes[2]->mVoltage);
392 ircdq =
393 Math::parkTransformPowerInvariant(theta, -**mSubResistorF->mIntfCurrent);
394 Matrix intfVoltageDQ =
395 Math::parkTransformPowerInvariant(mThetaN, **mIntfVoltage);
396 Matrix intfCurrentDQ =
397 Math::parkTransformPowerInvariant(mThetaN, **mIntfCurrent);
398 **mElecActivePower = -1. * (intfVoltageDQ(0, 0) * intfCurrentDQ(0, 0) +
399 intfVoltageDQ(1, 0) * intfCurrentDQ(1, 0));
400 **mElecReactivePower = -1. * (intfVoltageDQ(1, 0) * intfCurrentDQ(0, 0) -
401 intfVoltageDQ(0, 0) * intfCurrentDQ(1, 0));
402 //vector of voltages
403 **mVcd = vcdq(0, 0);
404 **mVcq = vcdq(1, 0);
405
406 //vector of currents
407 **mIrcd = ircdq(0, 0);
408 **mIrcq = ircdq(1, 0);
409
410 // add step of subcomponents
411 mVCO->signalStep(time, timeStepCount);
412 mVoltageControllerVSI->signalStep(time, timeStepCount);
413
414 // Transformation interface backward
415 **mVsref = Math::inverseParkTransformPowerInvariant(
416 mVCO->mOutputPrev->get(), mVoltageControllerVSI->mOutputCurr->get());
417 mThetaN = std::fmod(mThetaN + mTimeStep * **mOmegaN, 2 * PI);
418}
