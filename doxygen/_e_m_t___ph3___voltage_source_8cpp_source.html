26void EMT::Ph3::VoltageSource::setParameters(MatrixComp voltageRef,
27 Real srcFreq) {
28 auto srcSigSine = Signal::SineWaveGenerator::make(**mName + "_sw");
29 // Complex(1,0) is used as initialPhasor, since magnitude and phase of V_ref are taken into account by updateVoltage

37 "\nVoltage reference phasor [V]: {:s}"
38 "\nFrequency [Hz]: {:s}",

40 Logger::realToString(srcFreq));

78void EMT::Ph3::VoltageSource::initializeFromNodesAndTerminals(Real frequency) {
79 SPDLOG_LOGGER_INFO(mSLog, "\n--- Initialization from node voltages ---");
80 // TODO: this approach currently overwrites voltage reference set from outside, when not using setParameters
81 if (!mParametersSet) {
82 auto srcSigSine = Signal::SineWaveGenerator::make(**mName + "_sw");
83 srcSigSine->mFreq->setReference(mSrcFreq);
84 // Complex(1,0) is used as initialPhasor for signal generator as only phase is used
85 srcSigSine->setParameters(Complex(1, 0), frequency);
86 mSrcSig = srcSigSine;
87
88 **mVoltageRef = CPS::Math::singlePhaseVariableToThreePhase(
89 initialSingleVoltage(1) - initialSingleVoltage(0));
90
91 SPDLOG_LOGGER_INFO(mSLog,
92 "\nReference voltage: {:s}"
93 "\nTerminal 0 voltage: {:s}"
94 "\nTerminal 1 voltage: {:s}",
95 Logger::matrixCompToString(**mVoltageRef),
96 Logger::phasorToString(initialSingleVoltage(0)),
97 Logger::phasorToString(initialSingleVoltage(1)));
98 } else {
99 SPDLOG_LOGGER_INFO(
100 mSLog,
101 "\nInitialization from node voltages omitted (parameter already set)."
102 "\nReference voltage: {:s}",
103 Logger::matrixCompToString(**mVoltageRef));
104 }
105 SPDLOG_LOGGER_INFO(mSLog, "\n--- Initialization from node voltages ---");
106 mSLog->flush();
107}

192void EMT::Ph3::VoltageSource::updateVoltage(Real time) {
193 if (mSrcSig != nullptr) {
194 mSrcSig->step(time);
195 for (int i = 0; i < 3; i++) {
196 (**mIntfVoltage)(i, 0) = RMS3PH_TO_PEAK1PH *
197 Math::abs((**mVoltageRef)(i, 0)) *
198 cos(Math::phase(mSrcSig->getSignal()) +
199 Math::phase((**mVoltageRef)(i, 0)));
200 }
201 } else {
202 **mIntfVoltage = RMS3PH_TO_PEAK1PH * (**mVoltageRef).real();
203 }
204 SPDLOG_LOGGER_DEBUG(mSLog, "\nUpdate Voltage: {:s}",
205 Logger::matrixToString(**mIntfVoltage));
206}
