47void EMT::Ph3::VoltageSourceNorton::mnaCompInitialize(
48 Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
49 updateMatrixNodeIndices();
50 Complex voltageRef = mVoltageRef->get();
51 (**mIntfVoltage)(0, 0) = voltageRef.real() * cos(Math::phase(voltageRef));
52 (**mIntfVoltage)(1, 0) =
53 voltageRef.real() * cos(Math::phase(voltageRef) - 2. / 3. * M_PI);
54 (**mIntfVoltage)(2, 0) =
55 voltageRef.real() * cos(Math::phase(voltageRef) + 2. / 3. * M_PI);
56}

58void EMT::Ph3::VoltageSourceNorton::mnaCompApplySystemMatrixStamp(
59 SparseMatrixRow &systemMatrix) {
60 // Apply matrix stamp for equivalent resistance
61 if (terminalNotGrounded(0)) {
62 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 0),
63 matrixNodeIndex(0, 0), mConductance);
64 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 1),
65 matrixNodeIndex(0, 1), mConductance);
66 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 2),
67 matrixNodeIndex(0, 2), mConductance);
68 }
69 if (terminalNotGrounded(1)) {
70 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 0),
71 matrixNodeIndex(1, 0), mConductance);
72 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 1),
73 matrixNodeIndex(1, 1), mConductance);
74 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 2),
75 matrixNodeIndex(1, 2), mConductance);
76 }
77 if (terminalNotGrounded(0) && terminalNotGrounded(1)) {
78 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 0),
79 matrixNodeIndex(1, 0), -mConductance);
80 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 0),
81 matrixNodeIndex(0, 0), -mConductance);
82
83 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 1),
84 matrixNodeIndex(1, 1), -mConductance);
85 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 1),
86 matrixNodeIndex(0, 1), -mConductance);
87
88 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 2),
89 matrixNodeIndex(1, 2), -mConductance);
90 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 2),
91 matrixNodeIndex(0, 2), -mConductance);
92 }
93}

95void EMT::Ph3::VoltageSourceNorton::mnaCompApplyRightSideVectorStamp(
96 Matrix &rightVector) {
97 // Apply matrix stamp for equivalent current source
98 if (terminalNotGrounded(0)) {
99 Math::setVectorElement(rightVector, matrixNodeIndex(0, 0),
100 -mEquivCurrent(0, 0));
101 Math::setVectorElement(rightVector, matrixNodeIndex(0, 1),
102 -mEquivCurrent(1, 0));
103 Math::setVectorElement(rightVector, matrixNodeIndex(0, 2),
104 -mEquivCurrent(2, 0));
105 }
106 if (terminalNotGrounded(1)) {
107 Math::setVectorElement(rightVector, matrixNodeIndex(1, 0),
108 mEquivCurrent(0, 0));
109 Math::setVectorElement(rightVector, matrixNodeIndex(1, 1),
110 mEquivCurrent(1, 0));
111 Math::setVectorElement(rightVector, matrixNodeIndex(1, 2),
112 mEquivCurrent(2, 0));
113 }

116void EMT::Ph3::VoltageSourceNorton::updateState(Real time) {
117 // Check if set source was called
118 if (Math::abs(**mVoltageRef) > 0) {
119 (**mIntfVoltage)(0, 0) =
120 Math::abs(**mVoltageRef) *
121 cos(2. * PI * **mSrcFreq * time + Math::phase(**mVoltageRef));
122 (**mIntfVoltage)(1, 0) = Math::abs(**mVoltageRef) *
123 cos(2. * PI * **mSrcFreq * time +
124 Math::phase(**mVoltageRef) - 2. / 3. * M_PI);
125 (**mIntfVoltage)(2, 0) = Math::abs(**mVoltageRef) *
126 cos(2. * PI * **mSrcFreq * time +
127 Math::phase(**mVoltageRef) + 2. / 3. * M_PI);
128 }
129
130 mEquivCurrent = **mIntfVoltage / **mResistance;
131}

133void EMT::Ph3::VoltageSourceNorton::mnaCompUpdateVoltage(
134 const Matrix &leftVector) {
135 // Calculate v1 - v0
136 if (terminalNotGrounded(1)) {
137 (**mIntfVoltage)(0, 0) =
138 Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 0));
139 (**mIntfVoltage)(1, 0) =
140 Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 1));
141 (**mIntfVoltage)(2, 0) =
142 Math::realFromVectorElement(leftVector, matrixNodeIndex(1, 2));
143 }
144 if (terminalNotGrounded(0)) {
145 (**mIntfVoltage)(0, 0) =
146 (**mIntfVoltage)(0, 0) -
147 Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 0));
148 (**mIntfVoltage)(1, 0) =
149 (**mIntfVoltage)(1, 0) -
150 Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 1));
151 (**mIntfVoltage)(2, 0) =
152 (**mIntfVoltage)(2, 0) -
153 Math::realFromVectorElement(leftVector, matrixNodeIndex(0, 2));
154 }
155}

187void EMT::Ph3::VoltageSourceNorton::mnaCompUpdateCurrent(
188 const Matrix &leftVector) {
189 // signs are not verified
190 (**mIntfCurrent)(0, 0) =
191 mEquivCurrent(0, 0) - (**mIntfVoltage)(0, 0) / **mResistance;
192 (**mIntfCurrent)(1, 0) =
193 mEquivCurrent(1, 0) - (**mIntfVoltage)(1, 0) / **mResistance;
194 (**mIntfCurrent)(2, 0) =
195 mEquivCurrent(2, 0) - (**mIntfVoltage)(2, 0) / **mResistance;
196}
