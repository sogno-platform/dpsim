38void EMT::VTypeVariableSSNComp::initializeFromNodesAndTerminals(
39 Real frequency) {
40 if (!mParametersSet)
41 throw std::logic_error("setParameters() must be called before "
42 "initializeFromNodesAndTerminals().");
43
44 const MatrixComp uInit = buildInitialInputFromNodes(frequency);
45
46 **mIntfVoltage = uInit.real();
47 **mX = Matrix::Zero((**mX).rows(), 1);
48
49 SPDLOG_LOGGER_INFO(mSLog,
50 "\n--- Initialization from nodes and terminals ---");
51
52 // Fixed-point initialization for operating-point-dependent SSN models.
53 // This keeps the implementation generic while still allowing the component
54 // parameters to depend on the current state.
55 constexpr Int maxIterations = 10;
56 constexpr Real tolerance = 1e-9;
57 Bool converged = false;
58
59 for (Int iter = 0; iter < maxIterations; ++iter) {
60 const Matrix xPrev = **mX;
61
62 updateComponentParameters();
63
64 const MatrixComp xInit =
65 calculateSteadyStateStateFromInput(uInit, frequency);
66 **mX = xInit.real();
67
68 if ((**mX).isApprox(xPrev, tolerance)) {
69 converged = true;
70 break;
71 }
72 }
73
74 if (!converged) {
75 SPDLOG_LOGGER_WARN(
76 mSLog,
77 "Fixed-point initialization did not converge within {} iterations.",
78 maxIterations);
79 }
80
81 // Ensure the output matrices correspond to the final initialized state.
82 updateComponentParameters();
83
84 const MatrixComp xInit = (**mX).cast<Complex>();
85 const MatrixComp yInit = mC.cast<Complex>() * xInit +
86 mD.cast<Complex>() * uInit + mF.cast<Complex>();
87
88 **mIntfCurrent = yInit.real();
89 mParameterChanged = false;
90
91 SPDLOG_LOGGER_INFO(
92 mSLog,
93 "\nInput u: {:s}"
94 "\nOutput y: {:s}"
95 "\nState x: {:s}"
96 "\n--- Initialization from nodes and terminals finished ---",
97 Logger::matrixToString(**mIntfVoltage),
98 Logger::matrixToString(**mIntfCurrent), Logger::matrixToString(**mX));
99}
