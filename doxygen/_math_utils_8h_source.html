126  static void setMatrixElement(Matrix& mat, Matrix::Index row, Matrix::Index column, Complex value, Int maxFreq = 1, Int freqIdx = 0) {
127  // Assume square matrix
128  Eigen::Index harmonicOffset = mat.rows() / maxFreq;
129  Eigen::Index complexOffset = harmonicOffset / 2;
130  Eigen::Index harmRow = row + harmonicOffset * freqIdx;
131  Eigen::Index harmCol = column + harmonicOffset * freqIdx;
132 
133  mat(harmRow, harmCol) = value.real();
134  mat(harmRow + complexOffset, harmCol + complexOffset) = value.real();
135  mat(harmRow, harmCol + complexOffset) = - value.imag();
136  mat(harmRow + complexOffset, harmCol) = value.imag();
137  }
