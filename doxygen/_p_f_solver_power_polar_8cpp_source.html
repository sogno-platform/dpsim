23void PFSolverPowerPolar::generateInitialSolution(Real time,
24 bool keep_last_solution) {
25 // Undo Q-limit PV<->PQ conversions left over from a previous solve.
26 if (mEnforceReactiveLimits)
27 resetToOriginalClassification();
28
29 const UInt n = mSystem.mNodes.size();
30
31 bool can_keep = keep_last_solution && mHasLastConvergedSolution &&
32 mLastConvergedV.size() == n && mLastConvergedD.size() == n;
33
34 SPDLOG_LOGGER_INFO(mSLog,
35 "PF initialization: keep_last_solution={}, can_keep={}",
36 keep_last_solution, can_keep);
37
38 resize_sol(n);
39 resize_complex_sol(n);
40
41 if (can_keep) {
42 sol_V = mLastConvergedV;
43 sol_D = mLastConvergedD;
44 }
45
46 // update components
47 for (auto comp : mSystem.mComponents) {
48 if (auto load = std::dynamic_pointer_cast<CPS::SP::Ph1::Load>(comp)) {
49 if (load->use_profile)
50 load->updatePQ(time);
51
52 load->calculatePerUnitParameters(mBaseApparentPower,
53 mSystem.mSystemOmega);
54 }
55
56 if (auto gen =
57 std::dynamic_pointer_cast<CPS::SP::Ph1::SynchronGenerator>(comp)) {
58 gen->calculatePerUnitParameters(mBaseApparentPower, mSystem.mSystemOmega);
59 }
60 }
61
62 // PQ buses
63 for (auto pq : mPQBuses) {
64 UInt idx = pq->matrixNodeIndex();
65
66 sol_P(idx) = 0.0;
67 sol_Q(idx) = 0.0;
68
69 if (!can_keep) {
70 sol_V(idx) = 1.0;
71 sol_D(idx) = 0.0;
72 }
73
74 for (auto comp : mSystem.mComponentsAtNode[pq]) {
75 if (auto load = std::dynamic_pointer_cast<CPS::SP::Ph1::Load>(comp)) {
76 sol_P(idx) -= load->attributeTyped<CPS::Real>("P_pu")->get();
77 sol_Q(idx) -= load->attributeTyped<CPS::Real>("Q_pu")->get();
78 } else if (auto sst = std::dynamic_pointer_cast<
79 CPS::SP::Ph1::SolidStateTransformer>(comp)) {
80 sol_P(idx) -= sst->getNodalInjection(pq).real();
81 sol_Q(idx) -= sst->getNodalInjection(pq).imag();
82 } else if (auto vsi = std::dynamic_pointer_cast<
83 CPS::SP::Ph1::AvVoltageSourceInverterDQ>(comp)) {
84 sol_P(idx) +=
85 vsi->attributeTyped<CPS::Real>("P_ref")->get() / mBaseApparentPower;
86 sol_Q(idx) +=
87 vsi->attributeTyped<CPS::Real>("Q_ref")->get() / mBaseApparentPower;
88 } else if (auto gen =
89 std::dynamic_pointer_cast<CPS::SP::Ph1::SynchronGenerator>(
90 comp)) {
91 sol_P(idx) += gen->attributeTyped<CPS::Real>("P_set_pu")->get();
92 sol_Q(idx) += gen->attributeTyped<CPS::Real>("Q_set_pu")->get();
93 }
94 }
95
96 sol_S_complex(idx) = CPS::Complex(sol_P(idx), sol_Q(idx));
97 sol_V_complex(idx) = Math::polar(sol_V(idx), sol_D(idx));
98 }
99
100 // PV buses
101 for (auto pv : mPVBuses) {
102 UInt idx = pv->matrixNodeIndex();
103
104 sol_P(idx) = 0.0;
105 sol_Q(idx) = 0.0;
106
107 if (!can_keep) {
108 sol_D(idx) = 0.0;
109 sol_V(idx) = 1.0;
110 }
111
112 for (auto comp : mSystem.mComponentsAtNode[pv]) {
113 if (auto gen = std::dynamic_pointer_cast<CPS::SP::Ph1::SynchronGenerator>(
114 comp)) {
115 sol_P(idx) += gen->attributeTyped<CPS::Real>("P_set_pu")->get();
116 sol_V(idx) = gen->attributeTyped<CPS::Real>("V_set_pu")->get();
117 } else if (auto load =
118 std::dynamic_pointer_cast<CPS::SP::Ph1::Load>(comp)) {
119 sol_P(idx) -= load->attributeTyped<CPS::Real>("P_pu")->get();
120 sol_Q(idx) -= load->attributeTyped<CPS::Real>("Q_pu")->get();
121 } else if (auto vsi = std::dynamic_pointer_cast<
122 CPS::SP::Ph1::AvVoltageSourceInverterDQ>(comp)) {
123 sol_P(idx) +=
124 vsi->attributeTyped<CPS::Real>("P_ref")->get() / mBaseApparentPower;
125 } else if (auto extnet =
126 std::dynamic_pointer_cast<CPS::SP::Ph1::NetworkInjection>(
127 comp)) {
128 sol_P(idx) += extnet->attributeTyped<CPS::Real>("p_inj")->get() /
129 mBaseApparentPower;
130 sol_V(idx) = extnet->attributeTyped<CPS::Real>("V_set_pu")->get();
131 }
132 }
133
134 sol_S_complex(idx) = CPS::Complex(sol_P(idx), sol_Q(idx));
135 sol_V_complex(idx) = Math::polar(sol_V(idx), sol_D(idx));
136 }
137
138 // VD / slack buses
139 for (auto vd : mVDBuses) {
140 UInt idx = vd->matrixNodeIndex();
141
142 sol_P(idx) = 0.0;
143 sol_Q(idx) = 0.0;
144 sol_D(idx) = 0.0;
145 sol_V(idx) = 1.0;
146
147 for (auto comp : mSystem.mComponentsAtNode[vd]) {
148 if (auto extnet =
149 std::dynamic_pointer_cast<CPS::SP::Ph1::NetworkInjection>(comp)) {
150 sol_V(idx) = extnet->attributeTyped<CPS::Real>("V_set_pu")->get();
151 } else if (auto load =
152 std::dynamic_pointer_cast<CPS::SP::Ph1::Load>(comp)) {
153 sol_P(idx) -= load->attributeTyped<CPS::Real>("P_pu")->get();
154 sol_Q(idx) -= load->attributeTyped<CPS::Real>("Q_pu")->get();
155 } else if (auto gen =
156 std::dynamic_pointer_cast<CPS::SP::Ph1::SynchronGenerator>(
157 comp)) {
158 sol_P(idx) += gen->attributeTyped<CPS::Real>("P_set_pu")->get();
159 sol_Q(idx) += gen->attributeTyped<CPS::Real>("Q_set_pu")->get();
160 sol_V(idx) = gen->attributeTyped<CPS::Real>("V_set_pu")->get();
161 }
162 }
163
164 sol_S_complex(idx) = CPS::Complex(sol_P(idx), sol_Q(idx));
165 sol_V_complex(idx) = Math::polar(sol_V(idx), sol_D(idx));
166 }
167
168 solutionInitialized = true;
169 solutionComplexInitialized = true;
170
171 Pesp = sol_P;
172 Qesp = sol_Q;
173}

175void PFSolverPowerPolar::calculateMismatch() {
176 UInt npqpv = mNumPQBuses + mNumPVBuses;
177 UInt k;
178 mF.setZero();
179
180 for (UInt a = 0; a < npqpv; ++a) {
181 // For PQ and PV buses calculate active power mismatch
182 k = mPQPVBusIndices[a];
183 mF(a) = Pesp.coeff(k) - P(k);
184
185 //only for PQ buses calculate reactive power mismatch
186 if (a < mNumPQBuses)
187 mF(a + npqpv) = Qesp.coeff(k) - Q(k);
188 }
189}

191void PFSolverPowerPolar::calculateJacobian() {
192 UInt npqpv = mNumPQBuses + mNumPVBuses;
193 double val;
194 UInt k, j;
195 UInt da, db;
196
197 mJ.setZero();
198
199 //J1
200 for (UInt a = 0; a < npqpv; ++a) { //rows
201 k = mPQPVBusIndices[a];
202 //diagonal
203 mJ.coeffRef(a, a) = -Q(k) - B(k, k) * sol_V.coeff(k) * sol_V.coeff(k);
204
205 //non diagonal elements
206 for (UInt b = 0; b < npqpv; ++b) {
207 if (b != a) {
208 j = mPQPVBusIndices[b];
209 val = sol_V.coeff(k) * sol_V.coeff(j) *
210 (G(k, j) * sin(sol_D.coeff(k) - sol_D.coeff(j)) -
211 B(k, j) * cos(sol_D.coeff(k) - sol_D.coeff(j)));
212 //if (val != 0.0)
213 mJ.coeffRef(a, b) = val;
214 }
215 }
216 }
217
218 //J2
219 da = 0;
220 db = npqpv;
221 for (UInt a = 0; a < npqpv; ++a) { //rows
222 k = mPQPVBusIndices[a];
223 //diagonal
224 //std::cout << "J2D:" << (a + da) << "," << (a + db) << std::endl;
225 if (a < mNumPQBuses)
226 mJ.coeffRef(a + da, a + db) =
227 P(k) + G(k, k) * sol_V.coeff(k) * sol_V.coeff(k);
228
229 //non diagonal elements
230 for (UInt b = 0; b < mNumPQBuses; ++b) {
231 if (b != a) {
232 j = mPQPVBusIndices[b];
233 val = sol_V.coeff(k) * sol_V.coeff(j) *
234 (G(k, j) * cos(sol_D.coeff(k) - sol_D.coeff(j)) +
235 B(k, j) * sin(sol_D.coeff(k) - sol_D.coeff(j)));
236 //if (val != 0.0)
237 //std::cout << "J2ij:" << (a + da) << "," << (b + db) << std::endl;
238 mJ.coeffRef(a + da, b + db) = val;
239 }
240 }
241 }
242
243 //J3
244 da = npqpv;
245 db = 0;
246 for (UInt a = 0; a < mNumPQBuses; ++a) { //rows
247 k = mPQPVBusIndices[a];
248 //diagonal
249 //std::cout << "J3:" << (a + da) << "," << (a + db) << std::endl;
250 mJ.coeffRef(a + da, a + db) =
251 P(k) - G(k, k) * sol_V.coeff(k) * sol_V.coeff(k);
252
253 //non diagonal elements
254 for (UInt b = 0; b < npqpv; ++b) {
255 if (b != a) {
256 j = mPQPVBusIndices[b];
257 val = sol_V.coeff(k) * sol_V.coeff(j) *
258 (G(k, j) * cos(sol_D.coeff(k) - sol_D.coeff(j)) +
259 B(k, j) * sin(sol_D.coeff(k) - sol_D.coeff(j)));
260 //if (val != 0.0)
261 //std::cout << "J3:" << (a + da) << "," << (b + db) << std::endl;
262 mJ.coeffRef(a + da, b + db) = -val;
263 }
264 }
265 }
266
267 //J4
268 da = npqpv;
269 db = npqpv;
270 for (UInt a = 0; a < mNumPQBuses; ++a) { //rows
271 k = mPQPVBusIndices[a];
272 //diagonal
273 //std::cout << "J4:" << (a + da) << "," << (a + db) << std::endl;
274 mJ.coeffRef(a + da, a + db) =
275 Q(k) - B(k, k) * sol_V.coeff(k) * sol_V.coeff(k);
276
277 //non diagonal elements
278 for (UInt b = 0; b < mNumPQBuses; ++b) {
279 if (b != a) {
280 j = mPQPVBusIndices[b];
281 val = sol_V.coeff(k) * sol_V.coeff(j) *
282 (G(k, j) * sin(sol_D.coeff(k) - sol_D.coeff(j)) -
283 B(k, j) * cos(sol_D.coeff(k) - sol_D.coeff(j)));
284 if (val != 0.0) {
285 //std::cout << "J4:" << (a + da) << "," << (b + db) << std::endl;
286 mJ.coeffRef(a + da, b + db) = val;
287 }
288 }
289 }
290 }

293void PFSolverPowerPolar::updateSolution() {
294 UInt npqpv = mNumPQBuses + mNumPVBuses;
295
296 // Scale the whole step by one factor to bound the max change without altering direction.
297 const double maxDVpu = 0.1; // max |dV| per step [pu]
298 const double maxDThetaRad = 0.2; // max |dTheta| per step [rad]
299
300 // mX: [0,npqpv) angle incr (PQ+PV), then rel. voltage dV/V (PQ only).
301 double scale = 1.0;
302 for (UInt a = 0; a < npqpv; ++a) {
303 double dTheta = std::abs(mX.coeff(a));
304 if (dTheta > maxDThetaRad)
305 scale = std::min(scale, maxDThetaRad / dTheta);
306 }
307 for (UInt b = 0; b < mNumPQBuses; ++b) {
308 double dVrel = std::abs(mX.coeff(npqpv + b));
309 if (dVrel > maxDVpu)
310 scale = std::min(scale, maxDVpu / dVrel);
311 }
312
313 for (UInt a = 0; a < npqpv; ++a) {
314 UInt k = mPQPVBusIndices[a];
315 sol_D(k) += scale * mX.coeff(a);
316 // additive-relative update, consistent with the Jacobian
317 if (a < mNumPQBuses)
318 sol_V(k) *= (1.0 + scale * mX.coeff(a + npqpv));
319 }
320
321 for (auto node : mSystem.mNodes) {
322 UInt idx = node->matrixNodeIndex();
323 sol_V_complex(idx) = Math::polar(sol_V(idx), sol_D(idx));
324 }
325}

327void PFSolverPowerPolar::setSolution() {
328
329 if (!isConverged) {
330 SPDLOG_LOGGER_WARN(mSLog, "Not converged within {} iterations",
331 mIterations);
332
333 SPDLOG_LOGGER_WARN(mSLog, "Writing last iterate to result state "
334 "(not stored as warm-start solution).");
335 } else {
336 calculatePAndQAtSlackBus();
337 calculateQAtPVBuses();
338 calculatePAndQInjectionPQBuses();
339
340 mLastConvergedV = sol_V;
341 mLastConvergedD = sol_D;
342 mHasLastConvergedSolution = true;
343
344 SPDLOG_LOGGER_INFO(mSLog, "Converged in {} iterations", mIterations);
345 }
346
347 SPDLOG_LOGGER_INFO(mSLog, "Solution written to result state:");
348
349 SPDLOG_LOGGER_INFO(mSLog, "Name\tP\t\tQ\t\tV\t\tD");
350
351 for (auto node : mSystem.mNodes) {
352 UInt idx = node->matrixNodeIndex();
353
354 SPDLOG_LOGGER_INFO(
355 mSLog, "{}\t{}\t{}\t{}\t{}",
356 std::dynamic_pointer_cast<CPS::SimNode<CPS::Complex>>(node)->name(),
357 sol_P[idx], sol_Q[idx], sol_V[idx], sol_D[idx]);
358 }
359
360 mSLog->flush();
361
362 for (UInt i = 0; i < mSystem.mNodes.size(); ++i) {
363 sol_S_complex(i) = CPS::Complex(sol_P.coeff(i), sol_Q.coeff(i));
364 sol_V_complex(i) = Math::polar(sol_V.coeff(i), sol_D.coeff(i));
365 }
366
367 for (auto node : mSystem.mNodes) {
368 auto simNode = std::dynamic_pointer_cast<CPS::SimNode<CPS::Complex>>(node);
369
370 UInt idx = node->matrixNodeIndex();
371
372 simNode->setVoltage(sol_V_complex(idx) * mBaseVoltageAtNode[node]);
373
374 simNode->setPower(sol_S_complex(idx) * mBaseApparentPower);
375 }
376
377 calculateBranchFlow();
378 calculateNodalInjection();
379}

381void PFSolverPowerPolar::calculateBranchFlow() {
382 for (auto line : mLines) {
383 VectorComp v(2);
384 v(0) = sol_V_complex.coeff(line->node(0)->matrixNodeIndex());
385 v(1) = sol_V_complex.coeff(line->node(1)->matrixNodeIndex());
387 VectorComp current = line->Y_element() * v;
389 VectorComp flow_on_branch =
390 Math::elementwiseProduct(v, current.conjugate());
391 line->updateBranchFlow(current, flow_on_branch);
392 }
393 for (auto trafo : mTransformers) {
394 VectorComp v(2);
395 v(0) = sol_V_complex.coeff(trafo->node(0)->matrixNodeIndex());
396 v(1) = sol_V_complex.coeff(trafo->node(1)->matrixNodeIndex());
398 VectorComp current = trafo->Y_element() * v;
400 VectorComp flow_on_branch =
401 Math::elementwiseProduct(v, current.conjugate());
402 trafo->updateBranchFlow(current, flow_on_branch);
403 }
404}

406void PFSolverPowerPolar::calculateNodalInjection() {
407 for (auto node : mSystem.mNodes) {
408 std::list<std::shared_ptr<CPS::SP::Ph1::PiLine>> lines;
409 for (auto comp : mSystem.mComponentsAtNode[node]) {
410 if (std::shared_ptr<CPS::SP::Ph1::PiLine> line =
411 std::dynamic_pointer_cast<CPS::SP::Ph1::PiLine>(comp)) {
412 line->storeNodalInjection(sol_S_complex.coeff(node->matrixNodeIndex()));
413 lines.push_back(line);
414 break;
415 }
416 }
417 if (lines.empty()) {
418 for (auto comp : mSystem.mComponentsAtNode[node]) {
419 if (std::shared_ptr<CPS::SP::Ph1::Transformer> trafo =
420 std::dynamic_pointer_cast<CPS::SP::Ph1::Transformer>(comp)) {
421 trafo->storeNodalInjection(
422 sol_S_complex.coeff(node->matrixNodeIndex()));
423 break;
424 }
425 }
426 }
427 }

430Real PFSolverPowerPolar::P(UInt k) {
431 Real val = 0.0;
432 for (UInt j = 0; j < mSystem.mNodes.size(); ++j) {
433 val += sol_V.coeff(j) * (G(k, j) * cos(sol_D.coeff(k) - sol_D.coeff(j)) +
434 B(k, j) * sin(sol_D.coeff(k) - sol_D.coeff(j)));
435 }
436 return sol_V.coeff(k) * val;
437}
438
439Real PFSolverPowerPolar::Q(UInt k) {
440 Real val = 0.0;
441 for (UInt j = 0; j < mSystem.mNodes.size(); ++j) {
442 val += sol_V.coeff(j) * (G(k, j) * sin(sol_D.coeff(k) - sol_D.coeff(j)) -
443 B(k, j) * cos(sol_D.coeff(k) - sol_D.coeff(j)));
444 }
445 return sol_V.coeff(k) * val;
446}

448void PFSolverPowerPolar::calculatePAndQAtSlackBus() {
449 for (auto topoNode : mVDBuses) {
450 auto node_idx = topoNode->matrixNodeIndex();
451
452 // Net nodal injection into the network: S_inj = V * conj(YV)
453 CPS::Complex I(0.0, 0.0);
454 for (UInt j = 0; j < mSystem.mNodes.size(); ++j)
455 I += mY.coeff(node_idx, j) * sol_Vcx(j);
456
457 CPS::Complex S = sol_Vcx(node_idx) * conj(I);
458
459 // Generator/source power: S_gen = S_inj + S_load
460 CPS::Complex Sgen = S;
461 for (auto comp : mSystem.mComponentsAtNode[topoNode]) {
462 if (auto loadPtr = std::dynamic_pointer_cast<CPS::SP::Ph1::Load>(comp)) {
463 Sgen += CPS::Complex(**(loadPtr->mActivePowerPerUnit),
464 **(loadPtr->mReactivePowerPerUnit));
465 }
466 }
467
468 // Update connected VD source/generator with actual generated power
469 for (auto comp : mSystem.mComponentsAtNode[topoNode]) {
470 if (auto extnetPtr =
471 std::dynamic_pointer_cast<CPS::SP::Ph1::NetworkInjection>(comp)) {
472 extnetPtr->updatePowerInjection(Sgen * mBaseApparentPower);
473 break;
474 }
475
476 if (auto sgPtr =
477 std::dynamic_pointer_cast<CPS::SP::Ph1::SynchronGenerator>(
478 comp)) {
479 sgPtr->updatePowerInjection(Sgen * mBaseApparentPower);
480 break;
481 }
482 }
483
484 // Store net nodal injection, not generator power
485 sol_P(node_idx) = S.real();
486 sol_Q(node_idx) = S.imag();
487 }
488}
489void PFSolverPowerPolar::calculateQAtPVBuses() {
490 for (auto topoNode : mPVBuses) {
491 auto node_idx = topoNode->matrixNodeIndex();
492
493 // Net nodal injection into the network: S_inj = V * conj(YV)
494 CPS::Complex I(0.0, 0.0);
495 for (UInt j = 0; j < mSystem.mNodes.size(); ++j)
496 I += mY.coeff(node_idx, j) * sol_Vcx(j);
497
498 CPS::Complex S = sol_Vcx(node_idx) * conj(I);
499
500 // Generator power: S_gen = S_inj + S_load
501 CPS::Complex Sgen = S;
502 for (auto comp : mSystem.mComponentsAtNode[topoNode]) {
503 if (auto loadPtr = std::dynamic_pointer_cast<CPS::SP::Ph1::Load>(comp)) {
504 Sgen += CPS::Complex(**(loadPtr->mActivePowerPerUnit),
505 **(loadPtr->mReactivePowerPerUnit));
506 }
507 }
508
509 // Update PV generator with actual generator Q
510 for (auto comp : mSystem.mComponentsAtNode[topoNode]) {
511 if (auto sgPtr =
512 std::dynamic_pointer_cast<CPS::SP::Ph1::SynchronGenerator>(
513 comp)) {
514 sgPtr->updateReactivePowerInjection(Sgen * mBaseApparentPower);
515 break;
516 }
517 }
518
519 // Store net nodal Q injection, not generator Q
520 sol_Q(node_idx) = S.imag();
521 }
522}

524CPS::Real
525PFSolverPowerPolar::loadReactivePowerPerUnit(CPS::TopologicalNode::Ptr node) {
526 CPS::Real q = 0.0;
527 for (auto comp : mSystem.mComponentsAtNode[node])
528 if (auto load = std::dynamic_pointer_cast<CPS::SP::Ph1::Load>(comp))
529 q += load->attributeTyped<CPS::Real>("Q_pu")->get();
530 return q;
531}

533CPS::Real PFSolverPowerPolar::generatorReactivePowerPerUnit(
534 CPS::TopologicalNode::Ptr node) {
535 UInt k = node->matrixNodeIndex();
536 CPS::Complex I(0.0, 0.0);
537 for (UInt j = 0; j < mSystem.mNodes.size(); ++j)
538 I += mY.coeff(k, j) * sol_Vcx(j);
539 // Net nodal injection S = generator - load; add load back for generator Q.
540 CPS::Complex S = sol_Vcx(k) * conj(I);
541 return S.imag() + loadReactivePowerPerUnit(node);
542}

544CPS::Bool PFSolverPowerPolar::enforceReactiveLimits() {
545 // Returns false if the bus has no generator or no finite Q limit.
546 auto busLimits = [&](CPS::TopologicalNode::Ptr node, CPS::Real &qMaxPU,
547 CPS::Real &qMinPU, CPS::Real &vSetPU) -> bool {
548 qMaxPU = 0.0;
549 qMinPU = 0.0;
550 vSetPU = 0.0;
551 bool hasGen = false, anyFinite = false;
552 for (auto comp : mSystem.mComponentsAtNode[node]) {
553 if (auto gen = std::dynamic_pointer_cast<CPS::SP::Ph1::SynchronGenerator>(
554 comp)) {
555 CPS::Real gMax = gen->attributeTyped<CPS::Real>("Q_max_pu")->get();
556 CPS::Real gMin = gen->attributeTyped<CPS::Real>("Q_min_pu")->get();
557 qMaxPU += gMax;
558 qMinPU += gMin;
559 vSetPU = gen->attributeTyped<CPS::Real>("V_set_pu")->get();
560 hasGen = true;
561 // isFinite is bit-pattern based, so it's immune to -Ofast/-ffast-math.
562 if (CPS::Math::isFinite(gMax) || CPS::Math::isFinite(gMin))
563 anyFinite = true;
564 }
565 }
566 return hasGen && anyFinite;
567 };

569 auto frozen = [&](CPS::TopologicalNode::Ptr node) {
570 auto it = mQLimitSwitchCount.find(node);
571 return it != mQLimitSwitchCount.end() &&
572 it->second >= mMaxQLimitSwitchesPerBus;
573 };
574
575 std::vector<std::tuple<CPS::TopologicalNode::Ptr, bool, CPS::Real>> toPQ;
576 std::vector<CPS::TopologicalNode::Ptr> toPV;
577
578 // PV -> PQ: voltage-controlling generators that hit a reactive limit.
579 for (auto node : mPVBuses) {
580 if (frozen(node))
581 continue;
582 CPS::Real qMaxPU, qMinPU, vSetPU;
583 if (!busLimits(node, qMaxPU, qMinPU, vSetPU))
584 continue;
585 CPS::Real qGen = generatorReactivePowerPerUnit(node);
586 if (CPS::Math::isFinite(qMaxPU) && qGen > qMaxPU)
587 toPQ.emplace_back(node, true, qMaxPU);
588 else if (CPS::Math::isFinite(qMinPU) && qGen < qMinPU)
589 toPQ.emplace_back(node, false, qMinPU);
590 }
591
592 // PQ -> PV: pinned generators whose constraint is no longer binding.
593 for (auto &kv : mQLimitConvertedAtMax) {
594 auto node = kv.first;
595 bool atMax = kv.second;
596 if (frozen(node))
597 continue;
598 CPS::Real qMaxPU, qMinPU, vSetPU;
599 if (!busLimits(node, qMaxPU, qMinPU, vSetPU))
600 continue;
601 CPS::Real v = sol_V(node->matrixNodeIndex());
602 if (atMax && v > vSetPU)
603 toPV.push_back(node);
604 else if (!atMax && v < vSetPU)
605 toPV.push_back(node);
606 }
607
608 // Apply PV -> PQ switches (pin reactive injection at the limit).
609 for (auto &c : toPQ) {
610 auto node = std::get<0>(c);
611 bool atMax = std::get<1>(c);
612 CPS::Real qLimPU = std::get<2>(c);
613 mPVBuses.erase(std::remove(mPVBuses.begin(), mPVBuses.end(), node),
614 mPVBuses.end());
615 mPQBuses.push_back(node);
616 UInt idx = node->matrixNodeIndex();
617 Qesp(idx) = qLimPU - loadReactivePowerPerUnit(node);
618 sol_Q(idx) = Qesp(idx);
619 mQLimitConvertedAtMax[node] = atMax;
620 if (++mQLimitSwitchCount[node] >= mMaxQLimitSwitchesPerBus)
621 SPDLOG_LOGGER_WARN(
622 mSLog, "Q-limit: bus {} frozen at {} after {} switches", node->name(),
623 atMax ? "Qmax" : "Qmin", mQLimitSwitchCount[node]);
624 else
625 SPDLOG_LOGGER_INFO(mSLog, "Q-limit: PV bus {} -> PQ pinned at {}",
626 node->name(), atMax ? "Qmax" : "Qmin");
627 }
628
629 // Apply PQ -> PV switches (restore voltage control).
630 for (auto &node : toPV) {
631 mPQBuses.erase(std::remove(mPQBuses.begin(), mPQBuses.end(), node),
632 mPQBuses.end());
633 mPVBuses.push_back(node);
634 CPS::Real qMaxPU, qMinPU, vSetPU;
635 busLimits(node, qMaxPU, qMinPU, vSetPU);
636 sol_V(node->matrixNodeIndex()) = vSetPU;
637 mQLimitConvertedAtMax.erase(node);
638 ++mQLimitSwitchCount[node];
639 SPDLOG_LOGGER_INFO(mSLog, "Q-limit: PQ bus {} -> PV (constraint relaxed)",
640 node->name());
641 }
642
643 return !toPQ.empty() || !toPV.empty();
644}

651void PFSolverPowerPolar::calculatePAndQInjectionPQBuses() {
652 // calculates apparent power injection at PQ buses flowing to other nodes (i.e. S_inj_to_other = S_inj - S_shunt, with S_inj = S_gen - S_load)
653 for (auto topoNode : mPQBuses) {
654 auto node_idx = topoNode->matrixNodeIndex();
655
656 // calculate power flowing out of the node into the admittance matrix (i.e. S_inj)
657 CPS::Complex I(0.0, 0.0);
658 for (UInt j = 0; j < mSystem.mNodes.size(); ++j)
659 I += mY.coeff(node_idx, j) * sol_Vcx(j);
660 CPS::Complex S = sol_Vcx(node_idx) * conj(I);
661
662 // Subtracting shunt power to obtain power injection flowing from this node to the other nodes (i.e. S_inj_to_other)
663 CPS::Real V = sol_V.coeff(node_idx);
664 for (auto comp : mSystem.mComponentsAtNode[topoNode])
665 if (auto shuntPtr = std::dynamic_pointer_cast<CPS::SP::Ph1::Shunt>(comp))
666 // capacitive susceptance is positive --> q is injected into the node
667 S += std::pow(V, 2) * Complex(-**(shuntPtr->mConductancePerUnit),
668 **(shuntPtr->mSusceptancePerUnit));
669
670 // TODO: check whether S_inj_to_other should be stored in sol_P and sol_Q or rather S_inj
671 sol_P(node_idx) = S.real();
672 sol_Q(node_idx) = S.imag();
673 }
674}
