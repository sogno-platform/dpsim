33 Real Reader::unitValue(Real value, CIMPP::UnitMultiplier mult) {
34  switch (mult) {
35  case UnitMultiplier::p:
36  value *= 1e-12;
37  break;
38  case UnitMultiplier::n:
39  value *= 1e-9;
40  break;
41  case UnitMultiplier::micro:
42  value *= 1e-6;
43  break;
44  case UnitMultiplier::m:
45  value *= 1e-3;
46  break;
47  case UnitMultiplier::c:
48  value *= 1e-2;
49  break;
50  case UnitMultiplier::d:
51  value *= 1e-1;
52  break;
53  case UnitMultiplier::k:
54  value *= 1e3;
55  break;
56  case UnitMultiplier::M:
57  value *= 1e6;
58  break;
59  case UnitMultiplier::G:
60  value *= 1e9;
61  break;
62  case UnitMultiplier::T:
63  value *= 1e12;
64  break;
65  default:
66  break;
67  }
68  return value;
69 }

71 TopologicalPowerComp::Ptr Reader::mapComponent(BaseClass* obj) {
72  if (CIMPP::ACLineSegment *line = dynamic_cast<CIMPP::ACLineSegment*>(obj))
73  return mapACLineSegment(line);
74  if (CIMPP::EnergyConsumer *consumer = dynamic_cast<CIMPP::EnergyConsumer*>(obj))
75  return mapEnergyConsumer(consumer);
76  if (CIMPP::PowerTransformer *trans = dynamic_cast<CIMPP::PowerTransformer*>(obj))
77  return mapPowerTransformer(trans);
78  if (CIMPP::SynchronousMachine *syncMachine = dynamic_cast<CIMPP::SynchronousMachine*>(obj))
79  return mapSynchronousMachine(syncMachine);
80  if (CIMPP::ExternalNetworkInjection *extnet = dynamic_cast<CIMPP::ExternalNetworkInjection*>(obj))
81  return mapExternalNetworkInjection(extnet);
82  if (CIMPP::EquivalentShunt *shunt = dynamic_cast<CIMPP::EquivalentShunt*>(obj))
83  return mapEquivalentShunt(shunt);
84  return nullptr;
85 }

97 void Reader::parseFiles() {
98  try {
99  mModel->parseFiles();
100  }
101  catch (...) {
102  mSLog->error("Failed to parse CIM files");
103  return;
104  }
105 
106  mSLog->info("#### List of TopologicalNodes, associated Terminals and Equipment");
107  for (auto obj : mModel->Objects) {
108  if (CIMPP::TopologicalNode* topNode = dynamic_cast<CIMPP::TopologicalNode*>(obj)) {
109  if (mDomain == Domain::EMT)
110  processTopologicalNode<Real>(topNode);
111  else
112  processTopologicalNode<Complex>(topNode);
113  }
114  }
115 
116  // Collect voltage state variables associated to nodes that are used
117  // for various components.
118  mSLog->info("#### List of Node voltages and Terminal power flow data");
119  for (auto obj : mModel->Objects) {
120  // Check if object is of class SvVoltage
121  if (CIMPP::SvVoltage* volt = dynamic_cast<CIMPP::SvVoltage*>(obj)) {
122  processSvVoltage(volt);
123  }
124  // Check if object is of class SvPowerFlow
125  else if (CIMPP::SvPowerFlow* flow = dynamic_cast<CIMPP::SvPowerFlow*>(obj)) {
126  processSvPowerFlow(flow);
127  }
128  }
129 
130  mSLog->info("#### Create other components");
131  for (auto obj : mModel->Objects) {
132 
133  // Check if object is not TopologicalNode, SvVoltage or SvPowerFlow
134  if (!dynamic_cast<CIMPP::TopologicalNode*>(obj)
135  && !dynamic_cast<CIMPP::SvVoltage*>(obj)
136  && !dynamic_cast<CIMPP::SvPowerFlow*>(obj)) {
137 
138  if (CIMPP::IdentifiedObject* idObj = dynamic_cast<CIMPP::IdentifiedObject*>(obj)) {
139 
140  // Check if object is already in equipment list
141  if (mPowerflowEquipment.find(idObj->mRID) == mPowerflowEquipment.end()) {
142  TopologicalPowerComp::Ptr comp = mapComponent(obj);
143  if (comp)
144  mPowerflowEquipment.insert(std::make_pair(comp->uid(), comp));
145  }
146  }
147  }
148  }
149 
150  mSLog->info("#### Check topology for unconnected components");
151  for (auto pfe : mPowerflowEquipment) {
152  auto c = pfe.second;
153 
154  if (mDomain == Domain::EMT) {
155  if(SimPowerComp<Real>::Ptr powercomp = std::dynamic_pointer_cast<SimPowerComp<Real>>(c)) {
156  if (powercomp->terminalNumberConnected() < powercomp->terminalNumber())
157  throw InvalidTopology();
158  }
159  }
160  else {
161  if(SimPowerComp<Complex>::Ptr powercomp = std::dynamic_pointer_cast<SimPowerComp<Complex>>(c)) {
162  if (powercomp->terminalNumberConnected() < powercomp->terminalNumber())
163  throw InvalidTopology();
164  }
165  }
166  }
167 }

191 void Reader::processSvVoltage(CIMPP::SvVoltage* volt) {
192  CIMPP::TopologicalNode* node = volt->TopologicalNode;
193  if (!node) {
194  mSLog->warn("SvVoltage references missing Topological Node, ignoring");
195  return;
196  }
197  auto search = mPowerflowNodes.find(node->mRID);
198  if (search == mPowerflowNodes.end()) {
199  mSLog->warn("SvVoltage references Topological Node {}"
200  " missing from mTopNodes, ignoring", node->mRID);
201  return;
202  }
203 
204  Real voltageAbs = Reader::unitValue(volt->v.value, UnitMultiplier::k);
205 
206  try{
207  mSLog->info(" Angle={}", (float)volt->angle.value);
208  }catch(ReadingUninitializedField* e ){
209  volt->angle.value = 0;
210  std::cerr<< "Uninitialized Angle for SVVoltage at " << volt->TopologicalNode->name << ".Setting default value of " << volt->angle.value << std::endl;
211  }
212  Real voltagePhase = volt->angle.value * PI / 180;
213  mPowerflowNodes[node->mRID]->setInitialVoltage(std::polar<Real>(voltageAbs, voltagePhase));
214 
215  mSLog->info("Node {} MatrixNodeIndex {}: {} V, {} deg",
216  mPowerflowNodes[node->mRID]->uid(),
217  mPowerflowNodes[node->mRID]->matrixNodeIndex(),
218  std::abs(mPowerflowNodes[node->mRID]->initialSingleVoltage()),
219  std::arg(mPowerflowNodes[node->mRID]->initialSingleVoltage())*180/PI
220  );
221 }

236 SystemTopology Reader::systemTopology() {
237  SystemTopology system(mFrequency);
238 
239  for (auto comp : mPowerflowEquipment) {
240  system.addComponent(comp.second);
241  // TODO support Real
242  if (SimPowerComp<Complex>::Ptr powercomp = std::dynamic_pointer_cast<SimPowerComp<Complex>>(comp.second)) {
243  for (auto term : powercomp->topologicalTerminals()) {
244  TopologicalNode::Ptr node=term->topologicalNodes();
245  //TopologicalNode::Ptr node = powercomp->topologicalTerminals().back()->topologicalNodes();
246  if (system.mComponentsAtNode.find(node) == system.mComponentsAtNode.end()) {
247  TopologicalPowerComp::List complist;
248  complist.push_back(powercomp);
249  system.mComponentsAtNode.insert(std::make_pair(node, complist));
250  }
251  else {
252  system.mComponentsAtNode[node].push_back(powercomp);
253  }
254  }
255  }
256  }
257 
258  system.mNodes.resize(mPowerflowNodes.size());
259 
260  for (auto node : mPowerflowNodes) {
261  // The index of the node in the list should not matter anymore
262  //system.mNodes[node.second->matrixNodeIndex()] = node.second;
263  system.addNodeAt(node.second, node.second->matrixNodeIndex());
264  }
265 
266  return system;
267 }

317 TopologicalPowerComp::Ptr Reader::mapACLineSegment(CIMPP::ACLineSegment* line) {
318  mSLog->info(" Found ACLineSegment {} r={} x={} bch={} gch={}", line->name,
319  (float) line->r.value,
320  (float) line->x.value,
321  (float) line->bch.value,
322  (float) line->gch.value);
323 
324  Real resistance = line->r.value;
325  Real inductance = line->x.value / mOmega;
326 
327  // By default there is always a small conductance to ground to
328  // avoid problems with floating nodes.
329  Real capacitance = mShuntCapacitorValue;
330  Real conductance = mShuntConductanceValue;
331 
332  if(line->bch.value > 1e-9 && !mSetShuntCapacitor)
333  capacitance = Real(line->bch.value / mOmega);
334 
335  if(line->gch.value > 1e-9 && !mSetShuntConductance)
336  conductance = Real(line->gch.value);
337 
338  Real baseVoltage = determineBaseVoltageAssociatedWithEquipment(line);
339 
340  if (mDomain == Domain::EMT) {
341  if (mPhase == PhaseType::ABC) {
342  Matrix res_3ph = CPS::Math::singlePhaseParameterToThreePhase(resistance);
343  Matrix ind_3ph = CPS::Math::singlePhaseParameterToThreePhase(inductance);
344  Matrix cap_3ph = CPS::Math::singlePhaseParameterToThreePhase(capacitance);
345  Matrix cond_3ph = CPS::Math::singlePhaseParameterToThreePhase(conductance);
346 
347  auto cpsLine = std::make_shared<EMT::Ph3::PiLine>(line->mRID, line->name, mComponentLogLevel);
348  cpsLine->setParameters(res_3ph, ind_3ph, cap_3ph, cond_3ph);
349  return cpsLine;
350  }
351  else {
352  mSLog->info(" PiLine for EMT not implemented yet");
353  auto cpsLine = std::make_shared<DP::Ph1::PiLine>(line->mRID, line->name, mComponentLogLevel);
354  cpsLine->setParameters(resistance, inductance, capacitance, conductance);
355  return cpsLine;
356  }
357  }
358  else if (mDomain == Domain::SP) {
359  auto cpsLine = std::make_shared<SP::Ph1::PiLine>(line->mRID, line->name, mComponentLogLevel);
360  cpsLine->setParameters(resistance, inductance, capacitance, conductance);
361  cpsLine->setBaseVoltage(baseVoltage);
362  return cpsLine;
363  }
364  else {
365  auto cpsLine = std::make_shared<DP::Ph1::PiLine>(line->mRID, line->name, mComponentLogLevel);
366  cpsLine->setParameters(resistance, inductance, capacitance, conductance);
367  return cpsLine;
368  }

372 TopologicalPowerComp::Ptr Reader::mapPowerTransformer(CIMPP::PowerTransformer* trans) {
373  if (trans->PowerTransformerEnd.size() != 2) {
374  mSLog->warn("PowerTransformer {} does not have exactly two windings, ignoring", trans->name);
375  return nullptr;
376  }
377  mSLog->info("Found PowerTransformer {}", trans->name);
378 
379  // assign transformer ends
380  CIMPP::PowerTransformerEnd* end1 = nullptr, *end2 = nullptr;
381  for (auto end : trans->PowerTransformerEnd) {
382  if (end->Terminal->sequenceNumber == 1) end1 = end;
383  else if (end->Terminal->sequenceNumber == 2) end2 = end;
384  else return nullptr;
385  }
386 
387  // setting default values for non-set resistances and reactances
388  mSLog->info(" PowerTransformerEnd_1 {}", end1->name);
389  mSLog->info(" Srated={} Vrated={}", (float) end1->ratedS.value, (float) end1->ratedU.value);
390  try{
391  mSLog->info(" R={}", (float) end1->r.value);
392  }catch(ReadingUninitializedField* e1){
393  end1->r.value = 1e-12;
394  mSLog->warn(" Uninitialized value for PowerTrafoEnd1 setting default value of R={}", (float) end1->r.value);
395  }
396  try{
397  mSLog->info(" X={}", (float) end1->x.value);
398  }catch(ReadingUninitializedField* e1){
399  end1->x.value = 1e-12;
400  mSLog->warn(" Uninitialized value for PowerTrafoEnd1 setting default value of X={}", (float) end1->x.value);
401  }
402  mSLog->info(" PowerTransformerEnd_2 {}", end2->name);
403  mSLog->info(" Srated={} Vrated={}", (float) end2->ratedS.value, (float) end2->ratedU.value);
404  try{
405  mSLog->info(" R={}", (float) end2->r.value);
406  }catch(ReadingUninitializedField* e1){
407  end2->r.value = 1e-12;
408  mSLog->warn(" Uninitialized value for PowerTrafoEnd2 setting default value of R={}", (float) end2->r.value);
409  }
410  try{
411  mSLog->info(" X={}", (float) end2->x.value);
412  }catch(ReadingUninitializedField* e1){
413  end2->x.value = 1e-12;
414  mSLog->warn(" Uninitialized value for PowerTrafoEnd2 setting default value of X={}", (float) end2->x.value);
415  }
416 
417  if (end1->ratedS.value != end2->ratedS.value) {
418  mSLog->warn(" PowerTransformerEnds of {} come with distinct rated power values. Using rated power of PowerTransformerEnd_1.", trans->name);
419  }
420  Real ratedPower = unitValue(end1->ratedS.value, UnitMultiplier::M);
421  Real voltageNode1 = unitValue(end1->ratedU.value, UnitMultiplier::k);
422  Real voltageNode2 = unitValue(end2->ratedU.value, UnitMultiplier::k);
423 
424  Real ratioAbsNominal = voltageNode1 / voltageNode2;
425  Real ratioAbs = ratioAbsNominal;
426 
427  // use normalStep from RatioTapChanger
428  if (end1->RatioTapChanger) {
429  ratioAbs = voltageNode1 / voltageNode2 * (1 + (end1->RatioTapChanger->normalStep - end1->RatioTapChanger->neutralStep) * end1->RatioTapChanger->stepVoltageIncrement.value / 100);
430  }
431 
432  // if corresponding SvTapStep available, use instead tap position from there
433  if (end1->RatioTapChanger) {
434  for (auto obj : mModel->Objects) {
435  auto tapStep = dynamic_cast<CIMPP::SvTapStep*>(obj);
436  if (tapStep && tapStep->TapChanger == end1->RatioTapChanger) {
437  ratioAbs = voltageNode1 / voltageNode2 * (1 + (tapStep->position - end1->RatioTapChanger->neutralStep) * end1->RatioTapChanger->stepVoltageIncrement.value / 100);
438  }
439  }
440  }
441 
442  // TODO: To be extracted from cim class
443  Real ratioPhase = 0;
444 
445  // Calculate resistance and inductance referred to high voltage side
446  Real resistance = 0;
447  Real inductance = 0;
448  if (voltageNode1 >= voltageNode2 && abs(end1->x.value) > 1e-12) {
449  inductance = end1->x.value / mOmega;
450  resistance = end1->r.value;
451  } else if (voltageNode1 >= voltageNode2 && abs(end2->x.value) > 1e-12) {
452  inductance = end2->x.value / mOmega * std::pow(ratioAbsNominal, 2);
453  resistance = end2->r.value * std::pow(ratioAbsNominal, 2);
454  }
455  else if (voltageNode2 > voltageNode1 && abs(end2->x.value) > 1e-12) {
456  inductance = end2->x.value / mOmega;
457  resistance = end2->r.value;
458  }
459  else if (voltageNode2 > voltageNode1 && abs(end1->x.value) > 1e-12) {
460  inductance = end1->x.value / mOmega / std::pow(ratioAbsNominal, 2);
461  resistance = end1->r.value / std::pow(ratioAbsNominal, 2);
462  }
463 
464  if (mDomain == Domain::EMT) {
465  if (mPhase == PhaseType::ABC) {
466  Matrix resistance_3ph = CPS::Math::singlePhaseParameterToThreePhase(resistance);
467  Matrix inductance_3ph = CPS::Math::singlePhaseParameterToThreePhase(inductance);
468  Bool withResistiveLosses = resistance > 0;
469  auto transformer = std::make_shared<EMT::Ph3::Transformer>(trans->mRID, trans->name, mComponentLogLevel, withResistiveLosses);
470  transformer->setParameters(voltageNode1, voltageNode2, ratioAbs, ratioPhase, resistance_3ph, inductance_3ph);
471  return transformer;
472  }
473  else
474  {
475  mSLog->info(" Transformer for EMT not implemented yet");
476  return nullptr;
477  }
478  }
479  else if (mDomain == Domain::SP) {
480  auto transformer = std::make_shared<SP::Ph1::Transformer>(trans->mRID, trans->name, mComponentLogLevel);
481  transformer->setParameters(voltageNode1, voltageNode2, ratedPower, ratioAbs, ratioPhase, resistance, inductance);
482  Real baseVolt = voltageNode1 >= voltageNode2 ? voltageNode1 : voltageNode2;
483  transformer->setBaseVoltage(baseVolt);
484  return transformer;
485  }
486  else {
487  Bool withResistiveLosses = resistance > 0;
488  auto transformer = std::make_shared<DP::Ph1::Transformer>(trans->mRID, trans->name, mComponentLogLevel, withResistiveLosses);
489  transformer->setParameters(voltageNode1, voltageNode2, ratioAbs, ratioPhase, resistance, inductance);
490  return transformer;
491  }
492 }

494 TopologicalPowerComp::Ptr Reader::mapSynchronousMachine(CIMPP::SynchronousMachine* machine) {
495  mSLog->info(" Found Synchronous machine {}", machine->name);
496 
497  if (mDomain == Domain::DP) {
498  if (mGeneratorType == GeneratorType::TransientStability) {
499  mSLog->info(" GeneratorType is TransientStability.");
500  Real directTransientReactance;
501  Real inertiaCoefficient;
502  Real ratedPower;
503  Real ratedVoltage;
504  for (auto obj : mModel->Objects) {
505  // Check if object is not TopologicalNode, SvVoltage or SvPowerFlow
506  if (CIMPP::SynchronousMachineTimeConstantReactance* genDyn =
507  dynamic_cast<CIMPP::SynchronousMachineTimeConstantReactance*>(obj)) {
508  if (genDyn->SynchronousMachine->mRID == machine->mRID) {
509  directTransientReactance = genDyn->xDirectTrans.value;
510  inertiaCoefficient = genDyn->inertia.value;
511 
512  ratedPower = unitValue(machine->ratedS.value, UnitMultiplier::M);
513 
514  ratedVoltage = unitValue(machine->ratedU.value, UnitMultiplier::k);
515  auto gen = DP::Ph1::SynchronGeneratorTrStab::make(machine->mRID, machine->name, mComponentLogLevel);
516  gen->setStandardParametersPU(ratedPower, ratedVoltage, mFrequency,
517  directTransientReactance, inertiaCoefficient);
518  return gen;
519  }
520  }
521  }
522  mSLog->warn(" Dynamic parameters for {} not found. Not able to instantiate SynchronGeneratorTrStab.", machine->name);
523  } else if (mGeneratorType == GeneratorType::IdealVoltageSource) {
524  mSLog->info(" GeneratorType is IdealVoltageSource.");
525  return std::make_shared<DP::Ph1::SynchronGeneratorIdeal>(machine->mRID, machine->name, mComponentLogLevel);
526  } else if (mGeneratorType == GeneratorType::None) {
527  throw SystemError("GeneratorType is None. Specify!");
528  } else {
529  throw SystemError("GeneratorType setting unfeasible.");
530  }
531  } else if (mDomain == Domain::SP) {
532  if (mGeneratorType == GeneratorType::TransientStability) {
533  mSLog->info(" GeneratorType is TransientStability.");
534  Real directTransientReactance;
535  Real inertiaCoefficient;
536  Real ratedPower;
537  Real ratedVoltage;
538 
539  for (auto obj : mModel->Objects) {
540  // Check if object is not TopologicalNode, SvVoltage or SvPowerFlow
541  if (CIMPP::SynchronousMachineTimeConstantReactance* genDyn =
542  dynamic_cast<CIMPP::SynchronousMachineTimeConstantReactance*>(obj)) {
543  if (genDyn->SynchronousMachine->mRID == machine->mRID) {
544  directTransientReactance = genDyn->xDirectTrans.value;
545  inertiaCoefficient = genDyn->inertia.value;
546 
547  ratedPower = unitValue(machine->ratedS.value, UnitMultiplier::M);
548 
549  ratedVoltage = unitValue(machine->ratedU.value, UnitMultiplier::k);
550  auto gen = SP::Ph1::SynchronGeneratorTrStab::make(machine->mRID, machine->name, mComponentLogLevel);
551  gen->setStandardParametersPU(ratedPower, ratedVoltage, mFrequency,
552  directTransientReactance, inertiaCoefficient);
553  return gen;
554  }
555  }
556  }
557  } else if (mGeneratorType == GeneratorType::PVNode) {
558  mSLog->info(" GeneratorType is PVNode.");
559  for (auto obj : mModel->Objects) {
560  if (CIMPP::GeneratingUnit* genUnit = dynamic_cast<CIMPP::GeneratingUnit*>(obj)) {
561  for (auto syncGen : genUnit->RotatingMachine) {

700 template<typename VarType>
701 void Reader::processTopologicalNode(CIMPP::TopologicalNode* topNode) {
702  // Add this node to global node list and assign simulation node incrementally.
703  int matrixNodeIndex = Int(mPowerflowNodes.size());
704  mPowerflowNodes[topNode->mRID] = SimNode<VarType>::make(topNode->mRID, topNode->name, matrixNodeIndex, mPhase);
705 
706  if (mPhase == PhaseType::ABC) {
707  mSLog->info("TopologicalNode {} phase A as simulation node {} ", topNode->mRID, mPowerflowNodes[topNode->mRID]->matrixNodeIndex(PhaseType::A));
708  mSLog->info("TopologicalNode {} phase B as simulation node {}", topNode->mRID, mPowerflowNodes[topNode->mRID]->matrixNodeIndex(PhaseType::B));
709  mSLog->info("TopologicalNode {} phase C as simulation node {}", topNode->mRID, mPowerflowNodes[topNode->mRID]->matrixNodeIndex(PhaseType::C));
710  }
711  else
712  mSLog->info("TopologicalNode id: {}, name: {} as simulation node {}", topNode->mRID, topNode->name, mPowerflowNodes[topNode->mRID]->matrixNodeIndex());
713 
714  for (auto term : topNode->Terminal) {
715  // Insert Terminal if it does not exist in the map and add reference to node.
716  // This could be optimized because the Terminal is searched twice.
717  auto cpsTerm = SimTerminal<VarType>::make(term->mRID);
718  mPowerflowTerminals.insert(std::make_pair(term->mRID, cpsTerm));
719  cpsTerm->setNode(std::dynamic_pointer_cast<SimNode<VarType>>(mPowerflowNodes[topNode->mRID]));
720 
721  if (!term->sequenceNumber.initialized)
722  term->sequenceNumber = 1;
723 
724  mSLog->info(" Terminal {}, sequenceNumber {}", term->mRID, (int) term->sequenceNumber);
725 
726  // Try to process Equipment connected to Terminal.
727  CIMPP::ConductingEquipment *equipment = term->ConductingEquipment;
728  if (!equipment) {
729  mSLog->warn("Terminal {} has no Equipment, ignoring!", term->mRID);
730  }
731  else {
732  // Insert Equipment if it does not exist in the map and add reference to Terminal.
733  // This could be optimized because the Equipment is searched twice.
734  if (mPowerflowEquipment.find(equipment->mRID) == mPowerflowEquipment.end()) {
735  TopologicalPowerComp::Ptr comp = mapComponent(equipment);
736  if (comp) {
737  mPowerflowEquipment.insert(std::make_pair(equipment->mRID, comp));
738  } else {
739  mSLog->warn("Could not map equipment {}", equipment->mRID);
740  continue;
741  }
742  }
743 
744  auto pfEquipment = mPowerflowEquipment.at(equipment->mRID);
745  std::dynamic_pointer_cast<SimPowerComp<VarType>>(pfEquipment)->setTerminalAt(
746  std::dynamic_pointer_cast<SimTerminal<VarType>>(mPowerflowTerminals[term->mRID]), term->sequenceNumber-1);
747 
748  mSLog->info(" Added Terminal {} to Equipment {}", term->mRID, equipment->mRID);
749  }
750  }
751 }
