13SP::Ph1::AvVoltageSourceInverterDQ::AvVoltageSourceInverterDQ(
14 String uid, String name, Logger::Level logLevel, Bool withTrafo)
15 : CompositePowerComp<Complex>(uid, name, true, true, logLevel),
16 mOmegaN(mAttributes->create<Real>("Omega_nom")),
17 mPref(mAttributes->create<Real>("P_ref")),
18 mQref(mAttributes->create<Real>("Q_ref")),
19 mVcd(mAttributes->create<Real>("Vc_d", 0)),
20 mVcq(mAttributes->create<Real>("Vc_q", 0)),
21 mIrcd(mAttributes->create<Real>("Irc_d", 0)),
22 mIrcq(mAttributes->create<Real>("Irc_q", 0)),
23 mVsref(mAttributes->create<MatrixComp>("Vsref", MatrixComp::Zero(1, 1))),
24 mVs(mAttributes->createDynamic<MatrixComp>("Vs")),
25 mPllOutput(mAttributes->createDynamic<Matrix>("pll_output")),
26 mPowerctrlInputs(mAttributes->createDynamic<Matrix>("powerctrl_inputs")),
27 mPowerctrlOutputs(
28 mAttributes->createDynamic<Matrix>("powerctrl_outputs")),
29 mPowerctrlStates(mAttributes->createDynamic<Matrix>("powerctrl_states")) {
30 if (withTrafo) {
31 setVirtualNodeNumber(4);
32 mConnectionTransformer = SP::Ph1::Transformer::make(
33 **mName + "_trans", **mName + "_trans", mLogLevel, false);
34 addMNASubComponent(mConnectionTransformer,
35 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
36 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
37 } else {
38 setVirtualNodeNumber(3);
39 }
40 mWithConnectionTransformer = withTrafo;
41 setTerminalNumber(1);
42
43 SPDLOG_LOGGER_INFO(mSLog, "Create {} {}", type(), name);
44 **mIntfVoltage = MatrixComp::Zero(1, 1);
45 **mIntfCurrent = MatrixComp::Zero(1, 1);
46
47 // Create electrical sub components
48 mSubResistorF = SP::Ph1::Resistor::make(**mName + "_resF", mLogLevel);
49 mSubResistorC = SP::Ph1::Resistor::make(**mName + "_resC", mLogLevel);
50 mSubCapacitorF = SP::Ph1::Capacitor::make(**mName + "_capF", mLogLevel);
51 mSubInductorF = SP::Ph1::Inductor::make(**mName + "_indF", mLogLevel);
52 mSubCtrledVoltageSource =
53 SP::Ph1::VoltageSource::make(**mName + "_src", mLogLevel);
54 addMNASubComponent(mSubResistorF, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
55 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, false);
56 addMNASubComponent(mSubResistorC, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
57 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, false);
58 addMNASubComponent(mSubCapacitorF, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
59 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
60 addMNASubComponent(mSubInductorF, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
61 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
62
63 // Pre-step of the subcontrolled voltage source is handled explicitly in mnaParentPreStep
64 addMNASubComponent(mSubCtrledVoltageSource, MNA_SUBCOMP_TASK_ORDER::NO_TASK,
65 MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
66
67 SPDLOG_LOGGER_INFO(mSLog, "Electrical subcomponents: ");
68 for (auto subcomp : mSubComponents)
69 SPDLOG_LOGGER_INFO(mSLog, "- {}", subcomp->name());
70
71 // Create control sub components
72 mPLL = Signal::PLL::make(**mName + "_PLL", mLogLevel);
73 mPowerControllerVSI = Signal::PowerControllerVSI::make(
74 **mName + "_PowerControllerVSI", mLogLevel);
75
76 // Sub voltage source
77 mVs->setReference(mSubCtrledVoltageSource->mIntfVoltage);
78
79 // PLL
80 mPLL->mInputRef->setReference(mVcq);
81 mPllOutput->setReference(mPLL->mOutputCurr);
82
83 // Power controller
84 // input references
85 mPowerControllerVSI->mVc_d->setReference(mVcd);
86 mPowerControllerVSI->mVc_q->setReference(mVcq);
87 mPowerControllerVSI->mIrc_d->setReference(mIrcd);
88 mPowerControllerVSI->mIrc_q->setReference(mIrcq);
89 // input, state and output vector for logging
90 mPowerctrlInputs->setReference(mPowerControllerVSI->mInputCurr);
91 mPowerctrlStates->setReference(mPowerControllerVSI->mStateCurr);
92 mPowerctrlOutputs->setReference(mPowerControllerVSI->mOutputCurr);
93}

192void SP::Ph1::AvVoltageSourceInverterDQ::initializeFromNodesAndTerminals(
193 Real frequency) {
194
195 // set initial interface quantities
196 (**mIntfVoltage)(0, 0) = initialSingleVoltage(0);
197 (**mIntfCurrent)(0, 0) =
198 -std::conj(Complex(**mPref, **mQref) / (**mIntfVoltage)(0, 0));
199
200 Complex filterInterfaceInitialVoltage;
201 Complex filterInterfaceInitialCurrent;
202
203 if (mWithConnectionTransformer) {
204 // calculate quantities of low voltage side of transformer (being the interface quantities of the filter)
205 filterInterfaceInitialVoltage =
206 ((**mIntfVoltage)(0, 0) -
207 Complex(mTransformerResistance, mTransformerInductance * **mOmegaN) *
208 (**mIntfCurrent)(0, 0)) /
209 Complex(mTransformerRatioAbs, mTransformerRatioPhase);
210 filterInterfaceInitialCurrent =
211 (**mIntfCurrent)(0, 0) *
212 Complex(mTransformerRatioAbs, mTransformerRatioPhase);
213
214 // connect transformer
215 mVirtualNodes[3]->setInitialVoltage(filterInterfaceInitialVoltage);
216 mConnectionTransformer->connect({mTerminals[0]->node(), mVirtualNodes[3]});
217 } else {
218 // if no transformer used, filter interface equal to inverter interface
219 filterInterfaceInitialVoltage = (**mIntfVoltage)(0, 0);
220 filterInterfaceInitialCurrent = (**mIntfCurrent)(0, 0);
221 }
222
223 // derive initialization quantities of filter
224 Complex vcInit =
225 filterInterfaceInitialVoltage - filterInterfaceInitialCurrent * mRc;
226 Complex icfInit = vcInit * Complex(0., 2. * PI * frequency * mCf);
227 Complex vfInit = vcInit - (filterInterfaceInitialCurrent - icfInit) *
228 Complex(0., 2. * PI * frequency * mLf);
229 Complex vsInit =
230 vfInit - (filterInterfaceInitialCurrent - icfInit) * Complex(mRf, 0);
231 mVirtualNodes[0]->setInitialVoltage(vsInit);
232 mVirtualNodes[1]->setInitialVoltage(vfInit);
233 mVirtualNodes[2]->setInitialVoltage(vcInit);
234
235 // Set parameters electrical subcomponents
236 (**mVsref)(0, 0) = mVirtualNodes[0]->initialSingleVoltage();
237 mSubCtrledVoltageSource->setParameters((**mVsref)(0, 0));
238
239 // Connect electrical subcomponents
240 mSubCtrledVoltageSource->connect({SimNode::GND, mVirtualNodes[0]});
241 mSubResistorF->connect({mVirtualNodes[0], mVirtualNodes[1]});
242 mSubInductorF->connect({mVirtualNodes[1], mVirtualNodes[2]});
243 mSubCapacitorF->connect({mVirtualNodes[2], SimNode::GND});
244 if (mWithConnectionTransformer)
245 mSubResistorC->connect({mVirtualNodes[2], mVirtualNodes[3]});
246 else
247 mSubResistorC->connect({mVirtualNodes[2], mTerminals[0]->node()});
248
249 // Initialize electrical subcomponents
250 for (auto subcomp : mSubComponents) {
251 subcomp->initialize(mFrequencies);
252 subcomp->initializeFromNodesAndTerminals(frequency);
253 }
254
255 // Initialize control subcomponents
256 // current and voltage inputs to PLL and power controller
257 Complex vcdq, ircdq;
258 vcdq = Math::rotatingFrame2to1(
259 mVirtualNodes[3]->initialSingleVoltage(),
260 std::arg(mVirtualNodes[3]->initialSingleVoltage()), 0);
261 ircdq = Math::rotatingFrame2to1(
262 -1. * mSubResistorC->mIntfCurrent->get()(0, 0),
263 std::arg(mVirtualNodes[3]->initialSingleVoltage()), 0);
264 **mVcd = vcdq.real();
265 **mVcq = vcdq.imag();
266 **mIrcd = ircdq.real();
267 **mIrcq = ircdq.imag();
268 // angle input
269 Matrix matrixStateInit = Matrix::Zero(2, 1);
270 matrixStateInit(0, 0) = std::arg(mVirtualNodes[3]->initialSingleVoltage());
271 mPLL->setInitialValues(**mVcq, matrixStateInit, Matrix::Zero(2, 1));
272
273 SPDLOG_LOGGER_INFO(
274 mSLog,
275 "\n--- Initialization from powerflow ---"
276 "\nInterface voltage across: {:s}"
277 "\nInterface current: {:s}"
278 "\nTerminal 0 initial voltage: {:s}"
279 "\nTerminal 0 connected to {:s} = sim node {:d}"
280 "\nVirtual node 0 initial voltage: {:s}"
281 "\nVirtual node 1 initial voltage: {:s}"
282 "\nVirtual node 2 initial voltage: {:s}",
283 Logger::phasorToString((**mIntfVoltage)(0, 0)),
284 Logger::phasorToString((**mIntfCurrent)(0, 0)),
285 Logger::phasorToString(initialSingleVoltage(0)),
286 mTerminals[0]->node()->name(), mTerminals[0]->node()->matrixNodeIndex(),
287 Logger::phasorToString(mVirtualNodes[0]->initialSingleVoltage()),
288 Logger::phasorToString(mVirtualNodes[1]->initialSingleVoltage()),
289 Logger::phasorToString(mVirtualNodes[2]->initialSingleVoltage()));
290 if (mWithConnectionTransformer)
291 SPDLOG_LOGGER_INFO(
292 mSLog, "\nVirtual node 3 initial voltage: {:s}",
293 Logger::phasorToString(mVirtualNodes[3]->initialSingleVoltage()));
294 SPDLOG_LOGGER_INFO(mSLog, "\n--- Initialization from powerflow finished ---");
295}

345void SP::Ph1::AvVoltageSourceInverterDQ::controlStep(Real time,
346 Int timeStepCount) {
347 // Transformation interface forward
348 Complex vcdq, ircdq;
349 vcdq = Math::rotatingFrame2to1(
350 mVirtualNodes[3]->singleVoltage(),
351 mPLL->attributeTyped<Matrix>("output_prev")->get()(0, 0), mThetaN);
352 ircdq = Math::rotatingFrame2to1(
353 -1. * mSubResistorC->attributeTyped<MatrixComp>("i_intf")->get()(0, 0),
354 mPLL->attributeTyped<Matrix>("output_prev")->get()(0, 0), mThetaN);
355 **mVcd = vcdq.real();
356 **mVcq = vcdq.imag();
357 **mIrcd = ircdq.real();
358 **mIrcq = ircdq.imag();
359
360 // add step of subcomponents
361 mPLL->signalStep(time, timeStepCount);
362 mPowerControllerVSI->signalStep(time, timeStepCount);
363
364 // Transformation interface backward
365 (**mVsref)(0, 0) = Math::rotatingFrame2to1(
366 Complex(mPowerControllerVSI->attributeTyped<Matrix>("output_curr")
367 ->get()(0, 0),
368 mPowerControllerVSI->attributeTyped<Matrix>("output_curr")
369 ->get()(1, 0)),
370 mThetaN, mPLL->attributeTyped<Matrix>("output_prev")->get()(0, 0));
371
372 // Update nominal system angle
373 mThetaN = mThetaN + mTimeStep * **mOmegaN;
374}
