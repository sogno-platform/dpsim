82void SP::Ph1::ControlledCurrentSource::mnaCompApplyRightSideVectorStamp(
83 Matrix &rightVector) {
84 (**mIntfCurrent)(0, 0) = **mCurrentRef;
85
86 if (terminalNotGrounded(0))
87 Math::setVectorElement(rightVector, matrixNodeIndex(0),
88 -(**mIntfCurrent)(0, 0));
89 if (terminalNotGrounded(1))
90 Math::setVectorElement(rightVector, matrixNodeIndex(1),
91 (**mIntfCurrent)(0, 0));

108void SP::Ph1::ControlledCurrentSource::mnaCompUpdateVoltage(
109 const Matrix &leftVector) {
110 (**mIntfVoltage)(0, 0) = 0;
111 if (terminalNotGrounded(0))
112 (**mIntfVoltage)(0, 0) =
113 Math::complexFromVectorElement(leftVector, matrixNodeIndex(0));
114 if (terminalNotGrounded(1))
115 (**mIntfVoltage)(0, 0) =
116 (**mIntfVoltage)(0, 0) -
117 Math::complexFromVectorElement(leftVector, matrixNodeIndex(1));
118}
