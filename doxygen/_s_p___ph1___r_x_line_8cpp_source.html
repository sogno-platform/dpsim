157 void SP::Ph1::RXLine::initializeFromNodesAndTerminals(Real frequency) {
158 
159  (**mIntfVoltage)(0, 0) = initialSingleVoltage(1) - initialSingleVoltage(0);
160  Complex impedance = {**mSeriesRes, **mSeriesInd * 2. * PI * frequency};
161  (**mIntfCurrent)(0, 0) = (**mIntfVoltage)(0, 0) / impedance;
162  mVirtualNodes[0]->setInitialVoltage(initialSingleVoltage(0) +
163  (**mIntfCurrent)(0, 0) * **mSeriesRes);
164 
165  // Default model with virtual node in between
166  mSubResistor =
167  std::make_shared<SP::Ph1::Resistor>(**mName + "_res", mLogLevel);
168  mSubResistor->setParameters(**mSeriesRes);
169  mSubResistor->connect({mTerminals[0]->node(), mVirtualNodes[0]});
170  mSubResistor->initializeFromNodesAndTerminals(frequency);
171  addMNASubComponent(mSubResistor, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
172  MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
173 
174  mSubInductor =
175  std::make_shared<SP::Ph1::Inductor>(**mName + "_ind", mLogLevel);
176  mSubInductor->setParameters(**mSeriesInd);
177  mSubInductor->connect({mVirtualNodes[0], mTerminals[1]->node()});
178  mSubInductor->initializeFromNodesAndTerminals(frequency);
179  addMNASubComponent(mSubInductor, MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
180  MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
181 
182  mInitialResistor =
183  std::make_shared<SP::Ph1::Resistor>(**mName + "_snubber_res", mLogLevel);
184  mInitialResistor->setParameters(1e6);
185  mInitialResistor->connect({SimNode::GND, mTerminals[1]->node()});
186  mInitialResistor->initializeFromNodesAndTerminals(frequency);
187  addMNASubComponent(mInitialResistor,
188  MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT,
189  MNA_SUBCOMP_TASK_ORDER::TASK_BEFORE_PARENT, true);
190 
191  SPDLOG_LOGGER_INFO(mSLog,
192  "\n--- Initialization from powerflow ---"
193  "\nVoltage across: {:s}"
194  "\nCurrent: {:s}"
195  "\nTerminal 0 voltage: {:s}"
196  "\nTerminal 1 voltage: {:s}"
197  "\n--- Initialization from powerflow finished ---",
198  Logger::phasorToString((**mIntfVoltage)(0, 0)),
199  Logger::phasorToString((**mIntfCurrent)(0, 0)),
200  Logger::phasorToString(initialSingleVoltage(0)),
201  Logger::phasorToString(initialSingleVoltage(1)));
202 }
