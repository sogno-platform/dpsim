126 void SP::Ph1::ReducedOrderSynchronGeneratorVBR::mnaCompPostStep(
127  const Matrix &leftVector) {
128  // update armature voltage
129  (**mIntfVoltage)(0, 0) =
130  Math::complexFromVectorElement(leftVector, matrixNodeIndex(0));
131 
132  // convert armature voltage into dq reference frame
133  Matrix Vabc = Matrix::Zero(2, 1);
134  Vabc << (**mIntfVoltage)(0, 0).real(), (**mIntfVoltage)(0, 0).imag();
135  **mVdq = mComplexAToDq * Vabc / mBase_V_RMS;
136 
137  // update armature current
138  if (mModelAsNortonSource) {
139  (**mIntfCurrent)(0, 0) =
140  mIvbr -
141  Complex(mConductanceMatrix(0, 0) * (**mIntfVoltage)(0, 0).real() +
142  mConductanceMatrix(0, 1) * (**mIntfVoltage)(0, 0).imag(),
143  mConductanceMatrix(1, 0) * (**mIntfVoltage)(0, 0).real() +
144  mConductanceMatrix(1, 1) * (**mIntfVoltage)(0, 0).imag());
145  } else {
146  (**mIntfCurrent)(0, 0) = Math::complexFromVectorElement(
147  leftVector, mVirtualNodes[1]->matrixNodeIndex());
148  }
149 
150  // convert armature current into dq reference frame
151  Matrix Iabc = Matrix::Zero(2, 1);
152  Iabc << (**mIntfCurrent)(0, 0).real(), (**mIntfCurrent)(0, 0).imag();
153  **mIdq = mComplexAToDq * Iabc / mBase_I_RMS;
154 }
