29void SP::Ph1::ReducedOrderSynchronGeneratorVBR::initializeResistanceMatrix() {
30 // resistance matrix in dq reference frame
31 mResistanceMatrixDq = Matrix::Zero(2, 2);
32 mResistanceMatrixDq << 0.0, mA, mB, 0.0;
33
34 // initialize conductance matrix
35 mConductanceMatrix = Matrix::Zero(2, 2);
36}

38void SP::Ph1::ReducedOrderSynchronGeneratorVBR::calculateResistanceMatrix() {
39 Matrix resistanceMatrix = mDqToComplexA * mResistanceMatrixDq * mComplexAToDq;
40 resistanceMatrix = resistanceMatrix * mBase_Z;
41 mConductanceMatrix = resistanceMatrix.inverse();
42}

75void SP::Ph1::ReducedOrderSynchronGeneratorVBR::mnaCompApplySystemMatrixStamp(
76 SparseMatrixRow &systemMatrix) {
77
78 if (mModelAsNortonSource) {
79 // Stamp conductance matrix
80 // set buttom right block
81 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 0),
82 matrixNodeIndex(0, 0), mConductanceMatrix);
83 } else {
84 // Stamp voltage source
85 Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
86 mVirtualNodes[1]->matrixNodeIndex(), Complex(-1, 0));
87 Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(),
88 mVirtualNodes[0]->matrixNodeIndex(), Complex(1, 0));
89
90 // Stamp conductance matrix
91
92 // set upper left block
93 Math::addToMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
94 mVirtualNodes[0]->matrixNodeIndex(),
95 mConductanceMatrix);
96
97 // set buttom right block
98 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 0),
99 matrixNodeIndex(0, 0), mConductanceMatrix);
100
101 // Set off diagonal blocks
102 Math::addToMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
103 matrixNodeIndex(0, 0), -mConductanceMatrix);
104 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 0),
105 mVirtualNodes[0]->matrixNodeIndex(),
106 -mConductanceMatrix);
107 }
108}

110void SP::Ph1::ReducedOrderSynchronGeneratorVBR::
111 mnaCompApplyRightSideVectorStamp(Matrix &rightVector) {
112 if (mModelAsNortonSource) {
113 // compute equivalent northon circuit in abc reference frame
114 mIvbr = Complex(mConductanceMatrix(0, 0) * mEvbr.real() +
115 mConductanceMatrix(0, 1) * mEvbr.imag(),
116 mConductanceMatrix(1, 0) * mEvbr.real() +
117 mConductanceMatrix(1, 1) * mEvbr.imag());
118
119 Math::setVectorElement(rightVector, matrixNodeIndex(0, 0), mIvbr);
120 } else {
121 Math::setVectorElement(rightVector, mVirtualNodes[1]->matrixNodeIndex(),
122 mEvbr);
123 }
124}

126void SP::Ph1::ReducedOrderSynchronGeneratorVBR::mnaCompPostStep(
127 const Matrix &leftVector) {
128 // update armature voltage
129 (**mIntfVoltage)(0, 0) =
130 Math::complexFromVectorElement(leftVector, matrixNodeIndex(0));
131
132 // convert armature voltage into dq reference frame
133 Matrix Vabc = Matrix::Zero(2, 1);
134 Vabc << (**mIntfVoltage)(0, 0).real(), (**mIntfVoltage)(0, 0).imag();
135 **mVdq = mComplexAToDq * Vabc / mBase_V_RMS;
136
137 // update armature current
138 if (mModelAsNortonSource) {
139 (**mIntfCurrent)(0, 0) =
140 mIvbr -
141 Complex(mConductanceMatrix(0, 0) * (**mIntfVoltage)(0, 0).real() +
142 mConductanceMatrix(0, 1) * (**mIntfVoltage)(0, 0).imag(),
143 mConductanceMatrix(1, 0) * (**mIntfVoltage)(0, 0).real() +
144 mConductanceMatrix(1, 1) * (**mIntfVoltage)(0, 0).imag());
145 } else {
146 (**mIntfCurrent)(0, 0) = Math::complexFromVectorElement(
147 leftVector, mVirtualNodes[1]->matrixNodeIndex());
148 }
149
150 // convert armature current into dq reference frame
151 Matrix Iabc = Matrix::Zero(2, 1);
152 Iabc << (**mIntfCurrent)(0, 0).real(), (**mIntfCurrent)(0, 0).imag();
153 **mIdq = mComplexAToDq * Iabc / mBase_I_RMS;
154}

156Matrix
157SP::Ph1::ReducedOrderSynchronGeneratorVBR::get_DqToComplexATransformMatrix()
158 const {
159 Matrix dqToComplexA(2, 2);
160 dqToComplexA << cos(**mThetaMech - mBase_OmMech * mSimTime),
161 -sin(**mThetaMech - mBase_OmMech * mSimTime),
162 sin(**mThetaMech - mBase_OmMech * mSimTime),
163 cos(**mThetaMech - mBase_OmMech * mSimTime);
164
165 return dqToComplexA;
166}
