46void SP::Ph1::SynchronGenerator3OrderVBR::stepInPerUnit() {
47 if (mSimTime > 0.0) {
48 // calculate Eq_t at t=k
49 (**mEdq_t)(1, 0) = (**mIdq)(0, 0) * mLd_t + (**mVdq)(1, 0);
50 }
51
52 mDqToComplexA = get_DqToComplexATransformMatrix();
53 mComplexAToDq = mDqToComplexA.transpose();
54
55 // calculate resistance matrix at t=k+1
56 calculateResistanceMatrix();
57
58 // VBR history voltage
59 mEh_vbr(0, 0) = 0.0;
60 mEh_vbr(1, 0) = mAq_t * (**mIdq)(0, 0) + mBq_t * (**mEdq_t)(1, 0) +
61 mDq_t * mEf_prev + mDq_t * (**mEf);
62
63 // convert Edq_t into the abc reference frame
64 mEh_vbr = mDqToComplexA * mEh_vbr;
65 mEvbr = Complex(mEh_vbr(0, 0), mEh_vbr(1, 0)) * mBase_V_RMS;
66}
