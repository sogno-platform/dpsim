48void SP::Ph1::SynchronGenerator4OrderVBR::stepInPerUnit() {
49
50 if (mSimTime > 0.0) {
51 // calculate Edq_t at t=k
52 (**mEdq_t)(0, 0) = -(**mIdq)(1, 0) * mLq_t + (**mVdq)(0, 0);
53 (**mEdq_t)(1, 0) = (**mIdq)(0, 0) * mLd_t + (**mVdq)(1, 0);
54 }
55
56 // get transformation matrix
57 mDqToComplexA = get_DqToComplexATransformMatrix();
58 mComplexAToDq = mDqToComplexA.transpose();
59
60 // calculate resistance matrix at t=k+1
61 calculateResistanceMatrix();
62
63 // VBR history voltage
64 mEh_vbr(0, 0) = mAd_t * (**mIdq)(1, 0) + mBd_t * (**mEdq_t)(0, 0);
65 mEh_vbr(1, 0) = mAq_t * (**mIdq)(0, 0) + mBq_t * (**mEdq_t)(1, 0) +
66 mDq_t * mEf_prev + mDq_t * (**mEf);
67
68 // convert Edq_t into the abc reference frame
69 mEh_vbr = mDqToComplexA * mEh_vbr;
70 mEvbr = Complex(mEh_vbr(0, 0), mEh_vbr(1, 0)) * mBase_V_RMS;
71}
