32void SP::Ph1::SynchronGenerator5OrderVBR::specificInitialization() {
33
34 // initial voltage behind the transient reactance in the dq reference frame
35 (**mEdq_t)(0, 0) = 0.0;
36 (**mEdq_t)(1, 0) =
37 (1 - mTaa / mTd0_t) * **mEf - (mLd - mLd_t - mYd) * (**mIdq)(0, 0);
38
39 // initial dq behind the subtransient reactance in the dq reference frame
40 (**mEdq_s)(0, 0) = (**mVdq)(0, 0) - (mLq_s) * (**mIdq)(1, 0);
41 (**mEdq_s)(1, 0) = (**mVdq)(1, 0) + (mLd_s) * (**mIdq)(0, 0);
42
43 // initial history term behind the transient reactance
44 mEh_t(0, 0) = 0.0;
45 mEh_t(1, 0) = mAq_t * (**mIdq)(0, 0) + mBq_t * (**mEdq_t)(1, 0) +
46 mDq_t * (**mEf) + mDq_t * mEf_prev;
47
48 SPDLOG_LOGGER_DEBUG(mSLog,
49 "\n--- Model specific initialization ---"
50 "\nSG model: 5th order"
51 "\nInitial Ed_t (per unit): {:f}"
52 "\nInitial Eq_t (per unit): {:f}"
53 "\nInitial Ed_s (per unit): {:f}"
54 "\nInitial Eq_s (per unit): {:f}"
55 "\n--- Model specific initialization finished ---",
56 (**mEdq_t)(0, 0), (**mEdq_t)(1, 0), (**mEdq_s)(0, 0),
57 (**mEdq_s)(1, 0));
58 mSLog->flush();
59}

61void SP::Ph1::SynchronGenerator5OrderVBR::stepInPerUnit() {
62 // calculate Edq_t at t=k
63 (**mEdq_t)(0, 0) = 0.0;
64 (**mEdq_t)(1, 0) = mAq_t * (**mIdq)(0, 0) + mEh_t(1, 0);
65
66 // calculate Edq_s at t=k
67 (**mEdq_s)(0, 0) = (**mVdq)(0, 0) - mLq_s * (**mIdq)(1, 0);
68 (**mEdq_s)(1, 0) = (**mVdq)(1, 0) + mLd_s * (**mIdq)(0, 0);
69
70 mDqToComplexA = get_DqToComplexATransformMatrix();
71 mComplexAToDq = mDqToComplexA.transpose();
72
73 // calculate resistance matrix at t=k+1
74 calculateResistanceMatrix();
75
76 // calculate history term behind the transient reactance
77 mEh_t(0, 0) = 0.0;
78 mEh_t(1, 0) = mAq_t * (**mIdq)(0, 0) + mBq_t * (**mEdq_t)(1, 0) +
79 mDq_t * (**mEf) + mDq_t * mEf_prev;
80
81 // calculate history term behind the subtransient reactance
82 mEh_s(0, 0) = mAd_s * (**mIdq)(1, 0) + mCd_s * (**mEdq_s)(0, 0);
83 mEh_s(1, 0) = mAq_s * (**mIdq)(0, 0) + mBq_s * (**mEdq_t)(1, 0) +
84 mCq_s * (**mEdq_s)(1, 0) + mDq_s * (**mEf) + mDq_s * mEf_prev;
85
86 // convert Edq_t into the abc reference frame
87 mEh_s = mDqToComplexA * mEh_s;
88 mEvbr = Complex(mEh_s(0, 0), mEh_s(1, 0)) * mBase_V_RMS;
89}
