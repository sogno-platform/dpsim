80 void SP::Ph1::Transformer::initializeFromNodesAndTerminals(Real frequency) {
81  mNominalOmega = 2. * PI * frequency;
82  mReactance = mNominalOmega * mInductance;
83  mSLog->info("Reactance={} [Ohm] (referred to primary side)", mReactance);
84 
85  // Component parameters are referred to high voltage side.
86  // Switch terminals if transformer is connected the other way around.
87  if (Math::abs(mRatio) < 1.) {
88  mRatio = 1. / mRatio;
89  std::shared_ptr<SimTerminal<Complex>> tmp = mTerminals[0];
90  mTerminals[0] = mTerminals[1];
91  mTerminals[1] = tmp;
92  }
93 
94  // Set initial voltage of virtual node in between
95  mVirtualNodes[0]->setInitialVoltage(initialSingleVoltage(1) * mRatio);
96 
97  // Static calculations from load flow data
98  Complex impedance = { mResistance, mReactance };
99  mIntfVoltage(0, 0) = mVirtualNodes[0]->initialSingleVoltage() - initialSingleVoltage(0);
100  mIntfCurrent(0, 0) = mIntfVoltage(0, 0) / impedance;
101 
102  // Create series sub components
103  mSubInductor = std::make_shared<SP::Ph1::Inductor>(mUID + "_ind", mName + "_ind", Logger::Level::off);
104  mSubInductor->setParameters(mInductance);
105 
106  if (mNumVirtualNodes == 3) {
107  mVirtualNodes[2]->setInitialVoltage(initialSingleVoltage(0));
108  mSubResistor = std::make_shared<SP::Ph1::Resistor>(mUID + "_res", mName + "_res", Logger::Level::off);
109  mSubResistor->setParameters(mResistance);
110  mSubResistor->connect({ node(0), mVirtualNodes[2] });
111  mSubResistor->initialize(mFrequencies);
112  mSubResistor->initializeFromNodesAndTerminals(frequency);
113  mSubInductor->connect({ mVirtualNodes[2], mVirtualNodes[0] });
114  } else {
115  mSubInductor->connect({ node(0), mVirtualNodes[0] });
116  }
117  mSubInductor->initialize(mFrequencies);
118  mSubInductor->initializeFromNodesAndTerminals(frequency);
119 
120  // Create parallel sub components
121  // A snubber conductance is added on the low voltage side (resistance scaled with lower voltage side)
122  mSnubberResistance = mNominalVoltageEnd1 > mNominalVoltageEnd2 ? std::abs(mNominalVoltageEnd2)*1e6 : std::abs(mNominalVoltageEnd1)*1e6;
123  mSubSnubResistor = std::make_shared<SP::Ph1::Resistor>(mUID + "_snub_res", mName + "_snub_res", Logger::Level::off);
124  mSubSnubResistor->setParameters(mSnubberResistance);
125  mSubSnubResistor->connect({ node(1), SP::SimNode::GND });
126  mSubSnubResistor->initialize(mFrequencies);
127  mSubSnubResistor->initializeFromNodesAndTerminals(frequency);
128  mSLog->info("Snubber Resistance={} [Ohm] (connected to LV side)", mSnubberResistance);
129 
130  mSLog->info(
131  "\n--- Initialization from powerflow ---"
132  "\nVoltage across: {:s}"
133  "\nCurrent: {:s}"
134  "\nTerminal 0 voltage: {:s}"
135  "\nTerminal 1 voltage: {:s}"
136  "\nVirtual Node 1 voltage: {:s}"
137  "\n--- Initialization from powerflow finished ---",
138  Logger::phasorToString(mIntfVoltage(0, 0)),
139  Logger::phasorToString(mIntfCurrent(0, 0)),
140  Logger::phasorToString(initialSingleVoltage(0)),
141  Logger::phasorToString(initialSingleVoltage(1)),
142  Logger::phasorToString(mVirtualNodes[0]->initialSingleVoltage()));
143 }

154 void SP::Ph1::Transformer::calculatePerUnitParameters(Real baseApparentPower, Real baseOmega) {
155  mSLog->info("#### Calculate Per Unit Parameters for {}", mName);
156  mBaseApparentPower = baseApparentPower;
157  mBaseOmega = baseOmega;
158  mSLog->info("Base Power={} [VA] Base Omega={} [1/s]", baseApparentPower, baseOmega);
159 
160  mBaseImpedance = mBaseVoltage * mBaseVoltage / mBaseApparentPower;
161  mBaseAdmittance = 1.0 / mBaseImpedance;
162  mBaseCurrent = baseApparentPower / (mBaseVoltage*sqrt(3)); // I_base=(S_threephase/3)/(V_line_to_line/sqrt(3))
163  mSLog->info("Base Voltage={} [V] Base Impedance={} [Ohm]", mBaseVoltage, mBaseImpedance);
164 
165  mResistancePerUnit = mResistance / mBaseImpedance;
166  mReactancePerUnit = mReactance / mBaseImpedance;
167  mSLog->info("Resistance={} [pu] Reactance={} [pu]", mResistancePerUnit, mReactancePerUnit);
168 
169  mBaseInductance = mBaseImpedance / mBaseOmega;
170  mInductancePerUnit = mInductance / mBaseInductance;
171  mMagnetizing = mMagnetizingReactance / mBaseImpedance;
172  // omega per unit=1, hence 1.0*mInductancePerUnit.
173  mLeakagePerUnit = Complex(mResistancePerUnit,1.*mInductancePerUnit);
174  mMagnetizingPerUnit = mMagnetizing / mBaseImpedance;
175 
176  mRatioAbsPerUnit = mRatioAbs / mNominalVoltageEnd1 * mNominalVoltageEnd2;
177  mSLog->info("Tap Ratio={} [pu]", mRatioAbsPerUnit);
178 
179  // set snubber resistance
180  if (mBehaviour == Behaviour::Initialization) {
181  Real snubberResistance = 1e3;
182  mSubSnubResistor = std::make_shared<SP::Ph1::Resistor>(mUID + "_snub_res", mName + "_snub_res", mLogLevel);
183  mSubSnubResistor->setParameters(snubberResistance);
184  mSubSnubResistor->connect({ node(1), SP::SimNode::GND });
185  mSubSnubResistor->initializeFromNodesAndTerminals(mBaseOmega);
186  }
187  if (mSubSnubResistor) {
188  mSubSnubResistor->setBaseVoltage(mBaseVoltage);
189  mSubSnubResistor->calculatePerUnitParameters(baseApparentPower);
190  }
191 
192  mSLog->info("Leakage Impedance Per Unit={} [Ohm] ", mLeakagePerUnit);
193 }

195 void SP::Ph1::Transformer::pfApplyAdmittanceMatrixStamp(SparseMatrixCompRow & Y) {
196  // calculate matrix stamp
197  mY_element = MatrixComp(2, 2);
198  Complex y = Complex(1, 0) / mLeakagePerUnit;
199  Complex ys = Complex(1, 0) / mMagnetizingPerUnit;
200  mY_element(0, 0) = (y + ys);
201  mY_element(0, 1) = -y*mRatioAbsPerUnit;
202  mY_element(1, 0) = -y*mRatioAbsPerUnit;
203  mY_element(1, 1) = (y + ys)*std::pow(mRatioAbsPerUnit, 2);
204 
205  //check for inf or nan
206  for (int i = 0; i < 2; i++)
207  for (int j = 0; j < 2; j++)
208  if (std::isinf(mY_element.coeff(i, j).real()) || std::isinf(mY_element.coeff(i, j).imag())) {
209  std::cout << mY_element << std::endl;
210  std::cout << "Zm:" << mMagnetizing << std::endl;
211  std::cout << "Zl:" << mLeakage << std::endl;
212  std::cout << "tap:" << mRatioAbsPerUnit << std::endl;
213  std::stringstream ss;
214  ss << "Transformer>>" << this->name() << ": infinite or nan values in the element Y at: " << i << "," << j;
215  throw std::invalid_argument(ss.str());
216  }
217 
218  //set the circuit matrix values
219  Y.coeffRef(this->matrixNodeIndex(0), this->matrixNodeIndex(0)) += mY_element.coeff(0, 0);
220  Y.coeffRef(this->matrixNodeIndex(0), this->matrixNodeIndex(1)) += mY_element.coeff(0, 1);
221  Y.coeffRef(this->matrixNodeIndex(1), this->matrixNodeIndex(1)) += mY_element.coeff(1, 1);
222  Y.coeffRef(this->matrixNodeIndex(1), this->matrixNodeIndex(0)) += mY_element.coeff(1, 0);
223 
224  mSLog->info("#### Y matrix stamping: {}", mY_element);
225 }

274 void SP::Ph1::Transformer::mnaApplySystemMatrixStamp(Matrix& systemMatrix) {
275  // Ideal transformer equations
276  if (terminalNotGrounded(0)) {
277  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(), mVirtualNodes[1]->matrixNodeIndex(), Complex(-1.0, 0));
278  Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(), mVirtualNodes[0]->matrixNodeIndex(), Complex(1.0, 0));
279  }
280  if (terminalNotGrounded(1)) {
281  Math::setMatrixElement(systemMatrix, matrixNodeIndex(1), mVirtualNodes[1]->matrixNodeIndex(), mRatio);
282  Math::setMatrixElement(systemMatrix, mVirtualNodes[1]->matrixNodeIndex(), matrixNodeIndex(1), -mRatio);
283  }
284 
285  // Add inductive part to system matrix
286  mSubInductor->mnaApplySystemMatrixStamp(systemMatrix);
287  mSubSnubResistor->mnaApplySystemMatrixStamp(systemMatrix);
288 
289  if (mNumVirtualNodes == 3) {
290  mSubResistor->mnaApplySystemMatrixStamp(systemMatrix);
291  }
292 
293  if (terminalNotGrounded(0)) {
294  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(Complex(-1.0, 0)),
295  mVirtualNodes[0]->matrixNodeIndex(), mVirtualNodes[1]->matrixNodeIndex());
296  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(Complex(1.0, 0)),
297  mVirtualNodes[1]->matrixNodeIndex(), mVirtualNodes[0]->matrixNodeIndex());
298  }
299  if (terminalNotGrounded(1)) {
300  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(mRatio),
301  matrixNodeIndex(1), mVirtualNodes[1]->matrixNodeIndex());
302  mSLog->info("Add {:s} to system at ({:d},{:d})", Logger::complexToString(-mRatio),
303  mVirtualNodes[1]->matrixNodeIndex(), matrixNodeIndex(1));
304  }
305 }
