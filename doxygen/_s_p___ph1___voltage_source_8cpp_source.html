68 void SP::Ph1::VoltageSource::initializeFromNodesAndTerminals(Real frequency) {
70  if (**mVoltageRef == Complex(0, 0))
71  **mVoltageRef = initialSingleVoltage(1) - initialSingleVoltage(0);
72 
73  if (mSrcSig == nullptr) {
74  auto srcSigSine = Signal::SineWaveGenerator::make(**mName);
75  srcSigSine->mVoltageRef->setReference(mVoltageRef);
76  srcSigSine->mFreq->setReference(mSrcFreq);
77  srcSigSine->setParameters(**mVoltageRef);
78  mSrcSig = srcSigSine;
79  }
80 
81  SPDLOG_LOGGER_INFO(mSLog,
82  "\n--- Initialization from node voltages ---"
83  "\nVoltage across: {:s}"
84  "\nTerminal 0 voltage: {:s}"
85  "\nTerminal 1 voltage: {:s}"
86  "\n--- Initialization from node voltages ---",
87  Logger::phasorToString(mSrcSig->getSignal()),
88  Logger::phasorToString(initialSingleVoltage(0)),
89  Logger::phasorToString(initialSingleVoltage(1)));

127 void SP::Ph1::VoltageSource::mnaCompApplySystemMatrixStamp(
128  SparseMatrixRow &systemMatrix) {
129  for (UInt freq = 0; freq < mNumFreqs; freq++) {
130  if (terminalNotGrounded(0)) {
131  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
132  matrixNodeIndex(0), Complex(-1, 0), mNumFreqs,
133  freq);
134  Math::setMatrixElement(systemMatrix, matrixNodeIndex(0),
135  mVirtualNodes[0]->matrixNodeIndex(),
136  Complex(-1, 0), mNumFreqs, freq);
137  }
138  if (terminalNotGrounded(1)) {
139  Math::setMatrixElement(systemMatrix, mVirtualNodes[0]->matrixNodeIndex(),
140  matrixNodeIndex(1), Complex(1, 0), mNumFreqs,
141  freq);
142  Math::setMatrixElement(systemMatrix, matrixNodeIndex(1),
143  mVirtualNodes[0]->matrixNodeIndex(), Complex(1, 0),
144  mNumFreqs, freq);
145  }
146 
147  SPDLOG_LOGGER_INFO(mSLog, "-- Stamp frequency {:d} ---", freq);
148  if (terminalNotGrounded(0)) {
149  SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", -1.,
150  matrixNodeIndex(0),
151  mVirtualNodes[0]->matrixNodeIndex());
152  SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", -1.,
153  mVirtualNodes[0]->matrixNodeIndex(),
154  matrixNodeIndex(0));
155  }
156  if (terminalNotGrounded(1)) {
157  SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
158  mVirtualNodes[0]->matrixNodeIndex(),
159  matrixNodeIndex(1));
160  SPDLOG_LOGGER_INFO(mSLog, "Add {:f} to system at ({:d},{:d})", 1.,
161  matrixNodeIndex(1),
162  mVirtualNodes[0]->matrixNodeIndex());
163  }
164  }
165 }
