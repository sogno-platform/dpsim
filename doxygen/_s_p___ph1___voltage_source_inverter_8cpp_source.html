41void SP::Ph1::VoltageSourceInverter::modifyPowerFlowBusType(
42 PowerflowBusType powerflowBusType) {
43
44 mPowerflowBusType = powerflowBusType;
45 switch (powerflowBusType) {
46 case CPS::PowerflowBusType::PV:
47 throw std::invalid_argument(
48 " inverters currently cannot be set as PV bus.");
49 break;
50 case CPS::PowerflowBusType::PQ:
51 mPQ = std::make_shared<CPS::SP::Ph1::PQNode>(**mUID, **mName, mLogLevel);
52 break;
53 case CPS::PowerflowBusType::VD:
54 throw std::invalid_argument(
55 " inverters currently cannot be set as VD bus. ");
56 break;
57 case CPS::PowerflowBusType::None:
58 break;
59 default:
60 throw std::invalid_argument(" Invalid power flow bus type ");
61 break;
62 }
