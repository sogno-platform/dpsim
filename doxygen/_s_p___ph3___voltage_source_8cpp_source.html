41void SP::Ph3::VoltageSource::mnaCompInitialize(
42 Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
43 updateMatrixNodeIndices();
44 (**mIntfVoltage)(0, 0) = **mVoltageRef;
45 (**mIntfVoltage)(1, 0) =
46 Complex(Math::abs(**mVoltageRef) *
47 cos(Math::phase(**mVoltageRef) - 2. / 3. * M_PI),
48 Math::abs(**mVoltageRef) *
49 sin(Math::phase(**mVoltageRef) - 2. / 3. * M_PI));
50 (**mIntfVoltage)(2, 0) =
51 Complex(Math::abs(**mVoltageRef) *
52 cos(Math::phase(**mVoltageRef) + 2. / 3. * M_PI),
53 Math::abs(**mVoltageRef) *
54 sin(Math::phase(**mVoltageRef) + 2. / 3. * M_PI));
55}

57void SP::Ph3::VoltageSource::mnaCompApplySystemMatrixStamp(
58 SparseMatrixRow &systemMatrix) {
59 if (terminalNotGrounded(0)) {
60 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 0),
61 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
62 Complex(-1, 0));
63 Math::addToMatrixElement(systemMatrix,
64 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
65 matrixNodeIndex(0, 0), Complex(-1, 0));
66
67 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 1),
68 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
69 Complex(-1, 0));
70 Math::addToMatrixElement(systemMatrix,
71 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
72 matrixNodeIndex(0, 1), Complex(-1, 0));
73
74 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(0, 2),
75 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
76 Complex(-1, 0));
77 Math::addToMatrixElement(systemMatrix,
78 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
79 matrixNodeIndex(0, 2), -Complex(-1, 0));
80 }
81 if (terminalNotGrounded(1)) {
82 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 0),
83 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
84 Complex(1, 0));
85 Math::addToMatrixElement(systemMatrix,
86 mVirtualNodes[0]->matrixNodeIndex(PhaseType::A),
87 matrixNodeIndex(1, 0), Complex(1, 0));
88
89 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 1),
90 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
91 Complex(1, 0));
92 Math::addToMatrixElement(systemMatrix,
93 mVirtualNodes[0]->matrixNodeIndex(PhaseType::B),
94 matrixNodeIndex(1, 1), Complex(1, 0));
95
96 Math::addToMatrixElement(systemMatrix, matrixNodeIndex(1, 2),
97 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
98 Complex(1, 0));
99 Math::addToMatrixElement(systemMatrix,
100 mVirtualNodes[0]->matrixNodeIndex(PhaseType::C),
101 matrixNodeIndex(1, 2), Complex(1, 0));
102 }
103}
