139 template<typename VarType>
140 void SimPowerComp<VarType>::connect(typename SimNode<VarType>::List nodes) {
141  if (mNumTerminals < nodes.size()) {
142  mSLog->error("Number of Nodes is too large for Component {} - Ignoring", mName);
143  return;
144  }
145  for (UInt i = 0; i < nodes.size(); i++) {
146  String name = mName + "_T" + std::to_string(i);
147  typename SimTerminal<VarType>::Ptr terminal = SimTerminal<VarType>::make(name);
148  terminal->setNode(nodes[i]);
149  setTerminalAt(terminal, i);
150  }
151 }

153 template<typename VarType>
154 void SimPowerComp<VarType>::initialize(Matrix frequencies) {
155  mFrequencies = frequencies;
156  mNumFreqs = static_cast<UInt>(mFrequencies.size());
157 
158  if (mPhaseType != PhaseType::ABC) {
159  mIntfVoltage = MatrixVar<VarType>::Zero(1, mNumFreqs);
160  mIntfCurrent = MatrixVar<VarType>::Zero(1, mNumFreqs);
161  }
162  else {
163  mIntfVoltage = MatrixVar<VarType>::Zero(3, mNumFreqs);
164  mIntfCurrent = MatrixVar<VarType>::Zero(3, mNumFreqs);
165  }
166 
167  for (auto node : mVirtualNodes)
168  node->initialize(frequencies);
169 }
