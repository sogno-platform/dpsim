49 void Python::SystemTopology::updateDicts() {
50  for (auto node : sys->mNodes) {
51  if (!PyObject_HasAttrString(pyNodeDict, node->name().c_str()))
52  addCppNode(node);
53  }
54  for (auto comp : sys->mComponents) {

137 const char *Python::SystemTopology::docAutoDecouple =
138 "auto_decouple(timestep, threshold=1)\n"
139 "Automatically replace suitable transmission lines with decoupling lines in order "
140 "to speed up the simulation.\n"
141 "\n"
142 ":param timestep: Timestep to be used for the simulation.\n"
143 ":param threshold: Maximum ratio of resistance to surge impedance for a decoupling line. "
144 "Passing higher values leads to more lines being considered for decoupling, possibly at the "
145 "cost of simulation accuracy.\n";
146 PyObject* Python::SystemTopology::autoDecouple(SystemTopology* self, PyObject* args)
147 {
148  double timestep, threshold = 1;
149 
150  if (!PyArg_ParseTuple(args, "d|d", &timestep, &threshold))
151  return nullptr;
152 
153  CPS::IdentifiedObject::List newComponents;
154  for (auto it = self->sys->mComponents.begin(); it != self->sys->mComponents.end(); ) {
155  auto line = std::dynamic_pointer_cast<CPS::DP::Ph1::PiLine>(*it);
156  if (line) {
157  using Real = CPS::Real;
158 
159  Real delay = sqrt(line->attribute<Real>("L_series")->get() * line->attribute<Real>("C_parallel")->get());
160  Real surgeImpedance = sqrt(line->attribute<Real>("L_series")->get() / line->attribute<Real>("C_parallel")->get());
161  Real ratio = line->attribute<Real>("R_series")->get() / surgeImpedance;
162 
163  //std::cout << line->name() << " delay " << delay << " ratio " << ratio << std::endl;
164  if (delay >= timestep && ratio < threshold) {
165  auto decoupledLine = CPS::Signal::DecouplingLine::make(line->name(), line->node(0), line->node(1), line->attribute<Real>("R_series")->get(), line->attribute<Real>("L_series")->get(), line->attribute<Real>("C_parallel")->get(), CPS::Logger::Level::off);
166  newComponents.push_back(decoupledLine);
167  for (auto comp : decoupledLine->getLineComponents())
168  newComponents.push_back(comp);
169 
170  PyDict_DelItemString(self->pyComponentDict, line->name().c_str());
171  it = self->sys->mComponents.erase(it);
172  continue;
173  }
174  }
175  ++it;
176  }
177 
178  self->sys->addComponents(newComponents);
179  self->updateDicts();
180  Py_RETURN_NONE;
181 }

208 PyObject* Python::SystemTopology::addDecouplingLine(SystemTopology *self, PyObject *args)
209 {
210  PyObject *pyObj1, *pyObj2;
211  double resistance, inductance, capacitance;
212  int logLevel = 0;
213  const char* name;
214

216  return nullptr;
217 
218  Python::Node<CPS::Complex> *pyNode1, *pyNode2;
219  if (PyObject_TypeCheck(pyObj1, &Python::Node<CPS::Complex>::type)) {
220  pyNode1 = (Python::Node<CPS::Complex>*) pyObj1;
221  }
222  else {
223  PyErr_SetString(PyExc_TypeError, "First argument must be dpsim.Node");
224  return nullptr;
225  }
226 
227  if (PyObject_TypeCheck(pyObj2, &Python::Node<CPS::Complex>::type)) {
228  pyNode2 = (Python::Node<CPS::Complex>*) pyObj2;
229  }
230  else {
231  PyErr_SetString(PyExc_TypeError, "First argument must be dpsim.Node");
232  return nullptr;
233  }
234 
235  auto line = CPS::Signal::DecouplingLine::make(name, pyNode1->node, pyNode2->node, resistance, inductance, capacitance, CPS::Logger::Level(logLevel));
236  self->sys->addComponent(line);
237  self->sys->addComponents(line->getLineComponents());
238  self->updateDicts();
239 
240  Py_RETURN_NONE;

294 int Python::SystemTopology::init(Python::SystemTopology *self, PyObject *args, PyObject *kwds)
295 {
296  static const char *kwlist[] = {"frequency", "nodes", "components", nullptr};
297  double frequency = 50;
298 
299  PyObject *pyCompList = nullptr;
300  PyObject *pyNodeList = nullptr;
301 
302  if (!PyArg_ParseTupleAndKeywords(args, kwds, "d|OO", (char **) kwlist, &frequency, &pyNodeList, &pyCompList)) {
303  return -1;
304  }
305 
306  CPS::IdentifiedObject::List compList;
307  CPS::TopologicalNode::List nodeList;
308 
309  if (pyNodeList) {
310  try {
311  nodeList = nodesFromPython(pyNodeList);
312  } catch (...) {
313  PyErr_Format(PyExc_AttributeError, "Parameter 'nodes' must be a list of dpsim.Node");
314  return -1;
315  }
316 
317  for (int i = 0; i < PyList_Size(pyNodeList); i++) {
318  PyObject *pyObj = PyList_GetItem(pyNodeList, i);
319  PyObject *pyName;
320 
321  if (PyObject_TypeCheck(pyObj, &Python::Node<CPS::Complex>::type)) {
322  Python::Node<CPS::Complex> *pyNode = (Python::Node<CPS::Complex> *) pyObj;
323  pyName = PyUnicode_FromString(pyNode->node->name().c_str());
324  }
325  else if (PyObject_TypeCheck(pyObj, &Python::Node<CPS::Real>::type)) {
326  Python::Node<CPS::Real> *pyNode = (Python::Node<CPS::Real> *) pyObj;
327  pyName = PyUnicode_FromString(pyNode->node->name().c_str());
328  }
329  else {
330  PyErr_Format(PyExc_AttributeError, "Parameter 'nodes' must be a list of dpsim.Node");
331  return -1;
332  }
333 
334  if (PyDict_Contains(self->pyNodeDict, pyName)) {
335  PyErr_Format(PyExc_AttributeError, "Duplicated Node name");
336  return -1;
337  }
338 
339  PyDict_SetItem(self->pyNodeDict, pyName, pyObj);
340  }
341  }
342 
343  if (pyCompList) {
344  try {
345  compList = compsFromPython(pyCompList);
346  } catch (...) {
347  PyErr_Format(PyExc_AttributeError, "Parameter 'components' must be a list of dpsim.Component");
348  return -1;
349  }
350 
351  for (int i = 0; i < PyList_Size(pyCompList); i++) {
352  PyObject *pyObj = PyList_GetItem(pyCompList, i);
353  Python::Component *pyComp = (Python::Component *) pyObj;
354  PyObject *pyName = PyUnicode_FromString(pyComp->comp->name().c_str());
355 
356  if (PyDict_Contains(self->pyComponentDict, pyName)) {
357  PyErr_Format(PyExc_AttributeError, "Duplicated Component name");
358  return -1;
359  }
360 
361  PyDict_SetItem(self->pyComponentDict, pyName, pyObj);
362  }
363  }
364 
365  self->sys = std::make_shared<CPS::SystemTopology>(frequency, nodeList, compList);
366 
367  return 0;
368 }

371 {
372  Py_DECREF(self->pyComponentDict);
373  Py_DECREF(self->pyNodeDict);
374

376  // destructor of its members, we have to manually call the destructor of

381 
382  if (self->sys)
383  self->sys.~SharedSysPtr();
384 
385  Py_TYPE(self)->tp_free((PyObject*) self);
