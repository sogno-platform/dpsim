18 template<typename VarType>
19 void SystemTopology::multiplyPowerComps(Int numberCopies) {
20  typename SimNode<VarType>::List newNodes;
21  typename SimPowerComp<VarType>::List newComponents;
22 
23  for (int copy = 0; copy < numberCopies; copy++) {
24  std::unordered_map<typename SimNode<VarType>::Ptr, typename SimNode<VarType>::Ptr> nodeMap;
25  String copySuffix = "_" + std::to_string(copy+2);
26 
27  // copy nodes
28  typename SimNode<VarType>::Ptr nodePtr;
29  for (size_t nNode = 0; nNode < mNodes.size(); nNode++) {
30  auto nodePtr = this->node<SimNode<VarType>>(static_cast<UInt>(nNode));
31  if (!nodePtr)
32  continue;
33 
34  // GND is not copied
35  if (nodePtr->isGround()) {
36  nodeMap[nodePtr] = nodePtr;
37  } else {
38  auto nodeCpy = SimNode<VarType>::make(nodePtr->name() + copySuffix, nodePtr->phaseType());
39  nodeCpy->setInitialVoltage(nodePtr->initialVoltage());
40  nodeMap[nodePtr] = nodeCpy;
41  newNodes.push_back(nodeCpy);
42  }
43  }
44 
45  // copy components
46  for (auto genComp : mComponents) {
47  auto comp = std::dynamic_pointer_cast<SimPowerComp<VarType>>(genComp);
48  if (!comp)
49  continue;
50  auto copy = comp->clone(comp->name() + copySuffix);
51  if (!copy)
52  throw SystemError("copy() not implemented for " + comp->name());
53 
54  // map the nodes to their new copies, creating new terminals
55  typename SimNode<VarType>::List nodeCopies;
56  for (UInt nNode = 0; nNode < comp->terminalNumber(); nNode++) {
57  nodeCopies.push_back(nodeMap[comp->node(nNode)]);
58  }
59  copy->connect(nodeCopies);
60 
61  // update the terminal powers for powerflow initialization
62  for (UInt nTerminal = 0; nTerminal < comp->terminalNumber(); nTerminal++) {
63  copy->terminal(nTerminal)->setPower(comp->terminal(nTerminal)->power());
64  }
65  newComponents.push_back(copy);
66  }
67  }
68  for (auto node : newNodes)
69  addNode(node);
70  for (auto comp : newComponents)
71  addComponent(comp);
72 }

87 template <typename VarType>
88 void SystemTopology::splitSubnets(std::vector<SystemTopology>& splitSystems) {
89  std::unordered_map<typename SimNode<VarType>::Ptr, int> subnet;
90  int numberSubnets = checkTopologySubnets<VarType>(subnet);
91  if (numberSubnets == 1) {
92  splitSystems.push_back(*this);
93  } else {
94  std::vector<IdentifiedObject::List> components(numberSubnets);
95  std::vector<TopologicalNode::List> nodes(numberSubnets);
96 
97  // Split nodes into subnet groups
98  for (auto node : mNodes) {
99  auto pnode = std::dynamic_pointer_cast<SimNode<VarType>>(node);
100  if (!pnode || node->isGround())
101  continue;
102 
103  nodes[subnet[pnode]].push_back(node);
104  }
105 
106  // Split components into subnet groups
107  for (auto comp : mComponents) {
108  auto pcomp = std::dynamic_pointer_cast<SimPowerComp<VarType>>(comp);
109  if (!pcomp) {
110  // TODO this should only be signal components.
111  // Proper solution would be to pass them to a different "solver"
112  // since they are actually independent of which solver we use
113  // for the electric part.
114  // Just adding them to an arbitrary solver for now has the same effect.
115  components[0].push_back(comp);
116  continue;
117  }
118  for (UInt nodeIdx = 0; nodeIdx < pcomp->terminalNumber(); nodeIdx++) {
119  if (!pcomp->node(nodeIdx)->isGround()) {
120  components[subnet[pcomp->node(nodeIdx)]].push_back(comp);
121  break;
122  }
123  }
124  }
125  for (int currentNet = 0; currentNet < numberSubnets; currentNet++) {
126  splitSystems.emplace_back(mSystemFrequency,
127  nodes[currentNet], components[currentNet]);
128  }
129  }
130 }

132 template <typename VarType>
133 int SystemTopology::checkTopologySubnets(std::unordered_map<typename SimNode<VarType>::Ptr, int>& subnet) {
134  std::unordered_map<typename SimNode<VarType>::Ptr, typename SimNode<VarType>::List> neighbours;
135 
136  for (auto comp : mComponents) {
137  auto pcomp = std::dynamic_pointer_cast<SimPowerComp<VarType>>(comp);
138  if (!pcomp)
139  continue;
140 
141  for (UInt nodeIdx1 = 0; nodeIdx1 < pcomp->terminalNumberConnected(); nodeIdx1++) {
142  for (UInt nodeIdx2 = 0; nodeIdx2 < nodeIdx1; nodeIdx2++) {
143  auto node1 = pcomp->node(nodeIdx1);
144  auto node2 = pcomp->node(nodeIdx2);
145  if (node1->isGround() || node2->isGround())
146  continue;
147 
148  neighbours[node1].push_back(node2);
149  neighbours[node2].push_back(node1);
150  }
151  }
152  }
153 
154  int currentNet = 0;
155  size_t totalNodes = mNodes.size();
156  for (auto tnode : mNodes) {
157  auto node = std::dynamic_pointer_cast<SimNode<VarType>>(tnode);
158  if (!node || tnode->isGround()) {
159  totalNodes--;
160  }
161  }
162 
163  while (subnet.size() != totalNodes) {
164  std::list<typename SimNode<VarType>::Ptr> nextSet;
165 
166  for (auto tnode : mNodes) {
167  auto node = std::dynamic_pointer_cast<SimNode<VarType>>(tnode);
168  if (!node || tnode->isGround())
169  continue;
170 
171  if (subnet.find(node) == subnet.end()) {
172  nextSet.push_back(node);
173  break;
174  }
175  }
176  while (!nextSet.empty()) {
177  auto node = nextSet.front();
178  nextSet.pop_front();
179 
180  subnet[node] = currentNet;
181  for (auto neighbour : neighbours[node]) {
182  if (subnet.find(neighbour) == subnet.end())
183  nextSet.push_back(neighbour);
184  }
185  }
186  currentNet++;
187  }
188  return currentNet;
189 }
