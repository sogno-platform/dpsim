50void TurbineGovernorType1::initialize(Real TmRef) {
51 mTmRef = TmRef;
52 **mXg1 = TmRef;
53 **mXg2 = (1 - mT3 / mTc) * **mXg1;
54 **mXg3 = (1 - mT4 / mT5) * (**mXg2 + mT3 / mTc * **mXg1);
55 **mTm = **mXg3 + mT4 / mT5 * (**mXg2 + mT3 / mTc * **mXg1);
56
57 SPDLOG_LOGGER_INFO(mSLog,
58 "Governor initial values: \n"
59 "\nTorder: {:f}"
60 "\nXg1: {:f}"
61 "\nXg2: {:f}"
62 "\nXg3: {:f}"
63 "\nTm: {:f}",
64 mTmRef, **mXg1, **mXg2, **mXg3, **mTm);
65}

67Real TurbineGovernorType1::step(Real Omega, Real dt) {
68
70 mXg1_prev = **mXg1;
71 mXg2_prev = **mXg2;
72 mXg3_prev = **mXg3;
73
75 Real Tin = mTmRef + (mOmRef - Omega) / mR;
76 if (Tin > mTmax)
77 Tin = mTmax;
78 if (Tin < mTmin)
79 Tin = mTmin;
80
82 **mXg1 = Math::StateSpaceEuler(mXg1_prev, -1 / mTs, 1 / mTs, dt, Tin);
83
85 **mXg2 = Math::StateSpaceEuler(mXg2_prev, -1 / mTc, (1 - mT3 / mTc) / mTc, dt,
86 mXg1_prev);
87
89 **mXg3 = Math::StateSpaceEuler(mXg3_prev, -1 / mT5, (1 - mT4 / mT5) / mT5, dt,
90 (mXg2_prev + mT3 / mTc * mXg1_prev));
91
93 **mTm = **mXg3 + mT4 / mT5 * (**mXg2 + mT3 / mTc * **mXg1);
94
95 return **mTm;
96}
