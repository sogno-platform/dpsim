70void VCO::signalStep(Real time, Int timeStepCount) {
71 **mInputCurr = **mInputRef;
72
73 SPDLOG_LOGGER_DEBUG(
74 mSLog,
75 "Time {}\n: inputCurr = \n{}\n , inputPrev = \n{}\n , statePrev = \n{}",
76 time, **mInputCurr, **mInputPrev, **mStatePrev);
77
78 // calculate new states
79 **mStateCurr = (**mStatePrev + mTimeStep * **mInputCurr);
80 SPDLOG_LOGGER_DEBUG(mSLog, "stateCurr = \n {}", **mStateCurr);
81
82 // calculate new outputs
83 **mOutputCurr = **mStateCurr;
84 SPDLOG_LOGGER_DEBUG(mSLog, "Output values: outputCurr = \n{}", **mOutputCurr);
85}
