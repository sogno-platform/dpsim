45//setter for controller parameters and setting up of system matrices with VCO
46void VoltageControllerVSI::setControllerParameters(Real Kp_voltageCtrl,
47 Real Ki_voltageCtrl,
48 Real Kp_currCtrl,
49 Real Ki_currCtrl,
50 Real Omega_nominal) {
51
52 //Voltage Loop (First)
53 mKiVoltageCtrld = Ki_voltageCtrl;
54 mKiVoltageCtrlq = Ki_voltageCtrl;
55 mKpVoltageCtrld = Kp_voltageCtrl;
56 mKpVoltageCtrlq = Kp_voltageCtrl;
57
58 //Current Loop (Second)
59 mKiCurrCtrld = Ki_currCtrl;
60 mKiCurrCtrlq = Ki_currCtrl;
61 mKpCurrCtrld = Kp_currCtrl;
62 mKpCurrCtrlq = Kp_currCtrl;
63
64 //log loop parameters
65 SPDLOG_LOGGER_INFO(mSLog, "Control Parameters:");
66 SPDLOG_LOGGER_INFO(mSLog, "Voltage Loop: K_i = {}, K_p = {}", Ki_voltageCtrl,
67 Kp_voltageCtrl);
68 SPDLOG_LOGGER_INFO(mSLog, "Current Loop: K_i = {}, K_p = {}", Ki_currCtrl,
69 Kp_currCtrl);
70 SPDLOG_LOGGER_INFO(mSLog, "Cut-Off Frequency = {}", Omega_nominal);
71
72 // Set state space matrices using controller parameters
73
74 // [x] = [phid, phiq, gammad, gammaq]
75 // [u] = [vdref, vqref, vdc, vqc, idc, idq]
76 // [y] = [vdout, vqout]
77
78 mA << 0, 0, 0, 0, 0, 0, 0, 0, mKiVoltageCtrld, 0, 0, 0, 0, mKiVoltageCtrlq, 0,
79 0;
80
81 mB << 1, 0, -1, 0, 0, 0, 0, 1, 0, -1, 0, 0, mKpVoltageCtrld, 0,
82 -mKpVoltageCtrld, 0, -1, 0, 0, mKpVoltageCtrlq, 0, -mKpVoltageCtrlq, 0,
83 -1;
84
85 mC << mKpCurrCtrld * mKiVoltageCtrld, 0, mKiCurrCtrld, 0, 0,
86 mKpCurrCtrlq * mKiVoltageCtrlq, 0, mKiCurrCtrlq;
87
88 mD << mKpCurrCtrld * mKpVoltageCtrld, 0, -mKpCurrCtrld * mKpVoltageCtrld + 1,
89 0, -mKpCurrCtrld, 0, 0, mKpCurrCtrlq * mKpVoltageCtrlq, 0,
90 -mKpCurrCtrlq * mKpVoltageCtrlq + 1, 0, -mKpCurrCtrlq;
91
92 // Log state-space matrices
93 SPDLOG_LOGGER_INFO(mSLog, "State space matrices:");
94 SPDLOG_LOGGER_INFO(mSLog, "A = \n{}", mA);
95 SPDLOG_LOGGER_INFO(mSLog, "B = \n{}", mB);
96 SPDLOG_LOGGER_INFO(mSLog, "C = \n{}", mC);
97 SPDLOG_LOGGER_INFO(mSLog, "D = \n{}", mD);
98}

117//Creating state space model out of the variables
118void VoltageControllerVSI::initializeStateSpaceModel(
119 Real omega, Real timeStep, Attribute<Matrix>::Ptr leftVector) {
120
121 mTimeStep = timeStep;
122
123 // initialization of input --> [u]
124 **mInputCurr << mVdRef, mVqRef, **mVc_d, **mVc_q, **mIrc_d, **mIrc_q;
125 SPDLOG_LOGGER_INFO(mSLog, "Initialization of input: \n" +

127

133 // initialization of output --> [y]

135 SPDLOG_LOGGER_INFO(mSLog, "Initialization of output: \n" +
136 Logger::matrixToString(**mOutputCurr));

167void VoltageControllerVSI::signalStep(Real time, Int timeStepCount) {
168
169 // get current inputs
170 **mInputCurr << mVdRef, mVqRef, **mVc_d, **mVc_q, **mIrc_d, **mIrc_q;
171 SPDLOG_LOGGER_DEBUG(
172 mSLog,
173 "Time {}\n: inputCurr = \n{}\n , inputPrev = \n{}\n , statePrev = \n{}",
174 time, **mInputCurr, **mInputPrev, **mStatePrev);
175
176 // calculate new states
177 **mStateCurr = Math::StateSpaceTrapezoidal(**mStatePrev, mA, mB, mTimeStep,
178 **mInputCurr, **mInputPrev);
179 SPDLOG_LOGGER_DEBUG(mSLog, "stateCurr = \n {}", **mStateCurr);
180
181 // calculate new outputs
182 **mOutputCurr = mC * **mStateCurr + mD * **mInputCurr;
183 SPDLOG_LOGGER_DEBUG(mSLog, "Output values: outputCurr = \n{}", **mOutputCurr);
184}
